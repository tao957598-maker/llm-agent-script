// Check requirement
when('!fn.hasValue(test_no)', {
	// Load default requirement
	assign("test_no", "deposit_and_transfer_funds", "system", "XX")
})

// Use selenium
assign('pythonType', 'selenium', 'explore_mode', '{{fn.optional(explore_mode,1)}}', 'subfolder', '{{test_no}}')
// Send product mode information to RPA
assign('pythonType11','{"explore_mode":{{explore_mode}},"test_system":{{system}},"test_no":{{test_no}}}')
act_rpa.send?command=SwitchExploreMode&name={{fn.json('explore_mode,explore_mode,test_system,system,test_no,test_no')}}
// Load business information from KB
n_envinfo, act_kb.select?name=kb_testcase_biz&column=envinfo&where='{"name":"{{system}}"}'
// Query test steps from KB
n_steps, act_kb.select?name=kb_testcase_step&column=step&where='{"test_no":"{{test_no}}"}'\
	act_func?name=SortPlan&plans={{sys_last_result}}
// Verify test steps size
vrf_flag?value='n_steps?size > 0'

// Execute the test steps
foreach('n_test_step', 'n_steps', {
	// Reset system variables
	act_memory.forgetSystemVariable\
		act_memory.increase?var=count_{{n_test_step.index}}&max=10

	// Prepare to call agent based on product mode
	when('fn.equals(explore_mode, 0)', {
		// Call agent to run test case
		agt_test_step_run_prod?current_step={{n_test_step.item}}&test_no={{test_no}}&system={{system}}
		
		// End flow in advance
		on_failure({
			act_flow.go?plan=n_report
		})
	}, {
	    // agt_test_steps_run_current_step
	    assign('agt_test_steps_run_current_step', '{{n_test_step.item}}')
		// Call agent to build and run test case
		agt_run_test_case?current_step={{n_test_step.item}}&steps={{n_test_step.items}}&disableRead={{disableRead!}}

	})
})

// Update test result to KB
act_kb.merge?name=kb_testcase&columnValues='{"test_result":"Pass"}'&where='{"test_no":"{{test_no}}"}'

// test_notest_notest_no
assign('Generate_test', '{{test_no}}')
// Generate test report

n_report, agt_test_wordreport?test_no={{test_no}}

/////////// Load script information（Generate temporarily if not available） //////////////////////
// Parameters: 1.current_step ==> python tool; 2.steps ==> StepTestInfo
// Agent for building and running scripts
agent('agt_run_test_case', {
	// Prepare to run test step
	n_python_file, act_python.read?requirement={{current_step}}&disableRead={{disableRead}}

	// Check script type
	when("python_code=='jsonl'", {
		// Call new Agent to run subtasks
		agt_test_steps_run?test_no={{sub_test_no}}&system={{system}}&explore_mode={{explore_mode}}
	}, {
		// Check if script is generated
		when("!n_python_file.hasFilename()", {
			// Identify task type 验证任务
			n_gen_code_begin, tpl_get_testcase_type, act_func?name=SubmitFormData
	
			// Prepare to generate script
			when("tpl_get_testcase_type.taskType=='DATA_LOAD_TASK'", {
				// Analyse the requirement and generate subtasks
				agt_test_steps?test_no={{test_no}}_{{n_test_step.index+1}}&steps={{SubmitFormData}}&system={{system}}\
					act_python.write?requirement={{current_step}}&jsonl='{"sub_test_no":"{{test_no}}_{{n_test_step.index+1}}" }'
			}, {
				// Call selenium agent to generate script and run it
				n_python_file2, agt_gen_selenium_code?current_step={{current_step}}&steps={{steps}}&envinfo={{n_envinfo}}
			})
		}, {
			// Run test step
			n_python_result, act_rpa.send?name={{python_file}}&command=RunScript
			
			// Save execution result to KB
			act_kb.merge?name=kb_testcase_step&columnValues={{fn.json('result,n_python_result.screenshot_filename', true)}}&where={{fn.json('test_no,test_no,step,current_step')}}
	
			// Handle error
			on_failure({
				// Waiting 5 seconds
				expression(true, 'act_flow.wait?second=5')
	
				// Run test step again
				n_python_result2, act_rpa.send?name={{python_file}}&command=RunScript
	
				// Try to re-generate script
				act_flow.go?plan=n_gen_code_begin
			}, 2)
		})
	})
})

// Parameters: 1.current_step ==> python tool; 2.steps; 3.n_envinfo ==> build memory variables: step_test_info
// Agent for generating script
agent('agt_gen_selenium_code', {
	// Remove the old script
	act_python.remove?requirement={{current_step}}
    // n_gen_code_begin22n_gen_code_begin22n_gen_code_begin22
    assign('n_gen_code_begin22', '{{n_gen_code_begin}}')
	// Choose sub agent to generate script
	when("n_gen_code_begin22.taskType=='VERIFICATION_TASK'", {
		// Call find agent to verify element  步骤验证??
		agt_test_findelement?test_no={{test_no}}&current_step={{current_step}}&steps={{steps}}&envinfo={{n_envinfo}}
	}, {
		// Call selenium agent to generate code
		agt_test_selenium?test_no={{test_no}}&current_step={{current_step}}&steps={{steps}}&envinfo={{n_envinfo}}&system={{system}}&taskType={{tpl_get_testcase_type.taskType}}
		
		// Check if dialog exists
		when('agt_test_selenium.hasDialog', {
			// 考虑用tpl自动检测：n_test_step.nextItem
			// Add new step
			n_plan_result, act_plan.add?var=n_test_step&plan='点击关闭弹窗的按钮_{{n_test_step.index+1}}'
			
			// Save to KB
			when('n_plan_result.success', {
				// Clean old plan and save new plan to KB
				act_kb.delete?name=kb_testcase_step&where='{"test_no":"{{test_no}}"}'\
					act_kb.convert?data={{n_test_step.items}}&convert=map&storeColumn=step&parameters='{"test_no":"{{test_no}}"}'\
					act_kb.importData?name=kb_testcase_step&data={{sys_last_result}}
			})
		})
		
		// Return result
		assign('result', '{{agt_test_selenium.python_file}}')
	})
	
	on_failure({
		// Collect page information
		n_collect_page, act_rpa.send?command=CollectPageInfo\
			act_func?name=StepTestInfo

		// Load page information of current step
		n_task_page_info, act_kb.select?name=kb_testcase_step_explorer&where='{{fn.json("test_no, test_no, step, current_step")}}'
		
		// Check whether to roll back to the previous task
		tpl_flow_back
		
		// Check whether to abort the task and declare failure
		tpl_flow_fix_or_back, psr_json_python
		
		// Check screenshot
		rag_kb.select?name=kb_testcase_step&column=step&where='{{fn.json("test_no, test_no, step, current_step")}}', tpl_analyse_screenshot
		 
		when('tpl_analyse_screenshot.behavior=="giveup"', {
			// Announce task failure
			act_flow.fail
		}, {
			// Adjust plan
			act_plan.change?var=n_test_step&plan={{tpl_analyse_screenshot.next_step}}&operation={{tpl_analyse_screenshot.behavior}}
		})
	})
})
