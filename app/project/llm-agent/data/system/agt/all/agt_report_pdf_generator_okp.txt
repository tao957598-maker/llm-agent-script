n_clean_json,act_func?script='''
    import com.alibaba.fastjson2.JSON

    // ===== 递归删除指定字段 =====
    def removeFields(obj, keysToRemove){
        if(obj instanceof Map){
            keysToRemove.each { obj.remove(it) }
            obj.each { k, v ->
                removeFields(v, keysToRemove)
            }
        }else if(obj instanceof List){
            obj.each {
                removeFields(it, keysToRemove)
            }
        }
    }

    // ===== 判断是否为空�? =====
    def isEmptyValue
    isEmptyValue = { v ->
        if(v == null){
            return true
        }
        if(v instanceof String){
            return v.trim() == ""
        }
        if(v instanceof List){
            return v.isEmpty()
        }
        if(v instanceof Map){
            return v.isEmpty()
        }
        return false
    }

    // ===== 递归清理空�?�节�? =====
    def pruneEmpty
    pruneEmpty = { node ->
        if(node instanceof Map){
            def keys = new ArrayList(node.keySet())
            for(def k : keys){
                def cleaned = pruneEmpty(node[k])
                if(cleaned == null){
                    node.remove(k)
                }else{
                    node[k] = cleaned
                }
            }

            // 特殊规则：仅包含 value 且为�? �? 删除整个节点
            if(node.size() == 1 && node.containsKey("value") && isEmptyValue(node["value"])){
                return null
            }

            return node.isEmpty() ? null : node
        }

        if(node instanceof List){
            def out = []
            node.each { item ->
                def cleaned = pruneEmpty(item)
                if(cleaned != null){
                    out.add(cleaned)
                }
            }
            return out.isEmpty() ? null : out
        }

        return isEmptyValue(node) ? null : node
    }

    // ===== 初始�? JSON =====
    def json = n_report_obj
    if(json == null){
        json = [:]
    }else{
        // 深拷贝，避免修改原对�?
        json = JSON.parse(JSON.toJSONString(json))
    }

    // ===== 要删除的字段 =====
    def keysToRemove = ["aiRationale", "knowledgeBaseUsed", "dataSource"]

    // ===== 执行清理 =====
    removeFields(json, keysToRemove)
    json = pruneEmpty(json)

    if(json == null){
        json = [:]
    }

    // ===== 返回结果 =====
    return json
'''
