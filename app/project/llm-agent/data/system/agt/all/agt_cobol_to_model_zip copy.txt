// ==============================================
// 文件参数校验（zip_file 和 zip_file_url 必须传一个）
// ==============================================

when('{{!fn.hasValue(zip_file) && !fn.hasValue(zip_file_url)}}', {
	log('错误：zip_file 和 zip_file_url 必须传入其中一个')
	act_flow.fail?reason='参数错误'&message='请上传文件（zip_file）或提供文件URL（zip_file_url），两者必须传入一个'
})
log('task_id：{{task_id}}')
// ==============================================
// 文件参数处理
// ==============================================

when('{{fn.hasValue(zip_file)}}', {
	// 从知识库查询文件信息
	n_file_name, act_kb.selectBy?sql='select knowledge_name, store_uri from kb_spaces_folder_knowledge where id="{{zip_file}}"'
	assign("file_name", '{{n_file_name[0].knowledge_name}}')
	assign("file_path", '{{n_file_name[0].store_uri}}')
	log('成功获取文件信息：{{file_name}}')
	log('文件路径：{{file_path}}')
	// 调用COBOL转模型API（传zip_file）
	n_result, act_api.doExecuteDynamic?name="COBOL2Model_keyword4"&dynamicParamData='{"option":"{{option}}","bti_task_id":"{{bit_task_id}}","file":"{{file_path}}","name_space":"{{name_space}}","mode_name":"{{mode_name}}","language":"{{language}}"}'
})


when('{{fn.hasValue(zip_file_url)}}', {
	// 调用COBOL转模型API（传zip_file_url）
	log('文件URL：{{zip_file_url}}')
	n_result, act_api.doExecuteDynamic?name="COBOL2Model_keyword4"&dynamicParamData='{"option":"{{option}}","bti_task_id":"{{bit_task_id}}","zip_file_url":"{{zip_file_url}}","name_space":"{{name_space}}","mode_name":"{{mode_name}}","language":"{{language}}"}'
})


// 读取分析结果页模板（与 vue2_front/public/analysis-results.html 一致；占位 __RESULT_PLACEHOLDER__ 供下方注入 API 结果）
n_file_name, act_kb.selectBy?sql='select knowledge_name, store_uri from kb_spaces_folder_knowledge where file_name="analysis-results.html"'
assign("file_path", '{{n_file_name[0].store_uri}}')
n_analysis_html_tpl, act_func.read?filename='{{file_path}}'&version=1
// 使用 act_func.run 执行 Groovy：将 n_result 序列化为 JSON，替换模板中的占位符，得到可渲染的完整 HTML
n_html_result, act_func.run?script='''
    import groovy.json.JsonOutput

    def tpl = fn.get("n_analysis_html_tpl")
    def originalContent = fn.get("n_result")
    def payload
    if (originalContent instanceof CharSequence) {
        def s = originalContent.toString().trim()
        payload = (s.startsWith("[") || s.startsWith("{")) ? s : JsonOutput.toJson(originalContent)
    } else {
        payload = JsonOutput.toJson(originalContent)
    }
    // 只替换 renderAnalysisResult 的参数表达式，避免误替换 const 内字符串导致 JSON 引号打断 JS
    def needle = "RESULT_PLACEHOLDER==='__RESULT_PLACEHOLDER__'?[]:RESULT_PLACEHOLDER"
    return tpl.contains(needle) ? tpl.replace(needle, payload) : tpl
'''

n_timestamp, act_func.run?script='''
    // 获取当前时间，格式为 "yyyy-MM-dd HH:mm:ss"
    def now = fn.now()
    // 替换掉空格、冒号和短横线，得到 "yyyyMMdd_HHmmss" 格式
    def formattedTimestamp = now.replaceAll("[-: ]", "").replace(" ", "_")
    // 将结果返回给 agent 变量
    return formattedTimestamp
'''
assign('file_name', 'generated_{{bit_task_id}}_{{n_timestamp}}.html')

// 使用 act_func.write 将内容写入文件，filename 可以自定义
n_file_info, act_func.write?filename='{{file_name}}'&data='{{n_html_result}}'
assign('kbSpacePath', 'BTI/cobol/result')
// 2. 使用知识空间的上传工具，将文件上传到指定路径
// 工具 code: kbs_upload_kbs_path
// 参数 file: 上一步写入的文件信息
// 参数 kbsPath: 用户指定的知识空间目录路径
n_upload_result, act_api.doExecute?name='kbs_upload_kbs_path'&params='{{fn.json("file, n_file_info, kbsPath, kbSpacePath")}}'

//$ final answer
assign('final_result', '文件已上传到知识空间路径: {{kbSpacePath}}，上传结果: {{n_upload_result}}')

//$ final answer  
assign('final_html_output', '{{n_html_result}}')
