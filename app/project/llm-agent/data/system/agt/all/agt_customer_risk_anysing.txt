n_kb_query_final,act_func?script='''
    def provided = fn.optional(n_kb_query, "")
    if(provided != null && String.valueOf(provided).trim() != ""){
        return String.valueOf(provided).trim()
    }

    def customerType = fn.optional(n_customer.customerType, fn.optional(n_customer.partyType, "Unknown"))
    def lineOfBusiness = fn.optional(n_customer.lineOfBusiness, "Unknown")
    def kycRisk = fn.optional(n_kyc.riskLevel, "Unknown")

    def screeningSummary = ""
    if(n_screen instanceof Map){
        screeningSummary = fn.optional(n_screen.recommendation, fn.optional(n_screen.screeningScope, ""))
    }

    def gapSummary = ""
    if(n_gap instanceof Map){
        gapSummary = fn.optional(n_gap.recommendation, fn.optional(n_gap.summary, ""))
    }

    def txnSummary = ""
    if(n_txn instanceof Map){
        txnSummary = fn.optional(n_txn.rationale, fn.optional(n_txn.summary, ""))
    }

    return "AML / KYC / customer risk rating rules, review frequency, recommendation action matrix, enhanced due diligence requirements, STR escalation criteria. " +
           "CustomerType=" + customerType +
           ", LineOfBusiness=" + lineOfBusiness +
           ", KycRisk=" + kycRisk +
           ", Screening=" + screeningSummary +
           ", Gap=" + gapSummary +
           ", Transaction=" + txnSummary
'''
log('n_kb_query_final={{n_kb_query_final}}')

// ---------- Step 4: normalize KB result ----------
n_kb_result,act_func?script='''
    def api = [:]
    def data = [:]

    try{
        api = fn.map(n_kb_api_result)
    }catch(Exception e){
        api = [:]
    }

    try{
        data = fn.map(api.data)
    }catch(Exception e){
        data = [:]
    }

    def items = []
    if(data.results instanceof List){
        items = data.results
    }else if(data.items instanceof List){
        items = data.items
    }else if(data.records instanceof List){
        items = data.records
    }else if(api.results instanceof List){
        items = api.results
    }else{
        items = []
    }

    def normalized = []
    items.each{ it ->
        def row = it instanceof Map ? it : [:]
        normalized.add([
            title      : fn.optional(row.title, fn.optional(row.name, "")),
            content    : fn.optional(row.content, fn.optional(row.text, fn.optional(row.snippet, ""))),
            source     : fn.optional(row.source, fn.optional(row.documentName, fn.optional(row.fileName, ""))),
            kbName     : fn.optional(row.knowledgeBase, fn.optional(row.kbName, "")),
            score      : fn.optional(row.score, ""),
            referenceId: fn.optional(row.referenceId, fn.optional(row.id, ""))
        ])
    }

    return normalized.take(10)
'''
log('n_kb_result={{n_kb_result}}')
