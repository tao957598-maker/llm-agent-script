assign('n_agent_name', 'agt_Main_Ongoing_Review_Agent')
assign('n_scene_name', 'Main Ongoing Review Agent')

// =========================
// 1. 初始化变量
// =========================
assign('n_result', '{}')
assign('n_error', '')
assign('n_has_error', 'false')
assign('n_final_status', '')
assign('n_need_approval', 'false')

assign('n_customer_result', '{}')
assign('n_transaction_result', '{}')
assign('n_company_result', '{}')
assign('n_name_screening_result', '{}')
assign('n_gap_analysis_result', '{}')
assign('n_edd_result', '{}')
assign('n_risk_result', '{}')
assign('n_recommendation_result', '{}')
assign('n_status_result', '{}')

// =========================
// 2. 参数标准化
// =========================
when('{{!fn.hasValue(customerId)}}', {
	assign('customerId', '')
})
when('{{!fn.hasValue(customerType)}}', {
	assign('customerType', '')
})
when('{{!fn.hasValue(prOrTe)}}', {
	assign('prOrTe', '')
})
when('{{!fn.hasValue(teReason)}}', {
	assign('teReason', '')
})
when('{{!fn.hasValue(caseId)}}', {
	assign('caseId', '')
})

assign('customerType', '{{String.valueOf(customerType).trim().toUpperCase()}}')
assign('prOrTe', '{{String.valueOf(prOrTe).trim().toUpperCase()}}')

// =========================
// 3. 参数校验
// =========================
when('{{!fn.hasValue(customerId)}}', {
	assign('n_has_error', 'true')
	assign('n_error', 'customerId不能为空')
})

when('{{!n_has_error && !fn.hasValue(customerType)}}', {
	assign('n_has_error', 'true')
	assign('n_error', 'customerType不能为空')
})

when('{{!n_has_error && !fn.hasValue(prOrTe)}}', {
	assign('n_has_error', 'true')
	assign('n_error', 'prOrTe不能为空')
})

when('{{!n_has_error && !fn.hasValue(teReason)}}', {
	assign('n_has_error', 'true')
	assign('n_error', 'teReason不能为空')
})

when('{{!n_has_error && !fn.hasValue(caseId)}}', {
	assign('n_has_error', 'true')
	assign('n_error', 'caseId不能为空')
})

when('{{!n_has_error && !(customerType=="INDIVIDUAL" || customerType=="NON_INDIVIDUAL")}}', {
	assign('n_has_error', 'true')
	assign('n_error', 'customerType只允许INDIVIDUAL或NON_INDIVIDUAL')
})

when('{{!n_has_error && !(prOrTe=="PR" || prOrTe=="TE")}}', {
	assign('n_has_error', 'true')
	assign('n_error', 'prOrTe只允许PR或TE')
})

// =========================
// 4. 参数校验失败直接返回
// =========================
when('{{n_has_error=="true"}}', {
    log('参数校验失败: {{n_error}}')
    n_result,act_func.run?script='''
        def result = [
            success     : false,
            agentName   : 'agt_Main_Ongoing_Review_Agent',
            caseId      : caseId,
            customerId  : customerId,
            errorMessage: n_error
        ]
        return fn.json(result)
    '''
})

// =========================
// 5. 启动日志
// =========================
log('开始执行 Main Ongoing Review Agent')
log('caseId={{caseId}}, customerId={{customerId}}, customerType={{customerType}}, prOrTe={{prOrTe}}')

// =========================
// 6. Step 1 更新状态：Analysing Customer Information
// =========================
log('Step1: update status -> Analysing Customer Information')
n_status_result,agt_update_case_status?caseId="{{caseId}}"&status="Analysing Customer Information"&stage="CUSTOMER_PROFILE"

// =========================
// 7. Customer Profile Agent
// =========================
log('调用 Customer Profile Agent')
n_customer_result,agt_customer_profile_agent?customerId="{{customerId}}"&customerType="{{customerType}}"&caseId="{{caseId}}"&prOrTe="{{prOrTe}}"&teReason="{{teReason}}"

// =========================
// 8. Transaction Analysis Agent
// =========================
log('调用 Transaction Analysis Agent')
n_transaction_result,agt_transaction_analysis_agent?customerId="{{customerId}}"&customerType="{{customerType}}"&caseId="{{caseId}}"&customerAgentResult="{{n_customer_result}}"

// =========================
// 9. Company Search Agent（仅非个人客户）
// =========================
when('{{customerType=="NON_INDIVIDUAL"}}', {
	log('客户类型为NON_INDIVIDUAL，调用 Company Search Agent')
	n_company_result,agt_company_search_agent?customerId="{{customerId}}"&customerType="{{customerType}}"&caseId="{{caseId}}"&customerAgentResult="{{n_customer_result}}"
})

when('{{customerType!="NON_INDIVIDUAL"}}', {
	log('客户类型不是NON_INDIVIDUAL，跳过 Company Search Agent')
	assign('n_company_result', '{}')
})

// =========================
// 10. Step 2 更新状态：Name Screening
// =========================
log('Step2: update status -> Name Screening')
n_status_result,agt_update_case_status?caseId="{{caseId}}"&status="Name Screening"&stage="NAME_SCREENING"

// =========================
// 11. Name Screening Agent
// =========================
log('调用 Name Screening Agent')
n_name_screening_result,agt_name_screening_agent?customerId="{{customerId}}"&customerType="{{customerType}}"&caseId="{{caseId}}"&customerAgentResult="{{n_customer_result}}"&companySearchResult="{{n_company_result}}"

// =========================
// 12. Step 3 更新状态：Customer Gap Analysing
// =========================
log('Step3: update status -> Customer Gap Analysing')
n_status_result,agt_update_case_status?caseId="{{caseId}}"&status="Customer Gap Analysing"&stage="GAP_ANALYSIS"

// =========================
// 13. Gap Analysis Agent
// =========================
log('调用 Gap Analysis Agent')
n_gap_analysis_result,agt_gap_analysis_agent?customerId="{{customerId}}"&customerType="{{customerType}}"&caseId="{{caseId}}"&customerAgentResult="{{n_customer_result}}"&transactionReport="{{n_transaction_result}}"&companySearchResult="{{n_company_result}}"&nameScreeningReport="{{n_name_screening_result}}"

// =========================
// 14. Step 4 更新状态：EDD Form Filling
// =========================
log('Step4: update status -> EDD Form Filling')
n_status_result,agt_update_case_status?caseId="{{caseId}}"&status="EDD Form Filling"&stage="EDD_FORM"

// =========================
// 15. CAR / EDD Agent
// =========================
log('调用 CAR / EDD Agent')
n_edd_result,agt_car_edd_agent?customerId="{{customerId}}"&customerType="{{customerType}}"&caseId="{{caseId}}"&customerAgentResult="{{n_customer_result}}"&transactionReport="{{n_transaction_result}}"&companySearchResult="{{n_company_result}}"&nameScreeningReport="{{n_name_screening_result}}"&gapAnalysisReport="{{n_gap_analysis_result}}"&prOrTe="{{prOrTe}}"&teReason="{{teReason}}"

// =========================
// 16. Step 5 更新状态：Customer Risk Analysing
// =========================
log('Step5: update status -> Customer Risk Analysing')
n_status_result,agt_update_case_status?caseId="{{caseId}}"&status="Customer Risk Analysing"&stage="RISK_ASSESSMENT"

// =========================
// 17. Risk Assessment Agent
// =========================
log('调用 Risk Assessment Agent')
n_risk_result,agt_risk_assessment_agent?customerId="{{customerId}}"&customerType="{{customerType}}"&caseId="{{caseId}}"&customerAgentResult="{{n_customer_result}}"&transactionReport="{{n_transaction_result}}"&companySearchResult="{{n_company_result}}"&nameScreeningReport="{{n_name_screening_result}}"&gapAnalysisReport="{{n_gap_analysis_result}}"&eddFormJson="{{n_edd_result}}"

// =========================
// 18. Recommendation Agent
// =========================
log('调用 Recommendation Agent')
n_recommendation_result,agt_recommendation_agent?customerId="{{customerId}}"&customerType="{{customerType}}"&caseId="{{caseId}}"&riskAssessmentResult="{{n_risk_result}}"&customerAgentResult="{{n_customer_result}}"&transactionReport="{{n_transaction_result}}"&gapAnalysisReport="{{n_gap_analysis_result}}"&eddFormJson="{{n_edd_result}}"

// =========================
// 19. 根据风险等级决定最终状态
// =========================
// 这里假设 n_risk_result 中会返回 riskLevel 字段
assign('n_final_status', 'Completed')
assign('n_need_approval', 'false')

when('{{fn.hasValue(n_risk_result)}}', {
	n_risk_level,act_func.run?script='''
		def rs = [:]
		try{
			rs = fn.map(n_risk_result)
		}catch(Exception e){
			rs = [:]
		}
		return String.valueOf(rs.riskLevel ?: rs.output?.riskLevel ?: '').toUpperCase()
	'''
})

when('{{n_risk_level=="MEDIUM" || n_risk_level=="HIGH"}}', {
	assign('n_final_status', 'Pending for Approval')
	assign('n_need_approval', 'true')
})

when('{{n_risk_level=="NORMAL"}}', {
	assign('n_final_status', 'Completed')
	assign('n_need_approval', 'false')
})

// =========================
// 20. Step 6 更新最终状态
// =========================
log('Step6: update final status -> {{n_final_status}}')
n_status_result,agt_update_case_status?caseId="{{caseId}}"&status="{{n_final_status}}"&stage="FINAL"

// =========================
// 21. 汇总输出
// =========================
n_result,act_func.run?script='''
	def result = [:]
	def safeMap = { obj ->
		try{
			if(obj instanceof Map){
				return obj
			}
			if(obj instanceof String){
				def str = String.valueOf(obj).trim()
				if(str.startsWith("{")){
					return fn.map(str)
				}
			}
		}catch(Exception e){}
		return [:]
	}

	result.success = true
	result.agentName = 'agt_Main_Ongoing_Review_Agent'
	result.caseId = caseId
	result.customerId = customerId
	result.customerType = customerType
	result.prOrTe = prOrTe
	result.teReason = teReason
	result.finalStatus = n_final_status
	result.needApproval = n_need_approval

	result.customerProfileResult = safeMap(n_customer_result)
	result.transactionAnalysisResult = safeMap(n_transaction_result)
	result.companySearchResult = safeMap(n_company_result)
	result.nameScreeningResult = safeMap(n_name_screening_result)
	result.gapAnalysisResult = safeMap(n_gap_analysis_result)
	result.eddResult = safeMap(n_edd_result)
	result.riskAssessmentResult = safeMap(n_risk_result)
	result.recommendationResult = safeMap(n_recommendation_result)

	return fn.json(result)
'''

log('Main Ongoing Review Agent执行完成')
