        }

        def k = fn.optional(h.keywordEn, "")
        if(k != ""){
            def curr = keywordMap.containsKey(k) ? keywordMap.get(k) : 0
            keywordMap.put(k, curr + 1)
        }
    }

    def topKeywords = []
    keywordMap.each{ k, v ->
        topKeywords.add([keyword: k, count: v])
    }
    topKeywords.sort{ a, b -> b.count <=> a.count }

    return [
        rawHitCount          : rawHits.size(),
        relevantNegativeCount: finalHits.size(),
        involvedPartyCount   : partyMap.size(),
