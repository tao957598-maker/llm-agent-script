	function hasDocumentedResponseBody(meta, code) {
		const response = ((meta || {}).responses || {})[String(code)];
		if (!response || typeof response !== 'object') {
			return false;
		}
		const description = String(response.description || '');
		if (description === 'Documented HTTP ' + code + ' response' && Number(code) >= 300) {
			return false;
		}

			return true;
		}
		if (response.schema || response.examples || response.example) {
			return true;
		}
		return false;
	}

	function getPrimaryStatusCode(meta) {
		const responses = (meta || {}).responses || {};
		const codes = (meta.explicit_status_codes || meta.documented_status_codes || []).map(code => String(code)).filter(code => code);
		const codeWithBody = codes.find(code => hasDocumentedResponseBody(meta, code));
		return codeWithBody || codes[0] || Object.keys(responses)[0] || '';

	function hasAuthEvidence(item, meta) {
		if (Boolean((meta || {}).requires_auth) || Boolean((meta || {}).has_security_requirement)) {
			return true;
		}
		const parameter = (item || {}).parameter || {};

		const allHeaderNames = headerNames.concat(documentedHeaders);
		const authHeaderKeywords = ['authorization', 'token', 'x-access-token', 'access-token', 'api-key', 'x-api-key'];
		return allHeaderNames.some(name => authHeaderKeywords.includes(name));
	}

	function isDigit(ch) {
		return ch >= '0' && ch <= '9';
	}

	function collectStatusCodes(text) {
		const value = String(text || '');
		const codes = [];
		for (let index = 0; index <= value.length - 3; index++) {
			const prev = index > 0 ? value[index - 1] : '';
			const code = value.substring(index, index + 3);
			const next = index + 3 < value.length ? value[index + 3] : '';
			if (
				code[0] >= '1' && code[0] <= '5' &&
				isDigit(code[1]) && isDigit(code[2]) &&
				!isDigit(prev) && !isDigit(next)
			) {
				codes.push(code);
			}
		}
		return codes;
	}

	function extractVerifyStatusCodes(verify) {
		const text = String(verify || '');
		const codes = [];
		const newline = String.fromCharCode(10);
		let index = 0;
		while ((index = text.indexOf('response.status_code', index)) >= 0) {
			const lineEnd = text.indexOf(newline, index);
			const line = text.substring(index, lineEnd >= 0 ? lineEnd : text.length);
			collectStatusCodes(line).forEach(code => codes.push(code));
			index = index + 'response.status_code'.length;
		}
		return Array.from(new Set(codes));
	}

	function alignVerifyWithDocumentedResponseBody(item, meta) {
		const nextItem = deepClone(item);
		if (!nextItem.verify) {
			return nextItem;
		}
		const statusCodes = extractVerifyStatusCodes(nextItem.verify);
		if (statusCodes.length === 0) {
			return nextItem;
		}

		if (statusCodes.length === 1 && statusCodes[0] === '401' && !hasAuthEvidence(nextItem, meta) && primaryStatus) {
			nextItem.verify = `def verify_response(response):\n    assert response.status_code == ${primaryStatus}`;
			return nextItem;
		}
		const shouldUseStatusOnly = statusCodes.every(code => !hasDocumentedResponseBody(meta, code));
		if (!shouldUseStatusOnly) {
			return nextItem;
		}
		nextItem.verify = statusCodes.length === 1
			? `def verify_response(response):\n    assert response.status_code == ${statusCodes[0]}`
			: `def verify_response(response):\n    assert response.status_code in [${statusCodes.join(', ')}]`;
		return nextItem;
	}

## 字段生成规则
1. 方法对应字段：

     请根据下方 OpenAPI 接口定义，将每个 key 自动归类到正确的请求位置：
     · key 匹配 OpenAPI path 参数 → 放入 parameter.path
     · key 匹配 OpenAPI query 参数 → 放入 parameter.params
     · key 匹配 OpenAPI requestBody 属性 → 放入 parameter.data 或 parameter.json
     · key 匹配 HTTP 头 → 放入 parameter.headers
   - GET/DELETE：只在确实需要覆盖时输出 `path`、`headers`、`params`
   - POST/PUT：只在确实需要覆盖时输出 `path`、`headers`、`params`、`data`、`json`
   - 如果某字段与 `fixed_parameter` 的默认值相同，则不要在 `parameter` 中重复输出
   - 空的 `headers`、`params`、`data`、`json`、`path` 不要输出
2. 路径参数规则：
   - `fixed_parameter` 中的各 key 已经是默认请求参数；正常场景若直接复用默认值，不要在 `parameter` 中重复输出
   - 只有当场景需要覆盖动态路由参数时，才显式输出 `parameter.path`
   - 对缺失必填路径参数的场景，在 JSON 中将该路径参数值写为 `null`，表示请求层的 `None`
3. 测试数据要求：
   - 使用有意义的测试数据
   - 包含边界值测试
   - 符合字段类型约束
4. verify 字段是唯一允许使用 Python 语法的字段，其内容不得影响其他字段生成风格。
