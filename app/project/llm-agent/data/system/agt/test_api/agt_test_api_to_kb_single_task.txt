            const result = [];

                    singleEndpoint.paths = {
                        [path]: {
                            [method]: operation
                        }
                    };

                    const refs = findSchemaRefs(operation);
                    for (const ref of refs) {
                        const schemaName = ref.split('/').pop();
                        if (doc.components.schemas[schemaName]) {
                            singleEndpoint.components.schemas[schemaName] = doc.components.schemas[schemaName];
                        }
                    }

                }

                if (key === '$ref') {
                    refs.add(value);
                } else if (typeof value === 'object') {
                    findSchemaRefs(value, refs);
                }
            }

define('tpl_extract_common_fixed_parameter', '''
你是一个 API 文档结构化提取器。请读取下面的 Markdown 内容，只输出合法 JSON，不要输出解释、注释或 Markdown 代码块。

═══════════════════════════════════════
第一部分：固定参数提取 (fixed_parameter)
═══════════════════════════════════════

从文档中提取所有明确出现的固定参数，输出为一个**扁平 key-value 对象**（不要嵌套子对象）。

提取规则：
1. 文档中出现的所有参数 key-value 对都必须提取，一个不漏
2. key 统一转为英文 camelCase：
   - "用户名"       → "userName"
   - "密码"         → "password"
   - "租户ID"       → "tenantId"
   - "订单编号"     → "orderNo"
   - "认证令牌"     → "Authorization"（如果是 token 类，映射为 HTTP 标准头名）
   - 原始已是英文的保持原样，如 "pageSize"、"Content-Type"
3. 文档中有真实值的原样保留，没有值的不要凭空编造
4. 同一参数多处出现时，优先取 Fixed Parameter 区块中的值
5. **不要**将参数分类到 headers/params/data/json 等子对象中，就是一个扁平对象：
   ✅ {"Authorization": "Bearer xxx", "userName": "zhangsan", "pageSize": 10}
   ❌ {"headers": {"Authorization": "Bearer xxx"}, "params": {"pageSize": 10}}

═══════════════════════════════════════
第二部分：测试场景提取 (test_scenarios)
═══════════════════════════════════════

如果文档里存在测试场景数据，逐条提取：
- title      : 场景标题
- parameter  : 场景参数覆盖项
- verify     : 校验代码（原样保留）
如果文档没有测试场景，输出空数组 []

═══════════════════════════════════════
输出格式（严格 JSON，字符串内换行用 \\n）
═══════════════════════════════════════

{
  "name": "接口路径，如 /api/auth/login",
  "method": "HTTP 方法，如 POST",
  "description": "接口描述",
  "protocol": "协议，如 HTTP",
  "fixed_parameter": {
    "Authorization": "Bearer xxx",
    "userName": "zhangsan",
    "pageSize": 10
  },
  "route_parameter": {},
  "test_scenarios": [
    {
      "title": "场景标题",
      "parameter": {},
      "verify": "校验代码"
    }
  ]
}

规则：
- fixed_parameter 是一个扁平对象，不要嵌套 headers/params/data/json 等子对象
- 空字段用 "" 或 {} 或 []，不要省略任何字段
- 只输出 JSON，不要任何额外内容

Markdown 内容如下：
{{n_file_content}}
''')
