// Set search parameters
assign('params','{"question": "{{userInput}}", "folderIds": "{{kbSpaceId!}}", "knowledgeIds": "{{kbSpaceFileId!}}"}')

// assign('params','{"question": "{{userInput}}", "folderIds": "{{kbSpaceId!}}", "knowledgeIds": "273764152924340224"}')

//## Searching user file content text
n_file_context, rag_kbs.searchFileContent?paramData='{{params}}'

when('fn.hasValue(n_file_context)', {
	n_file_context_str, act_func?script='''
		return n_file_context.join("\\r\\n\\r\\n")
	'''
})

assign('n_report_path','')

// 查询知识信息
when("fn.hasValue(kbSpaceFileId)", {
	n_knowledge_data,act_kb.selectBy?sql='select CONCAT(full_path , "/Test Report") report_path from kb_spaces_folder where id in (select kb_spaces_folder_id from kb_spaces_folder_knowledge where id in ({{kbSpaceFileId}}));'
	assign('n_report_path','{{n_knowledge_data[0].report_path}}')
})

//## Parameters auto extraction (JSON format，For job use)
n_tpl_params_res, tpl_autotest_param_extract, psr_json

// 添加kbSpaceFileId 参数
n_autotest_params, act_func?script='''
        def data_map = fn.map(n_tpl_params_res)
        data_map.put("kbSpaceFileId", params.knowledgeIds)
		if(null != n_report_path && !"".equals(n_report_path)){
			data_map.put("n_report_path", n_report_path)
		}
        return data_map
'''

//# 正在打开 AUTO_TEST_VIEW 页面…
n_auto_test_view, act_flow.page?pageCode='native.AUTO_TEST_VIEW'&params={{n_autotest_params}}&isNative=true

when("fn.hasValue(n_auto_test_view.confirmed.test_type) && n_auto_test_view.confirmed.test_type == 'WEB'", {
	// 调用WEB执行模块
	agt_auto_test_execution_web.chatdoc.auto_test?userInput='{{n_auto_test_view.confirmed}}'&kbSpaceFileId='{{kbSpaceFileId}}'&n_plan_tasks='{{n_plan_tasks}}'
},{

})
