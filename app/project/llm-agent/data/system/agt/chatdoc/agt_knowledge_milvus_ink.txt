// 1.更新知识的状态todo->IN_PROGRESS
foreach('n_loop_id', 'knowledgeIdArr', {
    act_kb.merge?name='kb_spaces_folder_knowledge'&columnValues='{"status":"3"}'&where='{"id":{{n_loop_id.item}}}'
})

foreach('n_loop_id', 'knowledgeIdArr', {
    assign('knowledgeId', '{{n_loop_id.item}}')

    // 2.创建/更新知识库表与milvus
    // act_kb.defineKB?tableInfo={{tableInfo}}&embedding='emb_question'

    // 3.查询出需要处理的知识
    n_data,act_kb.select?name='kb_spaces_folder_knowledge'&column='id,kb_spaces_folder_id,extension,store_uri,knowledge_name'&maxSize=1&where='{"id":{{n_loop_id.item}}}'
    n_first_data,act_func?script='''
                            return n_data[0]
                        '''
    // 4.通过文件类型，获取对应的api后缀
    n_data_apiPostfix,act_kbs.getApiPostfix?extension='{{n_first_data.extension}}'

    when("n_data_apiPostfix!=null && n_data_apiPostfix !=''", {
        // 5.获取api编码
        n_api_code,act_func?script='''
                            def api_code = 'ink_upload_api'
                            api_code += "_" + n_data_apiPostfix
                            return api_code
                        '''

        // 6.调用Ink院的接口，将知识进行分片
        n_api_data,act_api.doExecute?name={{n_api_code}}&params='{"table_name":'kb_spaces_knowledge_ink_section',"document_id":'{{knowledgeId}}',"schema":'default',"folder_id":'{{n_first_data.kb_spaces_folder_id}}',"token":'{{token}}'}'&fileNames='{{n_first_data.store_uri}}'
        when("n_api_data.success==true", {
            // 7.将参数信息添加到指定的map中
            n_api_data,act_func?script='''
                    n_api_data.put("knowledge_name", n_first_data.knowledge_name)
                    n_api_data.put("storeUri", n_first_data.store_uri)
                    return n_api_data
                '''

            // 8.将转换后的内容，存储到指定的文件目录
            n_save_parse_file,act_kbs.writeParseFile?paramData='{{n_api_data}}'&subFolder='/kbs_parse'

            // 9.body相关的参数信息
            n_detail_data,act_func?script='''
                            def map = new HashMap()
                            map.put("transformed_file", n_save_parse_file.fileUri)
                            return map
                        '''

            // 10.将转换之后的文件，保存到detail里面
            act_kb.merge?name='kb_spaces_folder_knowledge'&columnValues='{{n_detail_data}}'&where='{"id":{{n_loop_id.item}}}'
        },{
            // 7.更新知识状态->fail
            act_kb.merge?name='kb_spaces_folder_knowledge'&columnValues='{"status":"6","msg":'{{n_api_data.msg}}'}'&where='{"id":{{n_loop_id.item}}}'
        })
    },{
          // 6.更新知识状态->fail
         act_kb.merge?name='kb_spaces_folder_knowledge'&columnValues='{"status":"6","msg":"fail,the file can't chunk."}'&where='{"id":{{n_loop_id.item}}}'
     })
})

assign('return', '{{sys_last_result}}')
