//## 正在分析用户输入选择工具
tpl_choose_tool, psr_json

// 判断选择的工具是否需要参数确认
n_if_need_confirm, act_func?script='''
	import com.alibaba.fastjson2.JSON
	import com.alibaba.fastjson2.JSONObject

	def paramJson = fn.toolDetail(tpl_choose_tool.type, tpl_choose_tool.name)

	// 使用 Fastjson 解析 JSON
	try {
		JSONObject parsedJson = JSON.parseObject(paramJson);
		JSONObject properties = parsedJson.getJSONObject("inputSchema").getJSONObject("properties");
		properties.containsKey("userInput");
		boolean hasOnlyUserInput = properties.size() == 1 && properties.containsKey("userInput");
		return hasOnlyUserInput;
	} catch (Exception e) {
		return false;
	}
'''

// 判断是否需要参数确认
when('tpl_choose_tool.type == "AGENT" && n_if_need_confirm', {
	// 执行agent
	n_tool_exec, act_plan.run?executor={{tpl_choose_tool.name}}&parameters='userInput={{userInput}}'

	// 执行结果
	assign('resultTip', '')
}, {
	//# 正在提取参数
	n_tool_param, tpl_parameter_extract, psr_json

	//# 进行人工确认运行参数
	n_param_confirm, act_flow.pageConfirm?toolType={{tpl_choose_tool.type}}&toolName={{tpl_choose_tool.name}}&params={{n_tool_param.params}}

	// 判断是否非确认按钮
	when('n_param_confirm.apiCode!="agent_manual_confirm"', {
		// 执行UI点击API
		act_plan.run?toolType=API&code={{n_param_confirm.apiCode}}&parameters={{n_param_confirm}}
	})

	// 根据人工确认后的参数运行工具
	when('n_param_confirm.type == "AGENT" && n_param_confirm.name == "agt_test_steps.test"', {
		// 构建JOB参数
		assign('jobData','{"agentName":"{{n_param_confirm.name}}","jobDesc":"ChatDocJob","notifyStrategy":"全部通知","executorParam":{{n_param_confirm.confirmedParams}}}')

		// 创建自动化测试agent JOB
		n_tool_exec, act_job.createAndTrigger?data={{jobData}}&isOneOffTask=true&isOnlyAdd=true

		//# Result Tip
		assign('resultTip', '任务已创建，任务名： {{n_tool_exec.jobDesc}}，点击<a href="/TaskManagement?id={{n_tool_exec.id}}" target="_blank">{{n_tool_exec.id}}</a>可查看任务')
	}, {
		//# 工具执行处理
		n_tool_exec, act_plan.run?toolType={{n_param_confirm.type}}&code={{n_param_confirm.name}}&parameters={{n_param_confirm.confirmedParams}}

		//# 执行结果
		assign('resultTip', '{{n_tool_exec}}')
	})	
})


define('tpl_choose_tool', '''
- Role: 熟悉智能办公系统的管家
- Background: 需要根据用户的输入，精确的定位使用哪种工具，符合用户的需求。
- Requirements: 
1. 结果以Json格式返回选择的工具名和工具所属分类；
2. json需要包含score分值，范围0-10，值越高则代表匹配度越高；
3. 如果工具列表都不符合用户的需求，则name为空，score的值为0；

# 返回示例：
{
	"type": "AGENT",
	"name": "agt_test_xxx.test",
	"score": 5
}

# 用户输入：
{{userInput}}

# 工具集列表：
{{fn.tools('toolset_multi_scenario')}}

''')

1. 工具参数规则按照properties描述填充；
2. 必填项需要有值，默认值一般为缺省的值；
3. 遇到日期业务的字段，格式一般都为yyyy-MM-dd或者yyyy-MM-dd HH:mm:ss
4. 遇到测试编号可以按照业务生成一个测试编号+序号，如测试登录案例，则测试编号为test_logincase_001;


# 返回示例：

		"envinfo": "系统地址：http://xxx.com",

	}
