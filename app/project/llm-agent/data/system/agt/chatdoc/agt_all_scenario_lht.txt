// 进行人工确认运行参数
//# Confirm parameters manually
n_param_confirm, act_flow.page?pageCode=sys_tool_param_confirm&toolType={{n_tool_confirm.confirmed.type}}&toolCode={{n_tool_confirm.confirmed.name}}&params={{n_tool_param.params}}

// 根据人工确认后的参数运行工具
when('n_param_confirm.confirmed.type == "AGENT" && ["agt_test_steps.test", "agt_test_doc_summary.test_req_poc"]?seq_contains(n_param_confirm.confirmed.name)', {
	// 构建JOB参数
	assign('jobData','{"agentName":"{{n_param_confirm.confirmed.name}}","jobDesc":"ChatDocJob","notifyStrategy":"全部通知","executorParam":{{n_param_confirm.confirmed}}}')

	// 创建自动化测试agent JOB
	n_tool_exec, act_job.createAndTrigger?data={{jobData}}&isOneOffTask=true&isOnlyAdd=true

	//# Result Tip
	assign('resultTip', '任务已创建，任务名： {{n_tool_exec.jobDesc}}，点击<a href="/chat-conversation?isFullScreen=true&bizId={{n_tool_exec.bizId}}" target="_blank">{{n_tool_exec.bizId}}</a>可查看任务')
}, {
	// 工具执行处理
	//## Tool execution
	n_tool_exec, act_plan.exec?toolType={{n_param_confirm.confirmed.type}}&code={{n_param_confirm.confirmed.name}}&parameters={{n_param_confirm.confirmed}}

	// 执行结果
	assign('resultTip', '')
})
