//## 获取编码规约
n_kb_projectrules, act_kb.selectBy?sql='''
	SELECT page, content from kbs_doc_page_parse where documentId in ( 
		SELECT id FROM kb_spaces_folder_knowledge t2 WHERE t2.kb_spaces_folder_id IN (
			SELECT id FROM kb_spaces_folder t1 WHERE t1.full_path = '/OA'
		)
	)
'''

//## 构建查询变量
assign('params','{"question":"{{userInput}}","folderPath":"/OA"}')

//## 查询项目需求
n_project_requiement, rag_kbs.searchQuestion?paramData='{{params}}'&similarity='0.5'

//## 过滤结果
