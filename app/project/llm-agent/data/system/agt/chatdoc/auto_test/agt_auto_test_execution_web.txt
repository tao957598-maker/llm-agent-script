			act_memory.assign?var=n_plan_summary&value={{n_plan_summary}}&type=plan_update

				def rows = n_all_test_steps_data
				if (rows == null || !(rows instanceof List)) rows = []
				return rows.collect { r ->
					def out = new LinkedHashMap()
					out["Test No"] = r.test_no == null ? "" : r.test_no
					out["System Name"] = r.system_name == null ? "" : r.system_name
					out["Title"] = r.title == null ? "" : r.title
					out["Test Step"] = r.steps == null ? "" : r.steps
					out["Data"] = r.data == null ? "" : r.data
					out["Expected Result"] = r.expected_result == null ? "" : r.expected_result
					out["Version"] = r.version == null ? "" : r.version
					return out
				}
			'''

			n_test_data_file_msg,act_artifact.write?filepath='{{n_export_paths.testDataPath}}'&data='{{n_test_data_export}}'
			n_test_data_sync_msg,act_artifact.triggerSync?filepath='{{n_export_paths.testDataPath}}'&forceSync=true&ensureKrd=true

		})

    log('Step {{current_plan_step}} completed with result: {{fn.get("step_result_" + current_plan_step)}}')
