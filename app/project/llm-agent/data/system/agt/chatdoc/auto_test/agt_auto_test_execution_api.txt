		})

							def n_tmp_map = new HashMap()
							n_tmp_map.put("enable", true);
							n_tmp_map.put("name", n_userInput.test_name);
							n_tmp_map.put("testType", "2");

            //### 执行plan
            n_plan_virtual_open,act_memory.assign?var=n_plan_virtual&value="{{n_api_result_data.data.name}}"&type=open_virtual&planName='n_plan_task'&index={{n_plan_task.index}}

			act_memory.assign?var=n_plan_summary&value={{n_plan_summary}}&type=plan_update&planName='n_plan_task'&index={{n_plan_task.index}}
		    assign("n_plan_virtual_value", "{{n_plan_summary}}")

		})

		on_failure({
			assign('n_plan_summary','执行plan失败');
			//## {{n_plan_task.index + 1}} / {{n_plan_tasks.size}} 执行测试(并发调度/自动编排)
			act_memory.assign?var=n_plan_summary&value={{n_plan_summary}}&type=plan_update
			assign("n_plan_virtual_value", "{{n_plan_summary}}")
			act_flow.fail?result='{{n_api_result_data.message}}'
		})
	},{
		assign('n_plan_summary','执行完成:创建Test Plan失败');
		//## {{n_plan_task.index + 1}} / {{n_plan_tasks.size}} 执行测试(并发调度/自动编排)
		act_memory.assign?var=n_plan_summary&value={{n_plan_summary}}&type=plan_update&planName='n_plan_task'&index={{n_plan_task.index}}
		assign("n_plan_virtual_value", "{{n_plan_summary}}")
	})

	//### 创建并执行plan 完成
    n_plan_virtual_close,act_memory.assign?var=n_plan_virtual&value="{{n_plan_virtual_value!}}"&type=close_virtual&planName='n_plan_task'&index={{n_plan_task.index}}

	assign('step_result_{{current_plan_step}}', '{{n_plan_summary}}')
