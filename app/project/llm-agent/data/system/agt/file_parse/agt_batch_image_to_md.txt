// 文件
when('!fn.hasValue(fileId)', {
	assign("fileId", "247386787614785536")
})

//# documentId
assign("documentId", "{{fileId}}")

// kbs_doc_page_parse取数据
n_kb_data,act_kb.selectBy?sql='SELECT * FROM kbs_doc_page_parse WHERE documentId = {{documentId}}'
//# 获取文件
assign("n_result_data", "{{n_kb_data}}")

when('!fn.hasValue(n_result_data)', {
  //# Chunk Failed
	act_flow.fail?result='Chunk Failed,Stop The Process'
},{
  //# 分页遍历文件
  foreach('n_data', 'n_result_data', {
    // 下标对应内容
    assign('table_item', '{{n_data.item}}')

    //# 获取图片路径及内容
    assign("image_path", "{{table_item.imagePath}}")
    //# ocr_content
    assign("ocr_content", "{{table_item.content}}")

    // 读取文件包含格式agent
    n_image_type,agt_image_format?imagePath="{{image_path}}"

    // 输出内容
    n_final_data,agt_read_image?imagePath="{{image_path}}"&ocrContent="{{ocr_content}}"&dataType="{{n_image_type}}"

    // 插入新数据
    act_kb.update?name=kbs_doc_page_parse&columnValues={{fn.json('analysis_content, n_final_data')}}&where='{"documentId":"{{fileId}}","page":"{{table_item.page}}"}'
  })
})
