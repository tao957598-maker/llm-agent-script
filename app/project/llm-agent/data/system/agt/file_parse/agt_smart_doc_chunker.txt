		ELSE p.analysis_content

    kbs_doc_page_parse p

    ON p.documentId = f.id

    p.documentId = "{{kbSpaceFileId}}" ) as a where a.content is not null

// 切片处理，避免单次请求超过multipart大小限制
n_content_batches, act_func?script='''
def sourceList = n_result
if (!sourceList) {
    return []
}
def maxBytes = 800 * 1024
def batches = []
def currentBatch = []
def currentBytes = 0
sourceList.each { item ->
    def itemBytes = fn.str(item).getBytes("UTF-8").length + 256
    if (currentBatch && currentBytes + itemBytes > maxBytes) {
        batches << currentBatch
        currentBatch = []
        currentBytes = 0
    }
    currentBatch << item
    currentBytes += itemBytes
}
if (currentBatch) {
    batches << currentBatch
}
return batches
'''
n_result_data, act_func?script='''return []'''
