assign("status", 'Agent is running')
// kbSpaceFileId22
assign("kbSpaceFileId22", '248398006873423872')
// file name
n_file_name, act_kb.selectBy?sql='select  knowledge_name  from kb_spaces_folder_knowledge where id="{{kbSpaceFileId22}}" '
assign("file_name", '{{n_file_name[0].knowledge_name}}')

when('!fn.hasValue(Pages)', {
  // 取全部数据
  n_kb_data, act_kb.selectBy?sql='select page,content from kbs_doc_page_parse where documentId="{{kbSpaceFileId22}}"   ORDER BY CAST(page AS SIGNED) ASC'

},{
    // 取分页数据
 n_kb_data, act_kb.selectBy?sql='select page,content from kbs_doc_page_parse where documentId="{{kbSpaceFileId22}}" and page in ({{Pages}}) ORDER BY CAST(page AS SIGNED) ASC'

})

// 循环优化提示词
foreach('n_optimization', '[1,2,3]', {
    // 生成答案
    n_generated_answer,tpl_prompt,psr_all
    // 优化建议
    n_prompt_optimization_suggestions,tpl_prompt_optimization_suggestions,psr_all
    
    // 优化后的提示词
    n_prompt_optimized,tpl_prompt_optimized,psr_all
    assign("prompt", '{{n_prompt_optimized}}')
})
