// Groovy: 机械校验所有实体
n_entity_validation, act_func?script='''
  def entities = n_entities ?: []
  def threshold = (confidenceThreshold as double) ?: 0.6
  def validChunkIds = n_valid_chunk_ids ?: []
  def results = []
  def seenIds = [] as Set

  entities.each { ent ->
    def issues = []

    // 必填字段检查
    if (!ent.entity_id) issues.add("missing entity_id")
    if (!ent.name || (ent.name as String).trim().isEmpty()) issues.add("missing or empty name")
    if (!ent.entity_type) issues.add("missing entity_type")

    // 证据检查
    if (!ent.evidence || ent.evidence.size() == 0) {
      issues.add("no evidence")
    } else {
      // 检查evidence中chunk_id是否有效
      ent.evidence.each { ev ->
        if (ev.chunk_id && !validChunkIds.contains(ev.chunk_id)) {
          issues.add("invalid chunk_id in evidence: " + ev.chunk_id)
        }
      }
    }

    // 置信度检查
    def conf = (ent.confidence as double) ?: 0.0
    def belowThreshold = conf < threshold

    // ID重复检查
    if (seenIds.contains(ent.entity_id)) {
      issues.add("duplicate entity_id: " + ent.entity_id)
    }
    seenIds.add(ent.entity_id)

    // 一致性检查：Document/Section实体必须是结构实体
    if (ent.entity_type == "Document" && !ent.entity_id?.startsWith("ent_doc_")) {
      issues.add("non-structural Document entity")
    }

    results.add([
      entity_id: ent.entity_id,
      name: ent.name,
      entity_type: ent.entity_type,
      confidence: conf,
      below_threshold: belowThreshold,
      evidence_count: ent.evidence?.size() ?: 0,
      issues: issues,
      valid: issues.size() == 0 && !belowThreshold
    ])
  }

  return results
'''

// Groovy: 机械校验所有关系
n_relation_validation, act_func?script='''
  def relations = n_relations ?: []
  def threshold = (confidenceThreshold as double) ?: 0.6
  def validEntityIds = [] as Set
  n_entities?.each { ent -> if (ent.entity_id) validEntityIds.add(ent.entity_id) }

  def results = []
  def seenIds = [] as Set

  relations.each { rel ->
    def issues = []

    // 必填字段检查
    if (!rel.relation_id) issues.add("missing relation_id")
    if (!rel.relation_type) issues.add("missing relation_type")
    if (!rel.source_entity_id) issues.add("missing source_entity_id")
    if (!rel.target_entity_id) issues.add("missing target_entity_id")

    // 自环检查
    if (rel.source_entity_id && rel.source_entity_id == rel.target_entity_id) {
      issues.add("self-loop relation")
    }

    // 实体引用存在性检查
    if (rel.source_entity_id && !validEntityIds.contains(rel.source_entity_id)) {
      issues.add("source entity not found: " + rel.source_entity_id)
    }
    if (rel.target_entity_id && !validEntityIds.contains(rel.target_entity_id)) {
      issues.add("target entity not found: " + rel.target_entity_id)
    }

    // 置信度检查
    def conf = (rel.confidence as double) ?: 0.0
    def belowThreshold = conf < threshold

    // ID重复检查
    if (seenIds.contains(rel.relation_id)) {
      issues.add("duplicate relation_id: " + rel.relation_id)
    }
    seenIds.add(rel.relation_id)

    results.add([
      relation_id: rel.relation_id,
      relation_type: rel.relation_type,
      source_entity_id: rel.source_entity_id,
      target_entity_id: rel.target_entity_id,
      confidence: conf,
      below_threshold: belowThreshold,
      issues: issues,
      valid: issues.size() == 0 && !belowThreshold
    ])
  }

  return results
'''

// Groovy: 机械校验所有事件
n_event_validation, act_func?script='''
  def events = n_events ?: []
  def threshold = (confidenceThreshold as double) ?: 0.6
  def results = []
  def seenIds = [] as Set

  events.each { evt ->
    def issues = []

    // 必填字段检查
    if (!evt.event_id) issues.add("missing event_id")
    if (!evt.event_type) issues.add("missing event_type")
    if (!evt.description || (evt.description as String).trim().isEmpty()) {
      issues.add("missing or empty description")
    }

    // 置信度检查
    def conf = (evt.confidence as double) ?: 0.0
    def belowThreshold = conf < threshold

    // ID重复检查
    if (seenIds.contains(evt.event_id)) {
      issues.add("duplicate event_id: " + evt.event_id)
    }
    seenIds.add(evt.event_id)

    results.add([
      event_id: evt.event_id,
      event_type: evt.event_type,
      confidence: conf,
      below_threshold: belowThreshold,
      issues: issues,
      valid: issues.size() == 0 && !belowThreshold
    ])
  }

  return results
'''

// Groovy: 汇总机械校验统计
n_mechanical_stats, act_func?script='''
  def entityValid = (n_entity_validation ?: []).findAll { it.valid }.size()
  def entityInvalid = (n_entity_validation ?: []).size() - entityValid
  def relationValid = (n_relation_validation ?: []).findAll { it.valid }.size()
  def relationInvalid = (n_relation_validation ?: []).size() - relationValid
  def eventValid = (n_event_validation ?: []).findAll { it.valid }.size()
  def eventInvalid = (n_event_validation ?: []).size() - eventValid

  return [
    entities: [total: n_entity_validation?.size() ?: 0, valid: entityValid, invalid: entityInvalid],
    relations: [total: n_relation_validation?.size() ?: 0, valid: relationValid, invalid: relationInvalid],
    events: [total: n_event_validation?.size() ?: 0, valid: eventValid, invalid: eventInvalid]
  ]
'''

n_entity_validation_final, act_func?script='''

// ===== Phase 3: Split validated vs review_queue =====

// Groovy: 拆分实体通过的进入validated, 未通过的进入review_queue
n_split_result, act_func?script='''
  def validEntities = []
  def reviewEntities = []
  def validRelations = []
  def reviewRelations = []
  def validEvents = []
  def reviewEvents = []

  // 构建entity_id → issues的查询表
  def entityIssuesMap = [:]
  n_entity_validation_final?.each { r ->
    entityIssuesMap[r.entity_id] = r.issues ?: []
  }
  def relationIssuesMap = [:]
  n_relation_validation_final?.each { r ->
    relationIssuesMap[r.relation_id] = r.issues ?: []
  }
  def eventIssuesMap = [:]
  n_event_validation_final?.each { r ->
    eventIssuesMap[r.event_id] = r.issues ?: []
  }

  // 拆分实体
  n_entities?.each { ent ->
    def issues = entityIssuesMap[ent.entity_id] ?: []
    if (issues.size() == 0 && (ent.confidence ?: 0.0) >= (confidenceThreshold as double)) {
      ent.validation_status = "validated"
      validEntities.add(ent)
    } else {
      ent.validation_status = "review"
      ent.validation_issues = issues
      if ((ent.confidence ?: 0.0) < (confidenceThreshold as double)) {
        ent.validation_issues = (ent.validation_issues ?: []) + ["confidence below threshold: ${ent.confidence}"]
      }
      reviewEntities.add(ent)
    }
  }

  // 拆分关系
  n_relations?.each { rel ->
    def issues = relationIssuesMap[rel.relation_id] ?: []
    if (issues.size() == 0 && (rel.confidence ?: 0.0) >= (confidenceThreshold as double)) {
      rel.validation_status = "validated"
      validRelations.add(rel)
    } else {
      rel.validation_status = "review"
      rel.validation_issues = issues
      if ((rel.confidence ?: 0.0) < (confidenceThreshold as double)) {
        rel.validation_issues = (rel.validation_issues ?: []) + ["confidence below threshold: ${rel.confidence}"]
      }
      reviewRelations.add(rel)
    }
  }

  // 拆分事件
  n_events?.each { evt ->
    def issues = eventIssuesMap[evt.event_id] ?: []
    if (issues.size() == 0 && (evt.confidence ?: 0.0) >= (confidenceThreshold as double)) {
      evt.validation_status = "validated"
      validEvents.add(evt)
    } else {
      evt.validation_status = "review"
      evt.validation_issues = issues
      if ((evt.confidence ?: 0.0) < (confidenceThreshold as double)) {
        evt.validation_issues = (evt.validation_issues ?: []) + ["confidence below threshold: ${evt.confidence}"]
      }
      reviewEvents.add(evt)
    }
  }

  fn.put("valid_entities", validEntities)
  fn.put("review_entities", reviewEntities)
  fn.put("valid_relations", validRelations)
  fn.put("review_relations", reviewRelations)
  fn.put("valid_events", validEvents)
  fn.put("review_events", reviewEvents)

  return [
    validated: [
      entities: validEntities.size(),
      relations: validRelations.size(),
      events: validEvents.size()
    ],
    review: [
      entities: reviewEntities.size(),
      relations: reviewRelations.size(),
      events: reviewEvents.size()
    ]
  ]
'''

// ===== Phase 4: Generate nodes.json =====

// Groovy: 生成图谱节点
n_graph_nodes, act_func?script='''
  def nodes = []
  def nextId = 1

  // 文档层节点 — 从document.json
  def docInfo = n_document ?: [:]
  nodes.add([
    node_id: "node_" + String.format("%04d", nextId++),
    labels: ["Document"],
    properties: [
      document_id: n_document_id ?: "",
      title: docInfo.title ?: "",
      language: docInfo.language ?: "zh",
      page_count: docInfo.page_count ?: 0,
      author: docInfo.author ?: "",
      date: docInfo.date ?: ""
    ]
  ])

  // Section节点 — 从document_structure
  def structure = n_structure ?: [:]
  def sections = structure?.sections ?: []
  def sectionNodeIdMap = [:]
  sections.each { sec ->
    def nodeId = "node_" + String.format("%04d", nextId++)
    sectionNodeIdMap[sec.section_id] = nodeId
    nodes.add([
      node_id: nodeId,
      labels: ["Section"],
      properties: [
        section_id: sec.section_id ?: "",
        title: sec.title ?: "",
        level: sec.level ?: 1,
        page_range: sec.page_range ?: [],
        parent_section_id: sec.parent_section_id ?: null
      ]
    ])
  }

  // Chunk节点 — 从chunks.json
  def chunks = n_chunks ?: []
  def chunkNodeIdMap = [:]
  chunks.each { c ->
    def nodeId = "node_" + String.format("%04d", nextId++)
    chunkNodeIdMap[c.chunk_id] = nodeId
    nodes.add([
      node_id: nodeId,
      labels: ["Chunk"],
      properties: [
        chunk_id: c.chunk_id ?: "",
        section_id: c.section_id ?: "",
        page_range: c.page_range ?: [],
        char_len: c.char_len ?: 0
      ]
    ])
  }

  // 知识层节点 — Entity (带二级标签)
  def entityNodeIdMap = [:]
  n_validated_entities?.each { ent ->
    def nodeId = "node_" + String.format("%04d", nextId++)
    entityNodeIdMap[ent.entity_id] = nodeId
    def labels = ["Entity"]
    if (ent.entity_type) labels.add(ent.entity_type)
    nodes.add([
      node_id: nodeId,
      labels: labels,
      properties: [
        entity_id: ent.entity_id ?: "",
        name: ent.name ?: "",
        entity_type: ent.entity_type ?: "",
        aliases: ent.aliases ?: [],
        attributes: ent.attributes ?: [:],
        confidence: ent.confidence ?: 0.0
      ]
    ])
  }

  // Event节点
  def eventNodeIdMap = [:]
  n_validated_events?.each { evt ->
    def nodeId = "node_" + String.format("%04d", nextId++)
    eventNodeIdMap[evt.event_id] = nodeId
    nodes.add([
      node_id: nodeId,
      labels: ["Event"],
      properties: [
        event_id: evt.event_id ?: "",
        event_type: evt.event_type ?: "",
        trigger: evt.trigger ?: "",
        time: evt.time ?: [:],
        description: evt.description ?: "",
        confidence: evt.confidence ?: 0.0
      ]
    ])
  }

  // Evidence节点 — 从entity evidence中提取
  n_validated_entities?.each { ent ->
    ent.evidence?.eachWithIndex { ev, idx ->
      nodes.add([
        node_id: "node_" + String.format("%04d", nextId++),
        labels: ["Evidence"],
        properties: [
          entity_id: ent.entity_id,
          chunk_id: ev.chunk_id ?: "",
          page_range: ev.page_range ?: [],
          quote: ev.quote ?: ""
        ]
      ])
    }
  }

  // 存入node映射供edges阶段使用
  fn.put("section_node_map", sectionNodeIdMap)
  fn.put("chunk_node_map", chunkNodeIdMap)
  fn.put("entity_node_map", entityNodeIdMap)
  fn.put("event_node_map", eventNodeIdMap)

  return nodes
'''

// ===== Phase 5: Generate edges.json =====

// Groovy: 生成图谱边
n_graph_edges, act_func?script='''
  def edges = []
  def nextEdgeId = 1

  def sectionNodeMap = fn.get("section_node_map") ?: [:]
  def chunkNodeMap = fn.get("chunk_node_map") ?: [:]
  def entityNodeMap = fn.get("entity_node_map") ?: [:]
  def eventNodeMap = fn.get("event_node_map") ?: [:]

  // 查找Document节点
  def docNodeId = ""
  def nodes = n_graph_nodes ?: []
  nodes.each { n ->
    if (n.labels?.contains("Document")) docNodeId = n.node_id
  }

  // 文档结构边: Document → HAS_SECTION → Section
  def structure = n_structure ?: [:]
  def sections = structure?.sections ?: []
  sections.each { sec ->
    def secNodeId = sectionNodeMap[sec.section_id]
    if (docNodeId && secNodeId) {
      edges.add([
        edge_id: "edge_" + String.format("%04d", nextEdgeId++),
        source_id: docNodeId,
        target_id: secNodeId,
        type: "HAS_SECTION",
        properties: [section_id: sec.section_id]
      ])
    }
  }

  // Section层级边: Section → HAS_SUBSECTION → Section
  sections.each { sec ->
    if (sec.parent_section_id) {
      def childNodeId = sectionNodeMap[sec.section_id]
      def parentNodeId = sectionNodeMap[sec.parent_section_id]
      if (childNodeId && parentNodeId) {
        edges.add([
          edge_id: "edge_" + String.format("%04d", nextEdgeId++),
          source_id: parentNodeId,
          target_id: childNodeId,
          type: "HAS_SUBSECTION",
          properties: [:]
        ])
      }
    }
  }

  // Section → HAS_CHUNK → Chunk
  def chunks = n_chunks ?: []
  chunks.each { c ->
    def secNodeId = sectionNodeMap[c.section_id]
    def chunkNodeId = chunkNodeMap[c.chunk_id]
    if (secNodeId && chunkNodeId) {
      edges.add([
        edge_id: "edge_" + String.format("%04d", nextEdgeId++),
        source_id: secNodeId,
        target_id: chunkNodeId,
        type: "HAS_CHUNK",
        properties: [chunk_id: c.chunk_id]
      ])
    }
  }

  // 知识关系边: Entity → {relation_type} → Entity
  n_validated_relations?.each { rel ->
    def sourceNodeId = entityNodeMap[rel.source_entity_id]
    def targetNodeId = entityNodeMap[rel.target_entity_id]
    if (sourceNodeId && targetNodeId) {
      edges.add([
        edge_id: "edge_" + String.format("%04d", nextEdgeId++),
        source_id: sourceNodeId,
        target_id: targetNodeId,
        type: rel.relation_type ?: "RELATED_TO",
        properties: [
          relation_id: rel.relation_id ?: "",
          qualifiers: rel.qualifiers ?: [:],
          confidence: rel.confidence ?: 0.0,
          implicit: rel.implicit ?: false
        ]
      ])
    }
  }

  // 事件参与边: Entity → PARTICIPATES_IN → Event
  n_validated_events?.each { evt ->
    def evtNodeId = eventNodeMap[evt.event_id]
    evt.participants?.each { p ->
      def entityNodeId = entityNodeMap[p.entity_id]
      if (entityNodeId && evtNodeId) {
        edges.add([
          edge_id: "edge_" + String.format("%04d", nextEdgeId++),
          source_id: entityNodeId,
          target_id: evtNodeId,
          type: "PARTICIPATES_IN",
          properties: [
            event_id: evt.event_id,
            role: p.role ?: "",
            entity_id: p.entity_id ?: ""
          ]
        ])
      }
    }
  }

  // 证据边: Chunk → MENTIONS → Entity
  n_validated_entities?.each { ent ->
    def entityNodeId = entityNodeMap[ent.entity_id]
    ent.evidence?.each { ev ->
      def chunkNodeId = chunkNodeMap[ev.chunk_id]
      if (chunkNodeId && entityNodeId) {
        edges.add([
          edge_id: "edge_" + String.format("%04d", nextEdgeId++),
          source_id: chunkNodeId,
          target_id: entityNodeId,
          type: "MENTIONS",
          properties: [
            page_range: ev.page_range ?: [],
            quote: ev.quote ?: ""
          ]

   - A PART_OF B 和 B PART_OF A 同时存在
   - A PRECEDES B 和 B PRECEDES A 同时存在
   - 同一实体对同时有 RELATED_TO 和更具体的关系

4. **事件一致性**：同一事件中参与者类型是否合理
   - 事件类型与参与方实体类型是否匹配
   - 事件时间是否在合理范围内
