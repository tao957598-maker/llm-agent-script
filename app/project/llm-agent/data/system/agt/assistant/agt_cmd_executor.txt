  "description": "统一命令执行器，支持本地AC和云端两种执行模式，带危险命令检测",
  "parameter": [
    {
      "name": "cmd",
      "label": "命令",
      "required": true
    },
    {
      "name": "cmd_executor_name",
      "label": "命令名称"
    },
    {
      "name": "server_id",
      "label": "服务器ID"
    },
    {
      "name": "async",
      "label": "是否异步执行"
    }
  ],
  "resultPlan": "",
  "life": "100"

assign('cmd_executor_name_text', '{{fn.optional(cmd_executor_name, "执行命令")}}')
assign('n_executor_mode', '{{fn.when(n_executor_type == "local_ac", "本地", "云端")}}')
assign('n_async', '{{fn.optional(async, "false")}}')
// 指数退避等待时间列表（秒），总和约 360s（6分钟）
// 前期慢查：[5, 8, 12]，中期过渡：[18, 25, 35]，后期稳定：[45, 45, 45, 45, 45, 45, 45, 45, 45, 45, 45, 45, 45, 45]
assign('n_poll_delays', '[5, 8, 12, 18, 25, 35, 45, 55, 65, 85, 95, 105, 30, 35, 55, 65, 75, 85, 35, 45, 45]')

//## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 风险检测
// 检测危险命令（避免复杂正则，改用字符串包含判断）
n_is_dangerous, act_func?script='''
    def c = (cmd ?: "").trim().toLowerCase()
    def d = false
    def reason = ""

    // 递归删除根目录
    if (c.contains("rm -rf /") || c.contains("rm -r -f /") || c.contains("rm -rf /*") || c.contains("rm -r -f /*")) {
        d = true
        reason = "递归删除根目录"
    } else if (c.contains("rm -r /")) {
        d = true
        reason = "删除根目录"
    } else if (c.contains("del /s") || c.contains("del /f") || c.contains("rmdir /s") || c.contains("rmdir /q")) {
        d = true
        reason = "删除系统目录"
    } else if (c.contains("dd if=") && (c.contains("of=/dev/") || c.contains("of=\\device"))) {
        d = true
        reason = "磁盘写入操作"
    } else if (c.contains("mkfs") || c.contains("format ")) {
        d = true
        reason = "格式化操作"
    } else if (c.contains(":(){") && c.contains(":|:") && c.contains("};:")) {
        d = true
        reason = "Fork炸弹"
    }

    fn.put("n_danger_reason", reason)
    return d
'''

when('fn.bool(n_is_dangerous)', {
  //## 检测到危险命令，已拦截
  n_danger_msg, act_func?script='''
    def msg = "⚠️ 危险命令已拦截：" + (n_danger_reason ?: "") + "\n\n命令：" + (cmd ?: "") + "\n\n如果必要，请人工操作。";
    return msg
  '''
  assign("n_executor_return", '{{n_danger_msg}}')
  act_flow.go?plan=break
})



//## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 执行命令
act_memory.assign?var=cmd_text&value="{{cmd}}"&type=code




// 执行模式为本地
when('n_executor_type == "local_ac"', {
  n_result, act_rpa.send?command=RunShell&parameters='{"shell_input":"{{cmd_text}}"}'
  //# n_result-log show
  log("-------------local----n_result:{{n_result}}")
  //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 输出结果
  assign("n_cmd_executor_result", '{{n_result.data}}')
}, {

  // 执行模式为云端，且异步
  when('n_async == "true"', {
    //## [{{n_executor_mode}}] {{cmd_executor_name_text}} 查看异步执行详情
    n_result, act_api.doExecuteDynamic?name="send_cmd"&dynamicParamData='{"command":"{{cmd_text}}","commands":"","async":true,"server_id":"{{server_id}}"}' 
    
    assign("n_result_data", '{{n_result.data}}')
    assign('n_task_id', '{{n_result_data.task_id!""}}')
    assign('n_task_status', '{{n_result_data.status!"pending"}}')
    assign('n_task_stdout', '{{n_result_data.stdout!""}}')
    assign('n_task_stderr', '{{n_result_data.stderr!""}}')
    assign('n_task_exit_code', '{{fn.optional(n_result_data.exit_code, "")}}')
    assign('n_poll_timeout', 'false')

    //当taskID为空或者状态是failed
    when('n_task_id == "" || n_task_status == "failed"', {
      //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 异步输出结果（失败）
      n_return, act_memory.assign?var=n_cmd_executor_result&value="{{((cmd_text == '')?then('', cmd_text))}}<br>{{n_task_stderr}}"&type=code
    }, {
    // 正常就进入循环  
      foreach('n_poll_loop', 'n_poll_delays', {
          when('n_task_status != "success" && n_task_status != "failed" && n_task_status != "done"', {
          act_flow.wait?second={{n_poll_loop}}
          n_query_result, act_api.doExecuteDynamic?name="get_cmd_res"&dynamicParamData='{"command":"{{cmd_text}}","commands":"","async":false,"server_id":"{{server_id}}","task_id":"{{n_task_id}}"}'
          assign('n_query_data', '{{n_query_result.data}}')
          assign('n_task_status', '{{n_query_data.status!n_task_status!"unknown"}}')
          assign('n_task_stdout', '{{n_query_data.stdout!n_task_stdout!""}}')
          assign('n_task_stderr', '{{n_query_data.stderr!n_task_stderr!""}}')
          assign('n_task_exit_code', '{{fn.optional(n_query_data.exit_code, n_task_exit_code)}}')
        })
      })
      // 循环结束还没有终止态就判断超时
      when('n_task_status != "success" && n_task_status != "failed" && n_task_status != "done"', {
        assign('n_poll_timeout', 'true')
      })

      // 超时
      when('n_poll_timeout == "true"', {
          //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 输出结果（超时）
          n_return, act_memory.assign?var=n_cmd_executor_result&value="命令执行超时（360s），请稍后手动查询结果。<br>{{cmd_text}}<br>{{n_task_stderr}}"&type=code
      }, {
        //没有超时
        when('n_task_status == "success" || (n_task_status == "done" && (n_task_exit_code!"") == "0")', {
            //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 轮询输出结果（成功）
            n_return, act_memory.assign?var=n_cmd_executor_result&value="{{n_task_stdout}}"&type=code
        }, {
            //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 轮询输出结果（失败）
            n_return, act_memory.assign?var=n_cmd_executor_result&value="{{((cmd_text == '')?then('', cmd_text))}}<br>{{n_task_stderr}}"&type=code
        })
      })
    })
  }, {
    // 执行模式为云端，且同步
    //## [{{n_executor_mode}}] {{cmd_executor_name_text}} 查看同步执行详情
    n_result, act_api.doExecuteDynamic?name="send_cmd"&dynamicParamData='{"command":"{{cmd_text}}","commands":"","async":false,"server_id":"{{server_id}}"}'
    assign("n_result_data", '{{n_result.data}}')
    when('n_result_data.exit_code == 0', {
        //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 同步输出结果（成功）
        n_return, act_memory.assign?var=n_cmd_executor_result&value="{{n_result_data.stdout}}"&type=code
    }, {
        //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 同步输出结果（失败）
        n_return, act_memory.assign?var=n_cmd_executor_result&value="{{((n_result_data.command == '')?then(n_result_data.commands, n_result_data.command))}}<br>{{n_result_data.stderr}}"&type=code
    })
  })
})
// 捕捉所有前序异常,将当前流程转为正常状态。（可使用act_flow.fail再次将流程转为异常状态）
on_failure({
	assign("n_return", "fail")
	// goto语句，跳转到代码结尾
	// act_flow.go?plan=end
})
