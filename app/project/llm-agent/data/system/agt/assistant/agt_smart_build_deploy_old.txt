// 定义内部 agent：find_project
// 参数: project_name, os, git_repo_url
// 输出: n_project_root, n_need_clone
agent('agt_inner_find_project',{
  // 设置默认变量
  assign('n_need_clone', 'false')
  when('n_os == "windows"', {
    assign('n_find_cmd', 'for %d in (Z Y X W V U T S R Q P O N M L K J I H G F E D C) do @if exist %d:\\ @for /f "delims=" %i in (\'dir /b /s /ad "%d:\\" 2^>nul ^| findstr /i "\\{{n_project_name}}$"\') do @if exist "%i\\deploy\\/build_thin_all.sh" echo %i')
  }, {
    // 搜索常用路径和临时目录
    assign('n_find_cmd', 'find /app /opt /etc/ /home /var /tmp -maxdepth 4 -type d -name "{{n_project_name}}" 2>/dev/null | while read d; do [ -f "$d/deploy/build_thin_all.sh" ] && echo "$d"; done | head -5')
  })
  // 执行查找命令
  n_find_result, agt_inner_cmd_executor?cmd={{n_find_cmd}}&cmd_executor_name="查找项目目录"
  assign('n_find_output', '{{n_find_result}}')
  n_find_ok, act_func?script='''
    def output = n_find_output ?: ""
    def lines = output.split("\\n")
    def path = lines?[0]?.trim()
    if (path == null || path.contains("NOT_FOUND") || path.isEmpty()) return false
    def forwardSlashPath = path.replace('\\\\\\\\', '/').replaceAll(/[\r\n]+$/, '')
    fn.put('n_project_root', forwardSlashPath)
    return true
  '''
  when('fn.bool(n_find_ok)', {
    assign('n_log', '{{n_log}}\n✓ 找到项目: {{n_project_root}}')
  }, {
    // 项目未找到，检查是否有 Git 仓库地址
    when('fn.hasValue(n_git_repo_url)', {
      assign('n_need_clone', 'true')
      assign('n_log', '{{n_log}}\n⚠️ 未找到本地项目，需要从 Git 克隆\n仓库地址: {{n_git_repo_url}}')
    }, {
      assign('n_deploy_result', '❌ 未找到项目 {{n_project_name}}\n\n建议:\n1. 检查项目是否存在\n2. 提供 Git 仓库地址进行克隆')
      act_flow.go?plan=break
    })
  })
})

// 定义内部 agent：clone_repo
// 参数: os, project_name, git_repo_url, branch_name, need_clone
// 功能: 克隆或更新 Git 仓库
// 优先使用 find_project 找到的路径，如果不存在才 clone 到 /tmp
agent('agt_inner_clone_repo',{
  // 检查 /tmp/git-repository-clone 下是否有同名项目（优先使用）
  when('n_os == "windows"', {
    assign('n_tmp_check_cmd', 'if exist "C:\\temp\\git-repository-clone\\{{n_project_name}}" (echo EXISTS) else (echo NOT_EXISTS)')
    assign('n_tmp_project_root', 'C:\\temp\\git-repository-clone\\{{n_project_name}}')
  }, {
    assign('n_tmp_check_cmd', 'if [ -d "/tmp/git-repository-clone/{{n_project_name}}" ]; then echo EXISTS; else echo NOT_EXISTS; fi')
    assign('n_tmp_project_root', '/tmp/git-repository-clone/{{n_project_name}}')
  })
  n_tmp_check, agt_inner_cmd_executor?cmd={{n_tmp_check_cmd}}&cmd_executor_name="检查/tmp临时目录"
  assign('n_tmp_exists', '{{fn.contains(n_tmp_check, "EXISTS")}}')
  
  // 优先使用 /tmp 下的项目
  when('fn.bool(n_tmp_exists)', {
    assign('n_project_root', '{{n_tmp_project_root}}')
    assign('n_log', '{{n_log}}\n📥 检测到 /tmp 下已有项目: {{n_project_root}}')
  })
  
  // 如果 /tmp 没有，检查 find_project 结果
  when('!fn.bool(n_tmp_exists) && fn.hasValue(n_project_root)', {
    assign('n_log', '{{n_log}}\n📥 使用找到的项目: {{n_project_root}}')
  })
  
  // 如果任何项目路径存在，尝试 git pull
  when('fn.hasValue(n_project_root)', {
    assign('n_log', '{{n_log}}\n📥 尝试 git pull 更新')
    assign('n_git_cmd', 'cd "{{n_project_root}}" && git fetch origin && git checkout {{n_branch_name}} && git pull')
    n_git_result, agt_inner_cmd_executor?cmd={{n_git_cmd}}&cmd_executor_name="Git更新代码"
    assign('n_git_pull_output', '{{n_git_result}}')
    n_git_ok, act_func?script='''
      def output = n_git_pull_output ?: ""
      def success = output.contains("Updating") || output.contains("Already up to date") || output.contains("Fast-forward") || output.contains("branch is up to date") || output.contains("Switched to a new branch") || output.contains("Switched to branch")
      fn.put("n_git_output", output)
      return success
    '''
    when('fn.bool(n_git_ok)', {
      assign('n_log', '{{n_log}}\n✓ 代码更新成功')
    }, {
      assign('n_log', '{{n_log}}\n⚠️ Git更新失败，将尝试继续使用现有目录')
    })
  }, {
    // 没有找到任何项目，需要 clone
    when('fn.bool(n_need_clone)', {
      assign('n_log', '{{n_log}}\n📥 未找到本地项目，准备克隆仓库')
      when('n_os == "windows"', {
        assign('n_clone_base', 'C:\\temp\\git-repository-clone')
        assign('n_project_root', '{{n_clone_base}}\\{{n_project_name}}')
      }, {
        assign('n_clone_base', '/tmp/git-repository-clone')
        assign('n_project_root', '{{n_clone_base}}/{{n_project_name}}')
      })
      // 确保 base 目录存在
      n_mkdir_result, agt_inner_cmd_executor?cmd='mkdir -p "{{n_clone_base}}"'&cmd_executor_name="创建基础目录"
      // 执行 clone
      assign('n_is_https_url', '{{fn.startsWith(n_git_repo_url, "https://")}}')
      assign('n_effective_git_url', "{{fn.gitHttpsToSsh(n_git_repo_url)}}")
      when('fn.bool(n_is_https_url)', {
        assign('n_clone_cmd', 'git clone -b {{n_branch_name}} "{{n_effective_git_url}}" "{{n_project_root}}"')
        assign('n_log', '{{n_log}}\n(已转换为SSH: {{n_effective_git_url}})')
      }, {
        assign('n_clone_cmd', 'git clone -b {{n_branch_name}} "{{n_git_repo_url}}" "{{n_project_root}}"')
      })
      n_clone_result, agt_inner_cmd_executor?cmd="{{n_clone_cmd}}"&cmd_executor_name="克隆Git仓库"
      assign('n_clone_output', '{{n_executor_return}}')
      // 验证 clone 成功
      n_clone_ok, act_func?script='''
        def output = n_clone_output ?: ""
        def success = output.contains("Cloning into") || output.contains("warning: You appear to have cloned an empty repository")
        return success
      '''
      when('fn.bool(n_clone_ok)', {
        assign('n_log', '{{n_log}}\n✓ 仓库克隆成功')
      }, {
        assign('n_deploy_result', '❌ 仓库克隆失败\n\n命令: {{n_clone_cmd}}\n\n输出: {{n_clone_result}}')
        act_flow.go?plan=break
      })
    }, {
      assign('n_deploy_result', '❌ 未找到项目且未提供 Git 仓库地址\n\n建议:\n1. 检查项目是否存在\n2. 提供 Git 仓库地址')
      act_flow.go?plan=break
    })
  })
})

// 功能: 检查 Git 环境，如未安装则自动安装

    }, {
      assign('n_git_install_cmd', 'sudo apt-get install -y git || sudo yum install -y git')
      n_git_install, agt_inner_cmd_executor?cmd={{n_git_install_cmd}}&cmd_executor_name="安装Git"
    })
    assign('n_log', '{{n_log}}\n✓ Git 安装完成')
  })
})

// 定义内部 agent：check_maven
// 参数: os
// 输出: n_mvn_bin
// 功能: 检查 Maven 环境，如未安装则自动安装
agent('agt_inner_check_maven',{
  // 检查 Maven 环境（直接用 which mvn 找）
  n_mvn_path, agt_inner_cmd_executor?cmd='which mvn 2>/dev/null || echo "NOT_FOUND"'&cmd_executor_name="检查Maven版本"
  assign('n_mvn_found', '{{fn.contains(n_mvn_path, "/mvn") && !fn.contains(n_mvn_path, "NOT_FOUND")}}')
  when('fn.bool(n_mvn_found)', {
    // 找到 Maven - 提取 Maven home 目录
    n_mvn_home, act_func?script='''
      def mvnCmd = n_mvn_path ?: ""
      if (mvnCmd.endsWith("/bin/mvn")) {
        return mvnCmd.substring(0, mvnCmd.length() - 8)
      } else if (mvnCmd.endsWith("/mvn")) {
        return mvnCmd.substring(0, mvnCmd.length() - 4)
      }
      return mvnCmd
    '''
    assign('n_mvn_bin', '{{n_mvn_home}}/bin')
    assign('n_log', '{{n_log}}\n✓ 找到 Maven: {{n_mvn_bin}}')
  }, {
    // Maven 未找到，尝试安装
    when('n_os == "windows"', {
      assign('n_mvn_install_cmd', 'winget install --id Apache.Maven -e --source winget || choco install maven -y')
    }, {
      assign('n_mvn_install_cmd', 'sudo apt-get update && sudo apt-get install -y maven || sudo yum install -y maven || (mkdir -p /opt && curl -fsSL https://archive.apache.org/dist/maven/maven-3/3.9.6/binaries/apache-maven-3.9.6-bin.tar.gz -o /tmp/maven.tar.gz && tar -xzf /tmp/maven.tar.gz -C /opt && rm /tmp/maven.tar.gz && echo "/opt/apache-maven-3.9.6/bin")')
    })
    n_mvn_install_result, agt_inner_cmd_executor?cmd={{n_mvn_install_cmd}}&cmd_executor_name="安装Maven"
    // 再次检查 Maven (安装后）
    n_mvn_install_check, agt_inner_cmd_executor?cmd='which mvn 2>/dev/null || echo "NOT_FOUND"'&cmd_executor_name="再次检查Maven"
    assign('n_mvn_install_ok', '{{fn.contains(n_mvn_install_check, "/mvn") && !fn.contains(n_mvn_install_check, "NOT_FOUND")}}')
    when('fn.bool(n_mvn_install_ok)', {
      assign('n_log', '{{n_log}}\n✓ Maven 安装完成')
      assign('n_mvn_bin', '/opt/apache-maven-3.9.6/bin')
    }, {
      assign('n_deploy_result', '❌ Maven 未安装且无法自动安装')
      act_flow.go?plan=break
    })
  })
})

// 定义内部 agent：switch_branch
// 参数: os, project_root, branch_name, git_repo_url
// 功能: 切换 Git 分支，如目录不存在则重新克隆
agent('agt_inner_switch_branch',{
  // 先检查目录是否存在
  when('n_os == "windows"', {
    assign('n_check_dir_cmd', 'if exist "{{n_project_root}}" (echo EXISTS) else (echo NOT_EXISTS)')
  }, {
    assign('n_check_dir_cmd', 'if [ -d "{{n_project_root}}" ]; then echo EXISTS; else echo NOT_EXISTS; fi')
  })
  n_dir_check, agt_inner_cmd_executor?cmd={{n_check_dir_cmd}}&cmd_executor_name="检查项目目录"
  assign('n_dir_exists', '{{n_dir_check?trim == "EXISTS"}}')
  when('!fn.bool(n_dir_exists)', {

  assign('n_skip_ssh', '{{fn.str(n_env) == "localhost"}}')
  when('!fn.bool(n_skip_ssh)', {
    // cloud 模式：检查 Git SSH 连通性
    when('n_executor_type == "cloud"', {
      // 检查 Git SSH 连通性（用 grep 判断，返回码 0=成功，1=失败）
      assign('n_git_ssh_test_cmd', 'ssh -T -o BatchMode=yes -o ConnectTimeout=10 git@github.com 2>&1 | grep -i "successfully authenticated"')
      n_git_ssh_result, agt_inner_cmd_executor?cmd={{n_git_ssh_test_cmd}}&cmd_executor_name="测试GitHub SSH连接"
      n_git_ssh_ok, act_func?script='''
        def output = n_git_ssh_result ?: ""
        fn.put('n_git_ssh_output', output)
        // grep 成功匹配返回 true
        return !output.isEmpty() && output.toLowerCase().contains("successfully authenticated")
      '''
      log("--------------------n_git_ssh_ok:{{n_git_ssh_ok}}")
      when('!fn.bool(n_git_ssh_ok)', {
        // 生成 RSA 公钥
        when('n_os == "windows"', {
          assign('n_keygen_cmd', 'if not exist "%USERPROFILE%\\.ssh\\id_rsa.pub" ssh-keygen -t rsa -b 4096 -N "" -f "%USERPROFILE%\\.ssh\\id_rsa"')

          assign('n_pubkey_cmd', 'type "%USERPROFILE%\\.ssh\\id_rsa.pub"')
        }, {
          assign('n_pubkey_cmd', '[ -f ~/.ssh/id_rsa.pub ] || ssh-keygen -t rsa -b 4096 -N "" -f ~/.ssh/id_rsa; cat ~/.ssh/id_rsa.pub')

        n_pubkey_result, agt_inner_cmd_executor?cmd={{n_pubkey_cmd}}&cmd_executor_name="获取SSH公钥"

          def output = n_pubkey_result ?: ""

          def pubkey = lines?.find { it.trim().contains("ssh-rsa") }

        '''

          def err = n_git_ssh_output ?: "未知错误"

          return msg
        '''
        assign('n_deploy_result', '{{n_err_msg}}')
        act_flow.go?plan=break
      })
      assign('n_log', '{{n_log}}\n✓ Git SSH 验证通过')
    })

    // local_ac 模式：检查云端机器到远程服务器的 SSH（用于部署）
    when('n_executor_type == "local_ac"', {
      assign('n_ssh_test_cmd', 'ssh -p {{n_ssh_port}} -o BatchMode=yes -o ConnectTimeout=5 -o StrictHostKeyChecking=no -o UserKnownHostsFile=/dev/null root@{{n_ssh_host}} echo OK && echo SSH_OK || echo SSH_FAIL')
      n_ssh_test_result, agt_inner_cmd_executor?cmd={{n_ssh_test_cmd}}&cmd_executor_name="测试远程服务器SSH连接"
      n_ssh_ok, act_func?script='''
        def output = n_ssh_test_result ?: ""
        fn.put('n_ssh_err', output)
        return output.contains("SSH_OK")
      '''
      when('!fn.bool(n_ssh_ok)', {
        // 生成 RSA 公钥
        when('n_os == "windows"', {
          assign('n_keygen_cmd', 'if not exist "%USERPROFILE%\\.ssh\\id_rsa.pub" ssh-keygen -t rsa -b 4096 -N "" -f "%USERPROFILE%\\.ssh\\id_rsa"')
          n_keygen_result, agt_inner_cmd_executor?cmd={{n_keygen_cmd}}&cmd_executor_name="生成SSH密钥"
          assign('n_pubkey_cmd', 'type "%USERPROFILE%\\.ssh\\id_rsa.pub"')
        }, {
          assign('n_pubkey_cmd', '[ -f ~/.ssh/id_rsa.pub ] || ssh-keygen -t rsa -b 4096 -N "" -f ~/.ssh/id_rsa; cat ~/.ssh/id_rsa.pub')
        })
        n_pubkey_result, agt_inner_cmd_executor?cmd={{n_pubkey_cmd}}&cmd_executor_name="获取SSH公钥"
        n_pubkey, act_func?script='''
          def output = n_pubkey_result ?: ""
          def lines = output.split("\\n")
          def pubkey = lines?.find { it.trim().contains("ssh-rsa") }
          return pubkey?.trim() ?: ""
        '''
        assign('n_deploy_result', '⚠️ 云端到远程服务器 SSH 未通过\n\n错误：{{n_ssh_err}}\n\n配置方法：将以下公钥添加到远程服务器 ~/.ssh/authorized_keys\n\n{{n_pubkey}}\n\n添加完成后重新执行部署。')
        act_flow.go?plan=break

  })

// 定义命令执行器 agent
agent('agt_inner_cmd_executor',{
  assign('cmd_executor_name_text', '{{fn.optional(cmd_executor_name, "执行命令")}}')
  assign('n_executor_mode', '{{fn.when(n_executor_type == "local_ac", "本地", "云端")}}')

  //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 执行命令
  act_memory.assign?var=cmd_text&value={{cmd}}&type=code

  when('n_executor_type == "local_ac"', {
    n_result, act_rpa.send?command=RunShell&parameters='{"shell_input":"{{cmd_text}}"}'
    //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 输出结果
    assign("n_executor_return", '{{n_result.data}}')
  }, {
    //## [{{n_executor_mode}}] {{cmd_executor_name_text}} 查看执行详情
    n_result, act_api.doExecuteDynamic?name="send_cmd"&dynamicParamData='{"command":"{{cmd_text}}","commands":"","server_id":"{{n_server_id}}"}'
    assign("n_result_data","{{n_result.data}}")
    when('n_result_data.exit_code == 0', {
        //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 成功
        act_memory.assign?var=n_executor_return&value={{n_result_data.stdout}}&type=code
    }, {
        //## [{{n_executor_mode}}] {{cmd_executor_name_text}} - 失败
        act_memory.assign?var=n_executor_return&value="{{((n_result_data.command == '')?then(n_result_data.commands, n_result_data.command))}}\n{{n_result_data.stderr}}"&type=code
    })
  })
})
