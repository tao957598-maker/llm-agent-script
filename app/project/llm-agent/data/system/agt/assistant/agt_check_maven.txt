/*{
  "title": "检查Maven环境",
  "description": "检查 Maven 环境是否安装，如未安装则自动安装，支持 JDK 21",
  "parameter": [
    {
      "name": "os",
      "label": "操作系统",
      "required": true
    },
    {
      "name": "server_id",
      "label": "服务器ID"
    }
  ],
  "resultPlan": "n_mvn_plan_result",
  "discoverable": false,
  "life": "100"
}*/

assign('n_log', '')
assign('n_mvn_bin_path', '')
assign('n_mvn_version', '')

// 检查 Maven 是否存在 
n_mvn_path, agt_cmd_executor.assistant?cmd='which mvn 2>/dev/null || echo "NOT_FOUND"'&cmd_executor_name="检查Maven路径"&server_id={{server_id}}
assign('n_mvn_found', '{{fn.contains(n_mvn_path, "/mvn") && !fn.contains(n_mvn_path, "NOT_FOUND")}}')

when('fn.bool(n_mvn_found)', {
  // 找到 Maven，检查版本
  n_mvn_version_raw, agt_cmd_executor.assistant?cmd='mvn -version | sed "s/\x1B\[[0-9;]*[A-Za-z]//g"'&cmd_executor_name="检查Maven版本号"&server_id={{server_id}}
  // 解析版本号
  n_mvn_version, act_func?script='''
    def raw = (n_mvn_version_raw ?: "").trim()
    def ver = ""
    if (raw.contains("Apache Maven ")) {
      def after = raw.substring(raw.indexOf("Apache Maven ") + 14)
      if (after.contains(" ")) {
        ver = after.substring(0, after.indexOf(" "))
      }
    }
    return ver
  '''
  // 判断是否是 3.9.x 版本 
  assign('n_is_39x', '{{fn.contains(n_mvn_version, "3.9")}}')
  when('fn.bool(n_is_39x)', {
    // 已是 3.9.x，直接使用提取 home 路径
    n_mvn_home, act_func?script='''
      def mvnCmd = (n_mvn_path ?: "")
      if (mvnCmd.contains("/bin/mvn")) {
        return mvnCmd.substring(0, mvnCmd.indexOf("/bin/mvn"))
      } else if (mvnCmd.contains("/mvn")) {
        return mvnCmd.substring(0, mvnCmd.indexOf("/mvn"))
      }
      return mvnCmd
    '''
    assign('n_mvn_bin_path', '{{n_mvn_home}}/bin')
    assign('n_log', '{{n_log}}\n✓ 已有 Maven {{n_mvn_version}}，路径: {{n_mvn_bin_path}}')
    assign("n_mvn_plan_result", '{"n_log":"{{n_log}}","n_data":"{{n_mvn_bin_path}}","is_plan_break":"false"}')
  }, {
    // 版本不是 3.9.x，需要下载 3.9.6
    assign('n_log', '{{n_log}}\n⚠ 当前 Maven 版本 {{n_mvn_version}} 不兼容 JDK 21，将下载 Maven 3.9.6')
    when('n_os == "windows"', {
      assign('n_mvn_install_cmd', 'winget install --id Apache.Maven -e --source winget || choco install maven -y')
    }, {
      assign('n_mvn_install_cmd', 'mkdir -p /tmp && curl -fsSL https://archive.apache.org/dist/maven/maven-3/3.9.6/binaries/apache-maven-3.9.6-bin.tar.gz -o /tmp/maven.tar.gz && tar -xzf /tmp/maven.tar.gz -C /tmp && rm /tmp/maven.tar.gz && /tmp/apache-maven-3.9.6/bin/mvn -version')
    })
    n_mvn_install_result, agt_cmd_executor.assistant?cmd={{n_mvn_install_cmd}}&cmd_executor_name="下载Maven 3.9.6"&server_id={{server_id}}
    assign('n_mvn_bin_path', '/tmp/apache-maven-3.9.6/bin')
    assign('n_log', '{{n_log}}\n✓ Maven 3.9.6 下载完成')
    assign("n_mvn_plan_result", '{"n_log":"{{n_log}}","n_data":"{{n_mvn_bin_path}}","is_plan_break":"false"}')
  })
}, {
  // Maven 不存在，直接下载 3.9.6
  assign('n_log', '{{n_log}}\n⚠ 未找到 Maven，将下载 Maven 3.9.6')
  when('n_os == "windows"', {
    assign('n_mvn_install_cmd', 'winget install --id Apache.Maven -e --source winget || choco install maven -y')
  }, {
    assign('n_mvn_install_cmd', 'mkdir -p /tmp && curl -fsSL https://archive.apache.org/dist/maven/maven-3/3.9.6/binaries/apache-maven-3.9.6-bin.tar.gz -o /tmp/maven.tar.gz && tar -xzf /tmp/maven.tar.gz -C /tmp && rm /tmp/maven.tar.gz && /tmp/apache-maven-3.9.6/bin/mvn -version')
  })
  n_mvn_install_result, agt_cmd_executor.assistant?cmd={{n_mvn_install_cmd}}&cmd_executor_name="下载Maven 3.9.6"&server_id={{server_id}}
  assign('n_mvn_bin_path', '/tmp/apache-maven-3.9.6/bin')
  assign('n_log', '{{n_log}}\n✓ Maven 3.9.6 下载完成')
  assign("n_mvn_plan_result", '{"n_log":"{{n_log}}","n_data":"{{n_mvn_bin_path}}","is_plan_break":"false"}')
})
