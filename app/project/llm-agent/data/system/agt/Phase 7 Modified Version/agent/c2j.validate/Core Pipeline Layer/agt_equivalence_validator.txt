#子模块 5.1：工具函数准备

// DETERMINISTIC: wc3270 scenario.json → Gherkin .feature + cucumber-manifest (no LLM).

import com.alibaba.fastjson2.JSON
import com.alibaba.fastjson2.JSONArray
import com.alibaba.fastjson2.JSONObject
import java.nio.charset.StandardCharsets
import java.nio.file.Files
import java.nio.file.Path
import java.time.Instant

def rootDir = com.llm.agent.common.constant.AppDirConsts.getRootDir()
def projectKey = fn.str(fn.get("n_project_key"))?.trim(); if (!projectKey) throw new IllegalStateException("Missing required parameter: projectKey")
def projectRoot = fn.str(fn.get("n_project_root"))?.trim(); if (!projectRoot) throw new IllegalStateException("Missing required parameter: projectRoot")
def version = fn.str(fn.get("n_ir_version")) ?: "v1"
def failOnUnmapped = (fn.str(fn.get("n_fail_on_unmapped")) ?: "true").toBoolean()
def catalogRel = fn.str(fn.get("n_trace_catalog_rel")) ?: "verification/trace-scenarios/trace-catalog.json"
def base = Path.of(rootDir, projectRoot).toAbsolutePath().normalize()
def projectRootPrefix = projectRoot.replace("\\", "/")
while (projectRootPrefix.endsWith("/")) projectRootPrefix = projectRootPrefix.substring(0, projectRootPrefix.length() - 1)
def normalizeProjectRel = { String r ->
    def s = (r ?: "").replace("\\", "/")
    return s.startsWith(projectRootPrefix + "/") ? s.substring(projectRootPrefix.length() + 1) : s
}
def rel = { String r ->
    def normalized = normalizeProjectRel(r)
    def p = Path.of(normalized)
    return p.isAbsolute() ? p.normalize() : base.resolve(normalized.replace("/", java.io.File.separator)).normalize()
}
def readJson = { String r ->
    def p = rel(r)
    if (!Files.isRegularFile(p)) throw new IllegalStateException("Missing JSON: " + p)
    return JSON.parseObject(Files.readString(p, StandardCharsets.UTF_8))
}
def writeText = { String r, String text ->
    def p = rel(r)
    Files.createDirectories(p.getParent())
    Files.writeString(p, text, StandardCharsets.UTF_8)
}
def esc = { String s -> (s ?: "").replace("\\", "\\\\").replace("\"", "\\\"") }
def tag = { String s -> (s ?: "unknown").replaceAll("[^a-zA-Z0-9_-]", "-").toLowerCase() }
def jsonArray = { JSONObject o, String k -> o?.getJSONArray(k) ?: new JSONArray() }
def asList = { JSONArray a ->
    def out = []
    if (a != null) for (int i = 0; i < a.size(); i++) out.add(a.get(i))
    return out
}
def asStringList = { JSONArray a ->
    def out = []
    if (a != null) for (int i = 0; i < a.size(); i++) {
        def v = a.get(i)
        if (v != null && !String.valueOf(v).isBlank()) out.add(String.valueOf(v))
    }
    return out
}
def mapOf = { Object... kv ->
    def m = new java.util.LinkedHashMap()
    for (int i = 0; i + 1 < kv.length; i += 2) m.put(kv[i], kv[i + 1])
    return m
}
def goldenRows = { JSONObject golden ->
    def rows = []
    if (golden == null || golden.isEmpty()) return rows
    golden.keySet().each { key ->
        def v = golden.get(key)
        if ("accounts".equals(key) && v instanceof JSONArray) {
            for (int i = 0; i < v.size(); i++) {
                def acc = v.getJSONObject(i)
                rows.add("| accountNumber | ${acc.getString("accountNumber")} | balance | ${acc.get("balance")} |")
            }
        } else if (!(v instanceof JSONObject) && !(v instanceof JSONArray)) {
            rows.add("| ${key} | ${v} |")
        }
    }
    return rows
}
def appendCredentialsTable = { StringBuilder sb, JSONObject cred ->
    if (cred == null || cred.isEmpty()) return
    sb.append("    And test credentials:\n")
    sb.append("      | field | value |\n")
    cred.keySet().each { k -> sb.append("      | ").append(k).append(" | ").append(cred.get(k)).append(" |\n") }
}
def appendApiScenario = { StringBuilder sb, JSONObject hint, JSONObject golden, String mappingStatus ->
    def path = hint?.getString("path") ?: "/api/UNMAPPED"
    def method = (hint?.getString("method") ?: "POST").toUpperCase()
    def op = hint?.getString("suggestedOperation") ?: "execute"
    def cap = hint?.getString("suggestedCapability") ?: ""
    sb.append("\n  @api-mapping-${mappingStatus} @operation-").append(tag(op)).append("\n")
    sb.append("  Scenario: API ").append(op).append(" (").append(path).append(")\n")
    sb.append("    When the client sends \"").append(method).append("\" to \"").append(path).append("\"")
    def body = hint?.getJSONObject("body")
    def query = hint?.getJSONObject("query")
    def pathParams = hint?.getJSONObject("pathParams")
    if (body != null && !body.isEmpty()) {
        sb.append(" with JSON body:\n")
        body.keySet().each { k -> sb.append("      | ").append(k).append(" | ").append(body.get(k)).append(" |\n") }
    } else if (query != null && !query.isEmpty()) {
        sb.append(" with query parameters:\n")
        query.keySet().each { k -> sb.append("      | ").append(k).append(" | ").append(query.get(k)).append(" |\n") }
    } else if (pathParams != null && !pathParams.isEmpty()) {
        sb.append(" with path parameters:\n")
        pathParams.keySet().each { k -> sb.append("      | ").append(k).append(" | ").append(pathParams.get(k)).append(" |\n") }
    } else {
        sb.append("\n")
    }
    if ("gap".equals(mappingStatus) || "batch-or-none".equals(mappingStatus) || "mapping-gap".equals(mappingStatus)) {
        sb.append("    Then the scenario should be marked as mapping-gap (explicit red in JUnit)\n")
        return
    }
    sb.append("    Then the response status should be 200\n")
    def rows = goldenRows(golden)
    if (!rows.isEmpty()) {
        sb.append("    And the JSON body should match golden:\n")
        sb.append("      | field | expected |\n")
        rows.each { sb.append("      ").append(it).append("\n") }
    } else {
        sb.append("    And the JSON body should satisfy trace-derived expectations\n")
    }
    if (cap) sb.append("    # verification-runtime fallback: capabilityId \"").append(cap).append("\"\n")
}
def buildFeature = { JSONObject scenario, String featureRel, String mappingStatus ->
    def moduleId = scenario.getString("moduleId") ?: "unknown"
    def scenarioId = scenario.getString("scenarioId") ?: "scenario"
    def bc = scenario.getString("boundedContext") ?: moduleId
    def traceFile = scenario.getString("traceFile")
    def desc = scenario.getString("description") ?: ""
    def validationKind = scenario.getString("validationKind")
    def sb = new StringBuilder()
    sb.append("# language: en\n")
    sb.append("@module-").append(tag(moduleId)).append(" @boundedContext-").append(tag(bc))
    if (traceFile == null || traceFile.isEmpty()) sb.append(" @no-3270-trace")
    else sb.append(" @trace-mock")
    sb.append("\n")
    sb.append("Feature: ").append(moduleId).append(" / ").append(scenarioId).append("\n")
    sb.append("  ").append(desc.replace("\n", " ")).append("\n\n")
    sb.append("  Background:\n")
    if (traceFile) {
        sb.append("    Given the wc3270 trace fixture \"").append(traceFile).append("\"\n")
    } else {
        sb.append("    Given there is no wc3270 trace for this module\n")
    }
    sb.append("    And the legacy trace scenario id \"").append(scenarioId).append("\"\n")
    def cred = scenario.getJSONObject("credentials")
    appendCredentialsTable(sb, cred)
    def pre = scenario.getString("preconditionTrace")
    if (pre) sb.append("    And the precondition trace \"").append(pre).append("\"\n")
    def steps = jsonArray(scenario, "steps")
    if (!steps.isEmpty()) {
        sb.append("\n  @trace-steps\n  Scenario: Legacy 3270 interaction sequence from trace\n")
        asList(steps).each { st ->
            if (!(st instanceof JSONObject)) return
            def action = st.getString("action") ?: "step"
            sb.append("    Given trace step ").append(st.getIntValue("stepIndex")).append(" action \"").append(action).append("\"\n")
            def assertions = st.getJSONObject("assertions")
            if (assertions != null) {
                def contains = assertions.getJSONArray("screenContains")
                if (contains != null) {
                    for (int i = 0; i < contains.size(); i++) {
                        sb.append("    Then the trace screen should contain \"").append(contains.getString(i)).append("\"\n")
                    }
                }
            }
        }
    }
    def hints = jsonArray(scenario, "apiMappingHints")
    def golden = scenario.getJSONObject("goldenApiResponse")
    if (hints.isEmpty()) {
        sb.append("\n  @mapping-gap @explicit-fail\n")
        sb.append("  Scenario: No API mapping hint — must fail until mapped\n")
        sb.append("    Then the scenario should be marked as mapping-gap (explicit red in JUnit)\n")
    } else {
        asList(hints).each { h ->
            if (h instanceof JSONObject) appendApiScenario(sb, h, golden, mappingStatus)
        }
    }
    if ("batch-equivalence".equals(validationKind)) {
        sb.append("\n  @batch-equivalence @mapping-gap\n")
        sb.append("  Scenario: Batch jobs equivalence (no 3270 trace)\n")
        def jobs = jsonArray(scenario, "batchJobs")
        asList(jobs).each { j -> sb.append("    Given batch job \"").append(j).append("\" should have Java batch parity\n") }
        sb.append("    Then the scenario should be marked as mapping-gap until Spring Batch parity exists\n")
    }
    writeText(featureRel, sb.toString())
}

def archCapsByModule = new java.util.LinkedHashMap()
try {
    def architectureRef = catalog.getString("architectureRef") ?: "design/architecture/target-architecture-${version}.json"
    def arch = readJson(architectureRef)
    asList(jsonArray(arch, "modules")).each { m ->
        if (m instanceof JSONObject) {
            def mid = m.getString("moduleId")
            def caps = asStringList(m.getJSONArray("capabilities"))
            if (mid && !caps.isEmpty()) archCapsByModule.put(mid, caps)
        }
    }
    asList(jsonArray(arch, "capabilityAssignments")).each { ca ->
        if (ca instanceof JSONObject) {
            def mid = ca.getString("moduleId")
            def cap = ca.getString("capabilityId")
            if (mid && cap) {
                def caps = archCapsByModule.get(mid)
                if (!(caps instanceof java.util.List)) caps = []
                if (!caps.contains(cap)) caps.add(cap)
                archCapsByModule.put(mid, caps)
            }
        }
    }
} catch (Exception ignored) {}

#子模块 5.4：注册场景 + 生成 Feature 文件
#这是最核心的逻辑，分为两个函数


def registerScenario = { JSONObject scenario, String featureRel, String traceId, Object fallbackCapabilities ->
    def scenarioCaps = asStringList(scenario.getJSONArray("capabilities"))
    def fallbackCaps = []
    if (fallbackCapabilities instanceof Collection) {
        fallbackCapabilities.each { c -> if (c != null && !String.valueOf(c).isBlank()) fallbackCaps.add(String.valueOf(c)) }
    }
    def effectiveCaps = !scenarioCaps.isEmpty() ? scenarioCaps : fallbackCaps
    def hints = jsonArray(scenario, "apiMappingHints")
    if (!effectiveCaps.isEmpty()) {
        if (hints.isEmpty()) {
            def h = new JSONObject()
            h.put("suggestedCapability", effectiveCaps[0])
            h.put("suggestedOperation", "execute")
            h.put("method", "POST")
            h.put("path", "/" + effectiveCaps[0])
            hints.add(h)
        }
        asList(hints).each { h ->
            if (h instanceof JSONObject && !h.getString("suggestedCapability")) {
                h.put("suggestedCapability", effectiveCaps[0])
            }
            if (h instanceof JSONObject && (!h.getString("method"))) h.put("method", "POST")
            if (h instanceof JSONObject && (!h.getString("suggestedOperation"))) h.put("suggestedOperation", "execute")
            if (h instanceof JSONObject && (!h.getString("path") || "/api/UNMAPPED".equals(h.getString("path")))) {
                h.put("path", "/" + h.getString("suggestedCapability"))
            }
        }
    }
    def traceFile = scenario.getString("traceFile")
    def validationKind = scenario.getString("validationKind")
    def mappingStatus = "batch-equivalence".equals(validationKind) ? "batch-or-none"
            : ((hints.isEmpty() && traceFile != null && !traceFile.isEmpty()) ? "gap"
            : (hints.isEmpty() ? "batch-or-none" : "present"))
    if ("gap".equals(mappingStatus) || "batch-or-none".equals(mappingStatus)) gapCount++
    else mappedCount++
    buildFeature(scenario, featureRel, mappingStatus)
    scenarios.add(mapOf(
            "scenarioId", scenario.getString("scenarioId"),
            "moduleId", scenario.getString("moduleId"),
            "boundedContext", scenario.getString("boundedContext"),
            "capabilities", !effectiveCaps.isEmpty() ? effectiveCaps : jsonArray(scenario, "capabilities"),
            "traceId", traceId,
            "traceFile", traceFile,
            "featureFile", featureRel,
            "mappingStatus", mappingStatus,
            "apiMappingHints", hints,
            "goldenApiResponse", scenario.getJSONObject("goldenApiResponse"),
            "credentials", scenario.getJSONObject("credentials")))
    features.add(featureRel)
}


#子模块 5.5：处理 catalog 中的模块
#通俗理解：把 catalog 里登记的每个模块逐个处理


def modules = jsonArray(catalog, "modules")
asList(modules).each { m ->
    if (!(m instanceof JSONObject)) return
    def moduleId = m.getString("moduleId")
    def trace = m.getJSONObject("trace")
    if (trace == null) return
    def scenarioRel = trace.getString("scenario")
    if (scenarioRel == null || scenarioRel.isEmpty()) return
    seenScenarioRels.add(normalizeProjectRel(scenarioRel))
    def scenario = readJson(scenarioRel)
    def featureRel = "test/cucumber/by-module/${moduleId}/${scenario.getString("scenarioId") ?: "scenario"}.feature".toString()
    def fallbackCaps = archCapsByModule.get(moduleId)
    if (!(fallbackCaps instanceof java.util.List) || fallbackCaps.isEmpty()) fallbackCaps = asStringList(m.getJSONArray("capabilities"))
    registerScenario(scenario, featureRel, trace.getString("traceId"), fallbackCaps)
}

#子模块 5.6：扫描遗漏的 scenario 文件

def scenarioRootRel = catalogRel.contains("/") ? catalogRel.substring(0, catalogRel.lastIndexOf("/")) : "verification/trace-scenarios"
def scenarioRootPath = rel(scenarioRootRel)
if (Files.isDirectory(scenarioRootPath)) {
    Files.walk(scenarioRootPath)
            .filter { p -> Files.isRegularFile(p) && p.getFileName().toString().endsWith(".scenario.json") }
            .forEach { p ->
        def scenarioRel = base.relativize(p).toString().replace("\\", "/")
        if (seenScenarioRels.contains(scenarioRel)) return
        def scenario = readJson(scenarioRel)
        def moduleId = scenario.getString("moduleId") ?: p.getParent().getFileName().toString()
        def scenarioId = scenario.getString("scenarioId") ?: p.getFileName().toString().replace(".scenario.json", "")
        def featureRel = "e2e".equals(moduleId)
                ? "test/cucumber/e2e/${scenarioId}.feature".toString()
                : "test/cucumber/by-module/${moduleId}/${scenarioId}.feature".toString()
        def fallbackCaps = archCapsByModule.get(moduleId)
        registerScenario(scenario, featureRel, scenario.getString("traceId") ?: scenarioId, fallbackCaps)
        seenScenarioRels.add(scenarioRel)
    }
}
