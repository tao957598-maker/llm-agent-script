       'n_project_root','{{projectRoot}}',
       'n_ir_version','{{fn.optional(irVersion,"v1")}}',
       'n_scenario_filter','{{fn.optional(scenarioFilter,"")}}')

//# 确定性：构造扰动输入并调用 API，收集基础结果

//# LLM：基于基础结果推导变化规律，生成验证断言

  vrf_flag?value='fn.hasValue(sys_last_result,"rules")'

//# 将规则写入报告
n_metamorphic_final_report, act_func.run?script='''
import com.alibaba.fastjson2.JSON
import java.nio.charset.StandardCharsets
import java.nio.file.Files
import java.nio.file.Path
import java.time.Instant

def rootDir = com.llm.agent.common.constant.AppDirConsts.getRootDir()
def projectRoot = fn.str(fn.get("n_project_root"))?.trim()
def base = Path.of(rootDir, projectRoot).toAbsolutePath().normalize()
def rules = JSON.parseObject(fn.str(fn.get("n_metamorphic_rules")))
def reportDir = base.resolve("test/equivalence")
Files.createDirectories(reportDir)

def report = [
    schema: "cobol2java.metamorphic-validation-report",
    version: "v1",
    generatedAt: Instant.now().toString(),
    generatedBy: "agt_metamorphic_validator.c2j.validate",
    rules: rules.getJSONArray("rules"),
    baseData: JSON.parseObject(fn.str(fn.get("n_metamorphic_base_data")))
]
Files.writeString(reportDir.resolve("metamorphic-validation-report-v1.json"),
    JSON.toJSONString(report), StandardCharsets.UTF_8)

fn.put("n_metamorphic_rules_count", rules.getJSONArray("rules")?.size() ?: 0)
return agentResult(JSON.toJSONString([status:"COMPLETE"]))
'''

assign('n_metamorphic_validator_report','''# 蜕变关系验证报告
