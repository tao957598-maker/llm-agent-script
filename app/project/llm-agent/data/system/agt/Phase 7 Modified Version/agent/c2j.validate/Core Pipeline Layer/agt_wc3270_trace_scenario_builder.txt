//# 模块 4：Trace 文件发现 — n_trace_discovery_json
#通俗理解：这是"扫描老录像文件夹，列出所有需要处理的录像文件"。

#执行逻辑：

#walk：遍历 verification/wc3270/ 目录（包括子文件夹）
#filter：只挑文件名以 .trace.txt 结尾的文件
#sorted：按文件名排序（保证处理顺序一致）
#limit：最多取 20 个（由 maxTraces 参数控制，防止一次性处理太多）
#forEach：把每个文件的路径记录下来


##——————————————————————————————————————————————————————————————————————————————




n_trace_discovery_json, act_func.run?script='''
import com.alibaba.fastjson2.JSON
import com.alibaba.fastjson2.JSONArray
import com.alibaba.fastjson2.JSONObject
import java.nio.file.Files
import java.nio.file.Path

def rootDir = com.llm.agent.common.constant.AppDirConsts.getRootDir()
def projectRoot = fn.str(fn.get("n_project_root"))?.trim(); if (!projectRoot) throw new IllegalStateException("Missing required parameter: projectRoot")
def inputRoot = fn.str(fn.get("n_trace_input_root")) ?: "verification/wc3270"
def maxTraces = 20
try { maxTraces = Integer.parseInt(fn.str(fn.get("n_max_traces")) ?: "20") } catch (Exception ignored) {}
def base = Path.of(rootDir, projectRoot, inputRoot.replace("/", java.io.File.separator)).toAbsolutePath().normalize()
if (!Files.isDirectory(base)) {
    fn.put("n_trace_entries", entries)
    fn.put("n_no_trace_fixture", "true")
    return JSON.toJSONString([schema: "cobol2java.trace-discovery", status: "SKIPPED_NO_TRACE_FIXTURE",
                              reason: "missing-trace-input-directory", inputRoot: inputRoot,
                              directory: base.toString(), count: 0, catalogArtifact: null])
}
def entries = new JSONArray()
Files.walk(base).filter { p -> Files.isRegularFile(p) && p.getFileName().toString().endsWith(".trace.txt") }
        .sorted { a, b -> a.toString() <=> b.toString() }
        .limit(maxTraces)
        .forEach { p ->
    def relToProject = base.relativize(p).toString().replace("\\", "/")
    def relFull = (inputRoot.endsWith("/") ? inputRoot : inputRoot + "/") + relToProject
    if (!relFull.startsWith(inputRoot)) relFull = inputRoot + "/" + relToProject
    def item = new JSONObject()
    item.put("traceRel", inputRoot + "/" + relToProject)
    item.put("fileName", p.getFileName().toString())
    entries.add(item)
}
if (entries.isEmpty()) {
    fn.put("n_trace_entries", entries)
    fn.put("n_no_trace_fixture", "true")
    return JSON.toJSONString([schema: "cobol2java.trace-discovery", status: "SKIPPED_NO_TRACE_FIXTURE",
                              reason: "no-trace-files", inputRoot: inputRoot,
                              directory: base.toString(), count: 0, catalogArtifact: null])
}
def catalogRel = "verification/wc3270/trace-catalog-v1.json"
def catalog = [schema: "cobol2java.trace-discovery-catalog", version: "v1", projectKey: fn.str(fn.get("n_project_key")),
               inputRoot: inputRoot, count: entries.size(), traces: entries]
def projectBase = Path.of(rootDir, projectRoot).toAbsolutePath().normalize()
Files.createDirectories(projectBase.resolve("verification/wc3270"))
Files.writeString(projectBase.resolve(catalogRel), com.alibaba.fastjson2.JSON.toJSONString(catalog), StandardCharsets.UTF_8)
fn.put("n_trace_catalog_rel", catalogRel)
fn.put("n_trace_entry_count", entries.size().toString())
return JSON.toJSONString([schema: "cobol2java.trace-discovery", status: "READY", count: entries.size(), catalogArtifact: catalogRel])
'''
