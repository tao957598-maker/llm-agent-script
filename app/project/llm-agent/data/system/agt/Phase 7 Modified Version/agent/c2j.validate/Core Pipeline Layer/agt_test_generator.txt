import com.alibaba.fastjson2.JSONArray

def rootDir = com.llm.agent.common.constant.AppDirConsts.getRootDir()
def projectKey = fn.str(fn.get("n_project_key"))?.trim(); if (!projectKey) throw new IllegalStateException("Missing required parameter: projectKey")
def projectRoot = fn.str(fn.get("n_project_root"))?.trim(); if (!projectRoot) throw new IllegalStateException("Missing required parameter: projectRoot")

def manifestRel = fn.str(fn.get("n_cucumber_manifest_rel")) ?: "test/equivalence/cucumber-manifest-${version}.json"

def base = Path.of(rootDir, projectRoot).toAbsolutePath().normalize()
def javaRoot = base.resolve("implementation/java")
def rel = { String r -> base.resolve(r.replace("/", java.io.File.separator)).normalize() }
def readJson = { String r -> JSON.parseObject(Files.readString(rel(r), StandardCharsets.UTF_8)) }
def writeText = { Path p, String text ->

    Files.writeString(p, text, StandardCharsets.UTF_8)
}
def javaIdent = { String s ->
    def t = (s ?: "x").replaceAll("[^a-zA-Z0-9]", "_")
    if (t.isEmpty() || !Character.isJavaIdentifierStart(t.charAt(0))) t = "s_" + t
    return t
}
def escJava = { String s ->
    if (s == null) return ""
    return s.replace("\\", "\\\\").replace("\"", "\\\"").replace("\n", "\\n").replace("\r", "")
}
def jsonArray = { JSONObject o, String k -> o?.getJSONArray(k) ?: new JSONArray() }
def asList = { JSONArray a ->
    def out = []
    if (a != null) for (int i = 0; i < a.size(); i++) out.add(a.get(i))
    return out
}

if (!Files.isRegularFile(rel(manifestRel))) {

def gapTests = 0

def suiteTests = 0
def methods = new StringBuilder()

asList(scenarios).each { sc ->
    if (!(sc instanceof JSONObject)) return
    def scenarioId = sc.getString("scenarioId") ?: "scenario"
    def moduleId = sc.getString("moduleId") ?: "module"
    def mappingStatus = sc.getString("mappingStatus") ?: "gap"
    def featureFile = sc.getString("featureFile") ?: ""
    def traceFile = sc.getString("traceFile") ?: ""
    def methodName = javaIdent(moduleId + "_" + scenarioId)

    if ("suite".equals(mappingStatus)) {
        suiteTests++
        methods.append("""
    /** Cucumber suite: ${featureFile} */
    @Test
    @DisplayName("E2E trace chain — ${scenarioId}")
    void ${methodName}_e2eChainDocumented() {

                "Missing Cucumber feature: ${featureFile}");

""")

    }

    if ("gap".equals(mappingStatus) || "batch-or-none".equals(mappingStatus)) {
        gapTests++
        methods.append("""
    /**
     * Cucumber: ${featureFile}
     * Trace: ${traceFile.isEmpty() ? "none" : traceFile}
     */
    @Test
    @DisplayName("UNMAPPED_TRACE: ${moduleId}/${scenarioId}")
    void ${methodName}_mappingGap() {
        fail("TRACE_UNMAPPED or missing apiMappingHints — see ${featureFile}");
    }
""")
        return

    mappedTests++
    def hints = jsonArray(sc, "apiMappingHints")
    def hint = hints.isEmpty() ? new JSONObject() : hints.getJSONObject(0)
    def cap = hint.getString("suggestedCapability")
    if (cap == null || cap.isEmpty()) {
        def caps = sc.getJSONArray("capabilities")
        if (caps != null && !caps.isEmpty()) cap = caps.getString(0)
    }
    def targetPath = hint.getString("path") ?: "/api/UNMAPPED"
    def golden = sc.getJSONObject("goldenApiResponse")
    def goldenLiteral = escJava(golden != null ? JSON.toJSONString(golden) : "{}")

    if (!useVerificationRuntime || cap == null || cap.isEmpty()) {
        gapTests++
        mappedTests--
        methods.append("""
    @Test
    @DisplayName("UNMAPPED_TRACE: ${moduleId}/${scenarioId} — no capability")
    void ${methodName}_noCapability() {
        fail("No suggestedCapability — target API ${targetPath}; see ${featureFile}");
    }
""")
        return
    }

    def paramsBlock = new StringBuilder()
    def body = hint.getJSONObject("body")
    def query = hint.getJSONObject("query")
    if (body != null && !body.isEmpty()) {
        body.keySet().each { k ->
            def v = body.get(k)
            def val = v instanceof String ? "\"${escJava(v.toString())}\"" : v.toString()
            paramsBlock.append("        params.put(\"${escJava(k.toString())}\", ${val});\n")
        }
    }
    if (query != null && !query.isEmpty()) {
        query.keySet().each { k ->
            def v = query.get(k)
            def val = v instanceof String ? "\"${escJava(v.toString())}\"" : v.toString()
            paramsBlock.append("        params.put(\"${escJava(k.toString())}\", ${val});\n")
        }
    }
    def pathParams = hint.getJSONObject("pathParams")
    if (pathParams != null && !pathParams.isEmpty()) {
        pathParams.keySet().each { k ->
            def v = pathParams.get(k)
            def key = "accountNumber".equals(k.toString()) ? "accountId" : k.toString()
            def val = v instanceof String ? "\"${escJava(v.toString())}\"" : v.toString()
            paramsBlock.append("        params.put(\"${escJava(key)}\", ${val});\n")
        }
    }
    def cred = sc.getJSONObject("credentials")
    if (cred != null && !cred.containsKey("customerId") && cred.getString("userId") != null) {
        paramsBlock.append("        params.put(\"customerId\", \"${escJava(cred.getString("userId"))}\");\n")
    }
    if (paramsBlock.length() == 0) {
        paramsBlock.append("        params.put(\"userId\", \"B0001\");\n")
        paramsBlock.append("        params.put(\"customerId\", \"B0001\");\n")
    }

    methods.append("""
    /**
     * Cucumber: ${featureFile}
     * Target API (future): ${targetPath}
     * P0: POST /verification/runtime/execute capabilityId=${cap}
     */
    @Test
    @DisplayName("${moduleId}/${scenarioId} — trace API equivalence")
    void ${methodName}_apiEquivalence() throws Exception {
        java.util.Map<String, Object> params = new java.util.LinkedHashMap<>();
${paramsBlock}
        String payload = objectMapper.writeValueAsString(java.util.Map.of(
                "capabilityId", "${escJava(cap)}",
                "parameters", params));
        var result = mockMvc.perform(post("/verification/runtime/execute")
                        .contentType(MediaType.APPLICATION_JSON)
                        .content(payload))
                .andExpect(status().isOk())
                .andExpect(jsonPath("$.capabilityId").value("${escJava(cap)}"))
                .andReturn();
        String responseBody = result.getResponse().getContentAsString();
        assertNotNull(responseBody);
        assertFalse(responseBody.contains("ERROR:"), () -> "Capability execution failed: " + responseBody);
        assertGolden_${methodName}(responseBody);
    }

    private void assertGolden_${methodName}(String responseBody) throws Exception {
        JSONObject golden = JSON.parseObject("${goldenLiteral}");
        if (golden == null || golden.isEmpty()) {
            return;
        }
        JSONObject envelope = JSON.parseObject(responseBody);
        String summary = envelope != null ? envelope.getString("resultSummary") : null;
        if (golden.getBooleanValue("authenticated")) {
            assertNotNull(summary);
            assertFalse(summary.startsWith("ERROR"), () -> summary);
        }
        JSONArray accounts = golden.getJSONArray("accounts");
        if (accounts != null && !accounts.isEmpty()) {
            assertNotNull(summary);
            accounts.forEach { acc ->
                if (acc instanceof JSONObject) {
                    String acctNo = acc.getString("accountNumber");

                        assertTrue(summary.contains(acctNo) || responseBody.contains(acctNo),
                                () -> "Expected account " + acctNo + " in response");
                    }
                }
            };
        }
    }
""")
}
