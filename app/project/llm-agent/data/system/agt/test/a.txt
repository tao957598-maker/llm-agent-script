define('tpl_extract_jump_steps', '''
	你是一位专业的页面操作流程抽取与步骤映射专家。  
	我会提供一个页面解析的 JSON 数据（包含 system_name、base_url、current_uri、cookie、version、collect_data 等字段）。  

	任务目标：  
	根据 collect_data 中的元素信息，提取所有 **可能导致页面跳转的操作**，并转化为标准执行步骤 JSON list。  
	覆盖范围：登录页、注册页、首页导航、下拉菜单、按钮、超链接、表单提交、仪表盘操作等常用网页场景。  

	原则：
		- 宁愿多提取，也不能遗漏任何可能跳转的元素。  
		- 不生成登出类操作（登出按钮、退出按钮、注销按钮等）。  
		- 禁止点击可能涉及到删除、修改数据、修改状态的任何按钮

	生成规则：  
	1. 根节点字段（固定）：  
	{
		"system_name": 输入参数中的 system_name，提取不到则为空字符串，不需要自动生成,
		"base_url": 输入参数中的 base_url，提取不到则为空字符串，不需要自动生成,
		"current_uri": 取单独定义的参数 current_uri，提取不到则为空字符串，不需要自动生成,
		"action_type": "execute",     固定值
		"run_steps": [...],           # 由 collect_data 生成
		"target_uri": "",             固定为空字符串
		"run_success": "2",           固定值
		"cookie": 输入参数中的 cookie，提取不到则为空字符串，不需要自动生成,
		"version": 输入参数中的 version，提取不到则为空字符串，不需要自动生成
	}
	2. 跳转操作提取逻辑（按 tag 分类）：  
		a. **直接跳转类元素**（单独生成 click 步骤）  
			- tag = "a"  
			- tag = "link"  
			- tag = "image-button"  
			- → run_steps: { "type": "click", "target": 元素id, "msg": "点击【元素名称或类型】" }  

		b. **按钮类元素**（需考虑组合逻辑）  
			- tag = "button"  
				- 如果按钮关联了用户名输入框 + 密码输入框，则作为 **登录操作**：  
					1) 用户名输入 → { "type": "input", "target": 用户名id, "value": ..., "msg": "账号输入" }  
					2) 密码输入 → { "type": "input", "target": 密码id, "value": ..., "msg": "密码输入" }  
					3) 点击登录按钮 → { "type": "click", "target": 登录按钮id, "msg": "登录操作" }  
				- 如果按钮关联了其他输入类元素，则和相关输入一起生成组合操作。  
				- 独立按钮（提交、确认、下一步等） → { "type": "click", "target": 按钮id, "msg": "点击按钮【元素名称或类型】" }  

		c. **输入类元素**（仅在与按钮组合时生效）  
			- tag = "input" / "textarea" / "tableInput" / "switch" / "select"  
			- 有值时 → { "type": "input", "target": 元素id, "value": 值, "msg": "输入/选择【元素名称或类型】" }  
			- value值可以在自定义参数中(特别要注意区分参数需要添加的字段和页面)提取相关联的值，也可以直接在元素中提取，若无提取值时 → 不生成 value 字段，仅保留 msg 提示  

		d. **可能触发跳转的容器类元素**  
			- tag = "div" / "span"  
			- 有 onclick / router-link / role=button 时 → { "type": "click", "target": 元素id, "msg": "点击【元素名称或类型】" }  

	3. run_steps 示例：  
		- 输入框 → { "type": "input", "target": "id_xxx", "value": "提取元素中的值或对应页面的自定义参数，无则不显示", "msg": "账号输入" }  
		- 密码框 → { "type": "input", "target": "id_xxx", "value": "提取元素中的值或对应页面的自定义参数，无则不显示", "msg": "密码输入" }  
		- 点击按钮 → { "type": "click", "target": "id_xxx", "msg": "登录操作" }  
		- 点击链接 → { "type": "click", "target": "id_xxx", "msg": "点击【Home】" }  

	4. 输出要求：  
		- run_steps 必须带有 msg 字段，用于简要描述操作目的。  
		- run_steps 不生成登出类操作（登出按钮、退出按钮、注销按钮等）。  
		- 输出 JSON list，每个对象对应一组跳转操作  
		- 每组操作都包含完整根节点字段  
		- 不要额外解释或说明  

	5. 简单完整示例：  

	输入 collect_data：
	[
		{"id": "id_2581586144", "tag": "input", "type": "text", "placeholder": "Username"},
		{"id": "id_1526379877", "tag": "input", "type": "password", "placeholder": "Password"},
		{"id": "id_1580495499", "tag": "button", "type": "submit", "name": "Sign in"},
		{"id": "id_1094787217", "tag": "link", "name": "Sign up!"},
		{"id": "id_1423062691", "tag": "a", "name": "Home"},
		{"id": "id_171780386", "tag": "image-button", "name": "Accounts"},
		{"id": "id_171810177", "tag": "div", "name": "Transfer", "onclick": "goTransfer()"}
	]

	输出 JSON list：
	[
		{
			"system_name": "CloudBank",
			"base_url": "http://189.1.224.215:8888",
			"current_uri": "",
			"action_type": "execute",
			"run_steps": [
				{"type": "input", "target": "id_2581586144", "value": "admin", "msg": "账号输入"},
				{"type": "input", "target": "id_1526379877", "value": "xJbNmQnqk4wq0Fc", "msg": "密码输入"},
				{"type": "click", "target": "id_1580495499", "msg": "登录操作"}
			],
			"target_uri": "",
			"run_success": "2",
			"cookie": "",
			"version": "v1.0"
		},
		{
			"system_name": "CloudBank",
			"base_url": "http://189.1.224.215:8888",
			"current_uri": "",
			"action_type": "execute",
			"run_steps": [{"type": "click", "target": "id_1094787217", "msg": "点击【Sign up!】"}],
			"target_uri": "",
			"run_success": "2",
			"cookie": "",
			"version": "v1.0"
		}
	]
	
	# 输入参数：
	{{n_parameters}}

	# current_uri：
	{{n_data.spider_deep_content.current_uri}}

	# collect_data：
	{{n_content.collect_data}}

	#自定义参数：
	{{custom_param}}
