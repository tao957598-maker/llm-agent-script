act_func?script='''
    def safeDocId = ""
    try {
        safeDocId = fn.optional(kbSpaceFileId, '')
    } catch (Throwable ignore) {
        try { safeDocId = (kbSpaceFileId != null ? kbSpaceFileId.toString() : "") } catch (Throwable ignore2) { safeDocId = "" }
    }
    def ver = ""
    try { ver = (version != null ? version.toString() : "") } catch (Throwable ignore) { ver = "" }

    def nonEmptyData = { mm ->
        if (mm == null || !(mm instanceof Map)) return false
        def x = mm.get("data")
        if (x instanceof List) return (x as List).size() > 0
        if (x instanceof Map) return true
        return false
    }
    def normalizeRoot = { raw ->
        if (raw == null) return null
        if (raw instanceof Map) return raw as Map
        if (raw instanceof List && (raw as List).size() > 0) {
            def z = (raw as List)[0]
            if (z instanceof Map) return z as Map
        }
        return null
    }
    def findMapWithData
    findMapWithData = { obj, depth ->
        if (depth > 8 || obj == null) return null
        if (obj instanceof Map) {
            def mm = obj as Map
            if (nonEmptyData(mm)) return mm
            for (def e : mm.entrySet()) {
                def r = findMapWithData(e.value, depth + 1)
                if (r != null) return r
            }
        }
        if (obj instanceof List) {
            for (def el : (obj as List)) {
                def r = findMapWithData(el, depth + 1)
                if (r != null) return r
            }
        }
        return null
    }
    def unwrapPayload = { raw ->
        def m = normalizeRoot(raw)
        if (m == null) return null
        if (nonEmptyData(m)) return m
        def inner = m
        if (inner.get("value") instanceof Map) inner = inner.get("value") as Map
        if (nonEmptyData(inner)) return inner
        if (inner.get("final_design_result") instanceof Map) inner = inner.get("final_design_result") as Map
        if (nonEmptyData(inner)) return inner
        if (inner.get("final_result") instanceof Map) inner = inner.get("final_result") as Map
        if (nonEmptyData(inner)) return inner
        if (inner.get("design_input_data") instanceof Map) inner = inner.get("design_input_data") as Map
        if (nonEmptyData(inner)) return inner
        def deep = findMapWithData(raw, 0)
        return deep != null ? deep : inner
    }
    def dataRows = { m ->
        if (m == null) return []
        def data = m.get("data")
        if (data instanceof List) return data as List
        if (data instanceof Map) return [data]
        return []
    }
    def tdl = fn.get('test_design_list')
    if (tdl == null || !(tdl instanceof List)) tdl = []

    def out = []
    tdl.each { raw ->
        def d = unwrapPayload(raw)
        if (d == null) return
        def rows = dataRows(d)
        rows.each { row ->
            if (!(row instanceof Map)) return
            def de = (row.get("design_element") ?: row.get("DesignElement") ?: "").toString()
            def items = row.get("test_content")
            if (items == null) items = row.get("TestContent")
            items = items instanceof List ? (items as List) : []
            items.each { tcRaw ->
                def tc = tcRaw
                if (tcRaw instanceof List && tcRaw.size() >= 2) {
                    tc = [sub_category: tcRaw[0] != null ? tcRaw[0].toString() : "", sub_id: tcRaw[1] != null ? tcRaw[1].toString() : ""]
                }
                if (!(tc instanceof Map)) return
                def sid = (tc.get("sub_id") ?: tc.get("Sub_id") ?: "").toString().trim()
                if (!sid) return
                out.add([
                    design_element: de,
                    sub_category: (tc.get("sub_category") ?: tc.get("Sub_category") ?: "").toString(),
                    sub_id: sid,
                    platform: d.get("platform"),
                    main_module_name: d.get("main_module_name"),
                    main_module_source: d.get("main_module_source"),
                    submodule_name: d.get("submodule_name"),
                    submodule_description: d.get("submodule_description"),
                    submodule_source: d.get("submodule_source"),
                    documentId: safeDocId,
                    version: ver
                ])
            }
        }
    }
    def seenKeys = new LinkedHashSet()
    def deduped = []
    out.each { m ->
        if (m == null || !(m instanceof Map)) return
        def k = [
            (m.get("submodule_name") ?: "").toString(),
            (m.get("design_element") ?: "").toString(),
            (m.get("sub_id") ?: "").toString()
        ].join("||")
        if (!seenKeys.contains(k)) {
            seenKeys.add(k)
            deduped.add(m)
        }
    }
    fn.put('result_list_design', deduped)
    fn.put('design_flattened_data', deduped)
    fn.put('design_flattened_data_one', deduped)
'''
