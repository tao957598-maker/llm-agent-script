assign('test_case_input_data', '{{input_data}}')
when('fn.hasValue(input_data)', {
  when('fn.hasValue(input_data.main_module_source)', {
    assign('design_pages', '{{input_data.main_module_source}}')
  }, {
    assign('design_pages', '{{input_data.submodule_source}}')
  })
  // 第三步性能/稳定性优化：不再调用 agt_get_page_data（会额外 LLM 解析并在 Web 侧更容易触发过载）
  // 直接从 "Page3, Page4" 提取数字并组装为 SQL IN 列表，例如 "(3,4)"。
  n_pages_sql,act_func?script='''
    def s = (design_pages ?: "").toString()
    def nums = s.findAll(/\\d+/)
    nums = (nums == null) ? [] : nums.unique()
    return "(" + nums.join(",") + ")"
  '''
  // 查询kb_okp_page_data表，按page字段升序排序，筛选system_name='OKP'且page等于提取的数字的数据
  n_pages_data, act_kb.selectBy?sql='''
    select 
    page,
    CASE
        WHEN analysis_content IS NULL OR analysis_content = '' THEN content
        ELSE analysis_content
    END AS content
    from kbs_doc_page_parse
    where documentId="{{documentId}}" and page in {{n_pages_sql}} order by page asc'''

  tpl_gen_test_case
  // 获取test case
  assign('processed_data', '{{tpl_gen_test_case.data}}')
  // 先写入，后续去重从 test_case_input_data 读取，避免 Groovy 脚本作用域丢失
  act_func?script="test_case_input_data.put('data', processed_data)"
  // 生成后去重（按 title + steps + expected_result）
  act_func?script='''
      def processed_data_local = test_case_input_data.get('data')
      if (processed_data_local == null) return
      def seen = new LinkedHashSet()
      def deduped = []
      processed_data_local.each { tc ->
          if (tc == null) return
          def t = (tc["title"] ?: "").toString().trim().replaceAll(/\\s+/, " ")
          def s = (tc["steps"] ?: "").toString().trim().replaceAll(/\\s+/, " ")
          def e = (tc["expected_result"] ?: "").toString().trim().replaceAll(/\\s+/, " ")
          def key = (t + "||" + s + "||" + e).toLowerCase()
          if (key == "||||") return
          if (seen.contains(key)) return
          seen.add(key)
          deduped.add(tc)
      }
      // 剔除「未注册/无功能权限」类负向用例（与第三步提示词一致，双保险）
      def dropUnregisteredNegative = { tc ->
          def b = ((tc["title"] ?: "") + " " + (tc["steps"] ?: "") + " " + (tc["expected_result"] ?: "")).toLowerCase()
          // 原则约束：第三步不输出“未注册/无权限”相关用例（无论是否写成 blocked/denied）
          if (b.contains("unregistered") || b.contains("not registered") || b.contains("feature not registered")
              || b.contains("no permission") || b.contains("without permission")
              || b.contains("未注册") || b.contains("无权限") || b.contains("無權限")) return true
          return false
      }
      deduped = deduped.findAll { tc -> !dropUnregisteredNegative(tc) }
      test_case_input_data.put('data', deduped)
  '''
 })
// final_test_case_result
assign("final_test_case_result",'{{test_case_input_data}}')

当且仅当**需要多语言**时：
- **必须拆成 3 条独立用例**，title 末尾分别为 **`[EN]`**、**`[TC]`**、**`[SC]`**（三条成组，缺一不可）。

## 禁止生成的用例类型（本批次一律不要输出）
- **未注册 / 无功能权限 / 负向拦截**：不要生成“unregistered profile / NOT registered / feature not registered / access blocked / navigation denied / button disabled because not registered”等场景的用例。
- 不要生成仅验证“无权限用户被拦截、按钮灰化不可点、禁止进入页面”的用例。
- 正向用例中**默认使用已注册且有权**的用户前提即可，无需单独写未注册类用例。

## 必测补充场景（在遵守上文禁止项的前提下）
1) **三语言/i18n**：仅当本设计点需要多语言 UI 验证时，按「多语言 / 标题后缀」规则输出 **恰好 3 条**（[EN][TC][SC]）。
2) **按钮跳转/导航**：若需求中出现 Upload / Download / Next / Back / 导航到某步骤，生成**正向**跳转/下载类用例（已注册用户）；Download 需验证下载成功、文件名/格式等。

## 轻量化表达（软约束，不得牺牲覆盖）
- 在不遗漏关键风险前提下，避免同义改写的“平行用例”；同一测试意图只保留表达最清晰的一条。

  - 默认**尽量 1～2 条**；
