// file name
n_file_name, act_kb.selectBy?sql='select  knowledge_name  from kb_spaces_folder_knowledge where id="{{kbSpaceFileId}}" '
assign("file_name", '{{n_file_name[0].knowledge_name}}')

// 初始化知识空间上下文与目录路径变量
assign("kb_space_context", "NaN")
assign("kb_dir_path", "")
// 判断是否有目录ID传入
when('fn.hasValue(kbSpaceId)', {
	n_filepath, act_kb.selectBy?sql='select  full_path  from kb_spaces_folder where id={{kbSpaceId}} '
	assign("filepath", '{{n_filepath[0].full_path}}')
	assign("kb_space_context", "NaN")
	// 获取目录下已有文件列表用作LLM上下文参考
	n_kb_dir_info,act_artifact.list?filepath='{{filepath}}'&maxDepth=1&maxResults=100
	// 将文件列表赋值给上下文变量
	assign("kb_space_context", "{{n_kb_dir_info}}")
	// 将提取的路径赋值给目录路径变量
	assign("kb_dir_path", "{{filepath}}")
})

assign("chat_history", "Nan")
when('is_read_chat_history==true', {
n_main_chat_session_history, act_memory.retrieve?var=main_chat_session_history&global=true
assign("chat_history", "{{n_main_chat_session_history}}")
})

// 生成文件名称与内容
n_gen_result,tpl_gen_file,psr_json

// 构建完整文件写入路径（目录路径 + 文件名）
n_final_path,act_func?script='''
	def dirPath = fn.str(kb_dir_path) ?: ""
	def fileName = fn.str(n_gen_result?.fileName) ?: "generated_file.txt"
	def fullPath = ""
	if (dirPath) {
		fullPath = dirPath + "/" + fileName
	} else {
		fullPath = fileName
	}
	return fullPath
'''

// 将生成的文件内容写入知识空间并同步索引
n_result,act_func.write?filename='{{n_final_path}}'&data='{{n_gen_result.fileContent}}'&syncKbs=true

// 组装最终返回结果
assign("finalResult", '{"success":true,"fileName":"{{n_gen_result.fileName}}","filePath":"{{n_final_path}}","fileType":"{{n_gen_result.fileType}}"}')


// 异常处理：生成或写入失败时记录错误信息
on_failure({
	// 记录失败原因
	assign("finalResult", '{"success":false,"error":"{{sys_failure_task.result}}"}')
})
