// n_write_filename: 从文件路径中提取纯文件名（不含路径部分）
n_write_filename, act_func.run?script=' return screenshot_filename.substring(screenshot_filename.lastIndexOf(File.separator) + 1)' 

// write_path: 构建上传时使用的存储路径，将文件放入upload目录下
