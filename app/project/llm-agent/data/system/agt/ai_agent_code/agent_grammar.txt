## AgentFlow 基础语法（基于简洁的groovy语法）
// 用途：编写Agent流程脚本，由引擎解析执行。脚本经预处理为执行图(Plan)再调度。
// 【生成 Agent 前必读】agent_grammar_paradigm.txt（范式、决策树、反模式、内嵌范例）。本文件为 API 手册，生成入口须先读 paradigm 再查本文件。
// 【范式·摘要】Agent编程=LLM推理：tpl/psr/vrf+工具+act_plan.run；when/foreach仅流程。不建议act_func?script代推理/分支。script用于：确定性逻辑、字符/格式处理、schema 合并。
// AgentFlow 是 Groovy 风格的受限 DSL，不是通用 Groovy；闭包内仅允许 DSL 语句，建议确定性 Groovy/JS 用 act_func.run。
// 续行：单条语句可用 \ 做换行续写，\ 结尾的行会与下一行拼接为一条语句。
// 变量来源：脚本内变量可自定义或来自顶部Agent定义块 /*{ "parameter":[...] }*/，如 userInput。
// 变量语义：变量是运行时值而非文本替换宏；DSL不声明类型，解析器在写入、能力调用、模板渲染等边界优先保留结构化语义并自动适配类型，业务信息缺失由agent补充。
// 术语：Plan=一条 expression 或等价链；n_xxx=plan name（步骤名/结果变量名），可用 {{n_xxx}} 引用步骤结果，也可作为步骤位置标记用于 act_flow.go 跳转；tpl_xxx=define 定义的 LLM 模板；sys_=系统内置变量。
// 参数解析：callpath 参数值默认解析 {{xxx}}；统一的语义参数（_ 前缀，引擎消费）：_plainProperty=p1,p2 指定字段保持字面量；_suppressFailure=true 失败不抛异常、封装 success 供 when 判断；_parameters={{var}} 或内联 JSON 可整包展开为子 Agent 顶层形参（等价 act_plan.run 的 parameters=JSON）。
// 不同类型的引号可内嵌，三引号有2种：'''和"""，可用于多行字符串。不支持转义符

### 变量与内联表达式
// {{xxx}} 等同于 FreeMarker 的 ${}，是内联表达式语法，平台用 {{}} 替代了 ${}
// {{}} 内是 FreeMarker 表达式语法，支持简短逻辑判断与内置函数
// {{}} 中可直接使用 fn.map()、fn.json()、fn.list()、fn.str()、fn.equals、fn.startWith()、fn.contains()、fn.bool()、fn.hasValue等工具方法(参考agent_grammar_fn.txt）
// {{}} 中禁止引入 fastjson/groovy.json 等外部 JSON 包，请使用 FunctionUtil 封装的方法
// FreeMarker 不支持 ?.、?:、Java 方法调用（如 .equals()/.trim()/.toUpperCase()）
// 字符串处理请使用 ?trim、?upper_case、?contains 等内置函数。
// 无参数时省略括号（推荐无参省略，如 ?trim 等价于 ?trim()），有参数时保留括号（如 ?contains("xxx")）
// 短逻辑示例

assign('n_upper', '{{(n_name!"")?upper_case}}')          // 字符串转大写
assign('n_has_value', '{{n_data??}}')                    // 判断变量是否存在
assign('n_trim_equals', '{{(n_status!"")?trim == "SUCCESS"}}')  // 去空格后精确匹配
assign('n_default', '{{n_value!"default"}}')             // 空值默认值
assign('n_size', '{{(n_list![])?size}}')                 // 集合大小
assign('n_contains', '{{(n_text!"")?contains("error")}}') 或 assign('n_contains', '{{fn.contains(n_text,"error")}}') // 子串判断（注意：精确匹配请使用 ==）
assign('n_math', '{{((n_price!0) * (n_quantity!0))}}')   // 数学运算
assign('n_ternary', '{{n_flag?string("yes", "no")}}')    // 三元判断（布尔转字符串）

// act_func?script 示例：确定性校验(非业务推理)；groovy语法，正则需转义，少用正则
n_result, act_func?script='''
  def output = fn.str(n_input) ?: ""
  def valid = output.trim().equals("VALID")
  fn.put('n_is_valid', valid)
  return valid
'''

### 引号嵌套规则
// 多层引号嵌套时，禁止嵌套相同的引号：
//   单引号 '...' 里面不能再放单引号 '
//   双引号 "..." 里面不能再放双引号 "
// 最佳实践：使用**内双外单**组合
// ✅ 正确：内双外单
act_memory.assign?var=n_result&value='{{n_cmd == ""? n_cmd: n_cmd2}}'
assign('n_msg', '结果：{{n_status.equals("OK") ? "成功" : "失败"}}')

// ❌ 错误：相同引号嵌套
// act_memory.assign?var=n_result&value="{{n_cmd == ''? n_cmd: n_cmd2}}"  // 报错
// assign('n_msg', "结果：{{n_status.equals('OK') ? '成功' : '失败'}}")   // 报错

// 变量与数据结构
// assign：首个左值既是写入 memory 的变量名，也是这条 assign 表达式在图中的 plan 名（同一语句多对 var,value 时仅首左值命名一条 plan）。
//   历史：fn.queryMessages=该plan原始消息；fn.queryRelevantHistory=按问题取相关历史包；fn.queryInstanceVarValues=同request各实例变量快照。
// value按JSON、字符串自动解析
assign('var', 'num')
// value为number/boolean时：用 fn.raw 保持类型写入，否则按字符串解析。
assign('num', '{{fn.raw(100)}}')
// 引用变量
assign('referenced_value', '{{var}}')
// 使用%{}%语法支持嵌套模板（fn为内置函数，freemarker表达式解析，参见：agent_grammar_fn.txt）
assign('nested_template', "{{fn.when('condition', '今天是2024-10-03', '今天是%{fn.today()}%')}}")
// 列表：结构化JSON自动解析为List
assign('list', '["a","b"]')
// JSON/Map对象：结构化JSON自动解析为Map
assign('json', '{"name":"jack"}')
// 链式定义(变量1,值1,变量2,值2)
assign("product", "oa",
		"requirement", '''无
		''',
		"requirement_score", ">=7")
// 给对象的属性赋值（map 语法）
assign("user.name", "user1")
// 计划表达式：URL格式，带模板的callpath。rag(检索)→tpl(LLM模板)→psr(解析)→vrf(校验执行)→act(直接执行)，psr或vrf环节出错，最多重试3次，提升LLM的成功率；可结合sys_hint进一步增强成功率
expression('rag_xxx', 'tpl_xxx', 'psr_xxx', 'vrf_xxx', 'act_xxx', 'act_xxx2')
// 简写：一行仅含 rag_/tpl_/psr_/vrf_/act_/n_ 逗号列表时可省略 expression(...)
rag_xxx, tpl_xxx, psr_xxx, vrf_xxx, act_xxx, act_xxx2
// 通过 memory 工具赋值（简写语法）。查询参数：var 目标变量名；value 待写入模板（可 {{xxx}} 引用变量并自动适配类型），省略则从最近一次步骤结果读取；type 可选，用于前端的展示类型（CodeBlockEnum.code，如 json、md、mmd、html、link、plan_update）；global 可选，是否写入全局记忆，默认 false
act_memory.assign?var=result&value=xxx
act_memory.assign?var=md_data&value={{tpl_md_output}}&type=md&global=true
// 拼接值val1,val2...为list，给到变量var
append('var', 'val1', 'val2'...)
// 记录日志
log('log info {{var}}')

rag_kb.select?name=kb_md_example, tpl_md_output, psr_md, vrf_flag?value='{{fn.when(fn.hasValue(sys_last_result,"optional_sub_property_name"),"true","false")}}',\

### 控制流
when('条件', { ... })
elsewhen('条件', { ... })
foreach ('optional_var_name', 'list|condition', { ... })
// 异常处理
on_failure({ ... })
// 条件判断，{{}}中为freemarker表达式. 支持 %{var}% 语法，将变量展开再做条件判断。
when('{{product_id == compared_product_id}}', {
	assign("score", "9")
})
// when支持if-else（省略{{}}不推荐）
when('list?size gt 0',{
	assign("score", "9")
},{
	assign("score", "10")
})
elsewhen('{{condition}}', {
	assign("score", "11")
})
// elsewhen支持else（无参数时）
elsewhen({
	assign("fallback_score", "12")
})
// for 循环，基于普通list数据，或task list数据. n_plan_tasks为json结构, 包含属性：index, item, size
foreach('n_plan_tasks', 'llm_data', {
    // index 下标
    assign('loop_task_index', '{{n_plan_tasks.index}}')
    // 下标对应内容
    assign('loop_task_item', '{{n_plan_tasks.item}}')
})
// 结合条件构建ReAct思维链 (tpl_react可使用fn.toolset(...)的工具描述，或自定义JSON描述)
foreach('n_plan_analysis', 'fn.get("tpl_react.name")! != "finish" && n_plan_analysis.index < 5', {
	// 思考和执行。run可无参，等价为code={{tpl_react.code}}&parameters={{tpl_react.parameters}}&answer={{tpl_react.answer}}
	tpl_react, act_plan.run
})
// 获取汇总的非空List结果
assign('n_summary', '{{n_plan_analysis.validSummary}}')

### 函数工具（act_func）
// 见【范式·重点】；业务推理/工具选择不建议写script。更适合：SQL/临时表、fn.put/get、确定性数据处理。
// 执行groovy脚本；可读agent变量；fn（fn.put 写回 agent，fn.run(callpath) 嵌套执行agent脚本）与 sql。sql 命名/语义贴近 JDBC PreparedStatement：占位符一律 ?、按参数顺序绑定；禁止多语句分号批处理。
//   sql.query(sql, ...params) → List<Map<列名, 值>>，行类型为有序 Map。仅只读 SELECT 或 WITH...SELECT；经知识库结构化算子或直连校验。
//   sql.update(sql, ...params) → int 影响行数。写操作仅允许作用域内已登记的临时表（本请求 createTempTable 等产出的 temp_* 等规则由 TempTableRegistry 约束），否则拒绝/记日志。
//   sql.createTempTable([logicalName,] sourceSql, ...params) → String 物理表名。sourceSql 为只读查询（CREATE TABLE t AS <select>）或（若配置允许）合法 CREATE TABLE 文本经改写挂到物理名；缺省 logicalName 时内部自动名。
n_result, act_func.run?script='''
	list.add('val3')
	fn.put('list2', new java.util.ArrayList<>())	// 将list2写入到agent
	def verified_sql = fn.run('act_plan.check?type=sql&script="select * from some_table where id=?"').result
	def t = sql.createTempTable('new_table', verified_sql, 1)
	fn.get('list2')
'''
// 支持%{}%语法，展开变量后执行，基于vrf激活最多自动重试3次
n_result, tpl_gen_script, psr_js, vrf_func.run?script='%{sys_last_result}%'
// 调用javascript脚本； 能力与groovy相同
act_func.run?js="""
	console.log(n_result);
"""

### 统一资源管理工具（act_artifact），对文件、db 等进行统一操作
// filepath：先按英文逗号拆成多段，再逐段处理。每一段 trim 后若整段可解析为 long，则对该段调用 searchKbSpaceName 还原为工作区路径；非 long 段原样当作路径。long 段默认按知识文件 id；仅 search/searchRows/list 可传 kbIdsAreFolderIds=true，使 long 段按目录/空间 id 解析。
// 例外：write、append、replaceLines 的 filepath 仅允许单段（不得英文逗号多路径）。syncState 可多段，多段时返回 List。
// SQL 库表挂载 filepath：system/db/sql/{schema}/{tableName}，{…} 为占位；
// write/append/delete/syncState/triggerSync/replaceLines/readRows 等仅文件语义：其中出现的 long 段一律按文件 id 解析（无 kbIdsAreFolderIds 参数）。可选 maxLength（默认 10000）：限制返回文本长度，更长需显式传入
// read：多段逗号拼接。文件=读正文；工作区文件不存在时返回空串（缺省值用 when+assign）。工作区目录且=第一层文件逐个读并拼接（目录展开与多段均带「=== FILE: path ===」头）。表/preferRepresentation=table→表 JSON。
// format=auto|markdown|map|matrix|matrices|raw（auto：xlsx→md、表→map）；format=raw：直读文件，文本=UTF-8，二进制=data URL(base64)。段尾:起始-结束行简写；全局start/endLine则禁用目录展开。maxLength=内容最大总长
n_text_content, act_artifact.read?filepath='agt/agt_core_test.txt'
n_url_content, act_artifact.read?filepath='https://www.baidu.com?a=1'
n_mixed_read, act_artifact.read?filepath='agt/,agt/agt_core_test.txt'
n_lines, act_artifact.read?filepath='agt/agt_core_test.txt'&startLine=1&endLine=10
// filepath 每段末尾可写行范围简写：: + 起始行-结束行（1-based），等价于对该段单独传 startLine/endLine
n_lines, act_artifact.read?filepath='agt/agt_core_test.txt:1-10'
// 结构化读与查：返回 List<Map>（需具表表示）；可选 column、where(JSON)、maxSize、order；多路径时按顺序合并行列表
n_rows, act_artifact.readRows?filepath='system/db/sql/schema_xxx/table_xxx'&column='id,user_id'&maxSize=1&where='{"id":"1"}'
// 统一检索：FileSearchResult 列表；目录命中=真实路径+行；表/向量/图多在 lineContent 的 JSON。preferRepresentation 空=自动短路：图(cypher)→表→向量→目录；table|vector=只走该分支。linesBefore/After=上下文行数
// query：目录=按行匹配，关键词或正则，配 ignoreCase；向量=自然语言；表=主要靠 where
n_search_result, act_artifact.search?query='keyword'&filepath='agt/'&maxResults=50&ignoreCase=true&linesBefore=2&linesAfter=2
// searchRows：返回 Map 行（表/向量/图）；目录无行。preferRepresentation/多路径与 search 同
n_row_hits, act_artifact.searchRows?filepath='agt/doc.krd.json'&query='需求描述'&preferRepresentation=vector&maxResults=20&isRerank=true&bm25Weight=0.3
// 写：data 为文件全文；或 unified diff（---/+++@@，会先读原文件再打补丁）；或 <<<<<<< SEARCH/=======/>>>>>>> REPLACE 块（同上）；或为 List<Map> JSON/对象（表 import 或 update 时配合 where、columnValues）；
act_artifact.write?filepath='{{file_path}}'&data='{{file_content}}'
// 追加：语义同 write 的 data；
act_artifact.append?filepath='{{file_path}}'&data='{{more_content}}'
// 按行替换（1-based；常与 search 命中行号联用）；data 为替换用内容
act_artifact.replaceLines?filepath='agt/xxx.txt'&startLine=3&endLine=3&data='new line content'
// list：列路径（类ls）；目录=子项；单文件存在则仅该项；默认路径不显式含.index则过滤.index侧车；多段合并去重。maxDepth/maxResults/maxLength
n_file_list, act_artifact.list?filepath='agt/'&maxDepth=1&maxResults=500
// describe：单JSON；files[]=artifact+linkedQualifiedTableKeys；tableDefinitions[]=qualified去重；多段/目录maxDepth聚合。system/db/sql或空filepath+tableKey仅tableDefinitions。tableKey过滤绑定
n_artifact_desc, act_artifact.describe?filepath='reports/weekly.md'
// 删除；可多id/路径批量
act_artifact.delete?filepath='agt/obsolete.txt'
// 判断资源是否存在：文件 or system/db/sql/{schema}/{table}
n_exists, act_artifact.exists?filepath='agt/agt_core_test.txt'
// 索引流水线进度（needOcr/ocrDone/textChunkDone/structureChunkDone/textVectorDone/structureDone/dbPersistDone 等）；filepath 可多段，多段返回数组
n_file_index_status, act_artifact.syncState?filepath='agt/xxx.txt'
// 触发索引/向量化等；底层默认 vector；可多 id/路径批量。forceSync=true 时先删index再全量重建；ensureKrd=true 时自当前目录向父级查 KRD，无则在资源最近目录写入缺省 .krd.json 后再同步
act_artifact.triggerSync?filepath='agt/xxx.txt'&forceSync=true&ensureKrd=true

### 计划工具
// markdown/多行文本转任务列表；子任务多缩进 2 空格或 1 tab；行首 [ ]/[x]；有子项的父行不执行
// 执行条件：任务下 when:，下一级列表写「- 前置任务名 success|failure」；遍历到该叶子时判定，不满足则跳过（视为成功继续），不改动叶子列表
n_plan_tasks, act_plan.parse?task='
# markdown 格式的任务规划（可多层级嵌套）
- [x] 已经完成的任务
- [ ] 阶段一：分析需求
  - [ ] 收集输入
  - [ ] 整理结论
- [ ] 阶段二：执行变更
  - [ ] 编写脚本
  - [ ] 运行验证
    when:
      - 编写脚本 success
  - [ ] 失败回滚
    when:
      - 运行验证 failure
'
// 将变量转换为 plan list，供 foreach 使用。必填：var=变量名。可选：maxConcurrent=同时未完成的异步任务数上限（≤0 不限制）；maxWaitSeconds=为单个任务的最大等待秒数（≤0 用系统默认）
n_plan_steps, act_plan.convert?var=steps&maxConcurrent=5&maxWaitSeconds=60
// 设置plan(var+index)的执行agent与参数；index空=当前plan，parameter为agent需要的参数串
act_plan.assignAgent?var=n_plan_tasks&index=1&executor=agt_findelement&parameter='step={{n_plan_tasks.item}}'
// 获取执行某个plan的agent名称
n_executor_agent, act_plan.agent?var=n_plan_tasks&index={{n_plan_tasks.index}}
// 执行 plan（async 见 FutureValue 节）
