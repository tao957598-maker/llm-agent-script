## 基本语法
### 变量与数据结构(sys_为系统内置变量，n_可兼做位置标记)
// 注释为//开头的行
// 数字
assign('num', '100')
// 字符串
assign('var', 'num')
// 引用变量
assign('referenced_value', '{{var}}')
// 嵌套模板使用%{}%语法
assign('nested_template', "{{fn.when('condition', '今天是2024-10-03', '今天是%{fn.today()}%')}}")
// 列表
assign('list', '["a","b"]')
// JSON
assign('json', '{"name":"jack"}')
// 链式定义(变量1,值1,变量2,值2)
assign("product", "oa",
		"requirement", '''无
		''',
		"requirement_score", ">=7")
// 通过memory工具赋值
act_memory.assign?var=result&value=xxx
// 拼接值val1,val2...给到变量var
append('var', 'val1', 'val2'...)
// 记录日志
log('log info {{var}}')

// fn.hasValue(n_data) 检查n_data是否有值 有值true / 无值false
assign('nested_template','{{fn.hasValue(n_data)}}')

// fn.isEmpty(n_data) 检查n_data是否有值 有值true / 无值false
assign('nested_template','{{fn.isEmpty(n_data)}}')

### 调用表达式
// 完整的Plan事务调用：读取知识库，调用LLM输出数据，抽取markdown内容，并检查LLM输出是否符合要求，最终保存为md_data变量，并在界面中的显示格式为markdown(注：最后的\为换行转义字符)
rag_kb.select?name=kb_md_example, tpl_md_output, psr_md, vrf_flag?value='{{fn.when(fn.hasValue(sys_last_result,"解释"),"true","false")}}',\
	act_memory.assign?var=md_data&value={{tpl_md_output}}&type=md

### 控制流
when('条件'， { ... })
elsewhen('条件', { ... })

loop('var_name', '条件', { ... })
// 异常处理
on_failure({ ... })
// 条件判断，{{}}中为freemarker表达式
when('{{product_id == compared_product_id}}', {
	assign("score", "9")
})
// when支持if-else（省略{{}}不推荐）
when('list?size gt 0',{
	assign("score", "9")
},{
	assign("score", "10")
})
elsewhen('{{condition}}', {
	assign("score", "11")
})
// elsewhen支持else（无参数时）
elsewhen({
	assign("fallback_score", "12")
})
// `foreach ('var_name', 'list', { ... })` for 循环，基于普通list数据，或task list数据. n_plan_tasks为json结构, 包含属性：index, item, size
foreach('n_plan_tasks', 'llm_data', {
    // index 下标
    assign('loop_task_index', '{{n_plan_tasks.index}}')
    // 下标对应内容
    assign('loop_task_item', '{{n_plan_tasks.item}}')
})

### 流程工具
// 跳转指定位置标记，同时支持内置标记：this(重新执行当前步骤)，next(下一步)，break，continue，end(跳转到agent的结束位置)
act_flow.go?plan=next
// 等待3秒
act_flow.wait?second=3
// 人与Agent的交互，展示UI页面(name来自UI模块的定义)，并暂停流程等待用户的输入
n_expense_form, act_flow.page?pageCode='expense_form'&params=
// LLM通过task()示范结构告诉引擎跳转节点（等价于tpl_plan_selector+act_flow.go）

用户输入为：{{userInput}}
直接进入下一步，JSON格式返回：{{fn.task().plan('next')}}
需要回到之前读文件的步骤，JSON格式返回：{{fn.task().plan('n_file_info')}}
结果错误，需要重新思考当前步骤，JSON格式返回：{{fn.task().rerun(true)}}
{{fn.chatFlag(5)}}
''')
// 捕捉所有前序异常,将当前流程转为正常状态。（可使用act_flow.fail再次将流程转为异常状态）
on_failure({
	assign("result", "fail")
	// goto语句，跳转到代码结尾
	act_flow.go?plan=end
})

### 执行审批节点（一次性执行的节点）
// `approve`用于构建审批节点，其注释为节点名。类似BPMN的UserTask，节点执行时会自动发送通知给相关人。assignee属性为json格式
// 常见的审批流程就是纯粹的approve节点，无需其他逻辑。比如3个节点的agent：小组长审批 ==> 项目经理审批 ==> 财务主管审批 
// 财务主管审批。type可选：STATIC，ROLE，EXPRESSION，HIERARCHY，DYNAMIC。 strategy可选：ALL，ANY，MAJORITY。fallbackAssignee为可选，后备审批用户ID。
// targetType可选：USER_ID/ROLE_NAME/DEPARTMENT_ID/EXPRESSION/REPORTING_LINE。value为具体角色名，condition为条件表达式。formKey为可选，关联报销表单
approve('n_financial_executive', '''
{
  "assigneeConfig": {
    "type": "ROLE",
    "strategy": "ALL",
    "fallbackAssignee": 2001,
    "targets": [
      {
        "targetType": "ROLE_NAME",
        "value": "财务主管",
        "condition": "amount > 10000"
      }
    ]
  },
  "formKey": "n_expense_report_form"
}
''')
