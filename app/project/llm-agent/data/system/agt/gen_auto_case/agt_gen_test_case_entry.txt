    if (rows == null || !(rows instanceof List)) rows = []
    def mode = "{{pipeline_mode}}"
    mode = (mode == null ? "" : mode.toString()).trim().toLowerCase()
    if (!mode) mode = "auto"
    if (!(mode in ["web", "api", "auto"])) mode = "auto"
    def out = []
    if (mode == "web") {
        rows.each { r -> if (r instanceof Map) out.add([st: "web", page_content: r]) }
        return out
    }
    if (mode == "api") {
        rows.each { r -> if (r instanceof Map) out.add([st: "api", page_content: r]) }
        return out
    }
    def wp = new HashSet()
    def ap = new HashSet()
    def ws = ""
    def asv = ""
    try { ws = web_pages_csv == null ? "" : web_pages_csv.toString() } catch (Throwable ignore) { ws = "" }
    try { asv = api_pages_csv == null ? "" : api_pages_csv.toString() } catch (Throwable ignore) { asv = "" }
    ws.split(",").each { String it2 -> def t = it2 == null ? "" : it2.trim(); if (t) wp.add(t) }
    asv.split(",").each { String it2 -> def t = it2 == null ? "" : it2.trim(); if (t) ap.add(t) }
    rows.each { r ->
        if (!(r instanceof Map)) return
        def p = r.get("page") == null ? "" : r.get("page").toString().trim()
        def inW = wp.contains(p)
        def inA = ap.contains(p)
        if (inW && !inA) out.add([st: "web", page_content: r])
        else if (inA && !inW) out.add([st: "api", page_content: r])
        else if (inW && inA) {
            out.add([st: "web", page_content: r])
            out.add([st: "api", page_content: r])
        } else {
            out.add([st: "web", page_content: r])
            out.add([st: "api", page_content: r])
        }
    }
    return out
'''
