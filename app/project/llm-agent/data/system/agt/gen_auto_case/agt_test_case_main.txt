//# Agent status
assign("status", 'Agent is running')
// Document ID
assign("documentId", '{{kbSpaceFileId}}')

n_design_flattened_data, act_kb.selectBy?sql='SELECT * FROM kb_gen_test_module_design WHERE documentId ="{{documentId}}" and version ="{{version}}"'

// 按 submodule_name 分批：同批多条 design 汇总后再治理导入
n_design_batches,act_func?script='''
    def d = fn.get('n_design_flattened_data')
    if (d == null || !(d instanceof List)) d = []
    def byKey = new LinkedHashMap()
    d.each { row ->
        if (!(row instanceof Map)) return
        def name = (row.get("submodule_name") ?: "").toString().trim()
        if (!byKey.containsKey(name)) byKey[name] = []
        byKey[name].add(row)
    }
    def batches = []
    byKey.each { k, v -> batches.add(v) }
    return batches
'''
assign('design_batches', '{{n_design_batches}}')

n_test_case_delete_rows,act_kb.selectBy?sql='select distinct document_id as doc_id from kb_gen_test_case where version = "{{version}}" and document_id in ("{{documentId}}","{{kbSpaceFileId}}")'
foreach('n_test_case_delete_row', 'n_test_case_delete_rows', {
	act_kb.delete?name=kb_gen_test_case&where='{"version":"{{version}}","document_id":"{{n_test_case_delete_row.item.doc_id}}"}'&maxSize=1000
	on_failure({
		// Force to success status
		act_flow.success
	})
})
assign('flush_data', '[]')
assign('seen_biz_global', '[]')
assign('seen_no_global', '{}')
assign('serial_counter', '{{fn.raw(0)}}')

foreach('n_batch', 'design_batches', {
    assign('batch_designs', '{{n_batch.item}}')
    act_func?script='fn.put("batch_case_rows", [])'
    foreach('n_test_design', 'batch_designs', {
        assign('test_design', '{{n_test_design.item}}')
        assign('design_case_rows', '[]')
        n_gen_case,agt_test_doc_gen_case?input_data='{{test_design}}'&documentId='{{documentId}}'&async=false
        assign('test_case_raw', '{{n_gen_case}}')
        assign('test_case_map', '{}')
        act_func?script='''
            def h = test_case_raw
            if (h == null) { fn.put("test_case_map", [:]); return }
            def v = null
            try {
                if (h instanceof Map) {
                    v = h.get("value")
                    if (v == null) v = h
                } else {
                    v = h
                }
            } catch (Exception ignore) {
                v = h
            }
            if (v == null || !(v instanceof Map)) v = [:]
            fn.put("test_case_map", v)
        '''
        foreach('n_test_case_map','test_case_map.data',{
            assign('test_case', '{{n_test_case_map.item}}')
            act_func?script='''
                // 根据 sub_id 和目标语言进行 sub_category 语言转换
                def subId = (test_case_map.sub_id ?: "").toString().trim()
                def lang = "{{outputLanguage ?: 'zh'}}".trim().toLowerCase()
                def targetSubCategory = test_case_map.sub_category

                def langMap = [
                    "10": ["Navigation flow", "导航流程", "導航流程"],
                    "11": ["Interaction", "交互功能", "交互功能"],
                    "13": ["Static content", "静态内容", "靜態內容"],
                    "15": ["Localization and internationalization", "本地化与国际化", "本地化與國際化"],
                    "20": ["Function validation: field input", "功能验证-栏位输入控制", "功能驗證-欄位輸入控制"],
                    "21": ["Function validation: field output display", "功能验证-栏位输出显示", "功能驗證-欄位輸出顯示"],
                    "22": ["Function validation: data rules", "功能验证-数据", "功能驗證-數據"],
                    "31": ["Process: enter feature flow", "流程-进入功能流程", "流程-進入功能流程"],
                    "32": ["Process: normal business", "流程-正常业务流程", "流程-正常業務流程"],
                    "33": ["Process: exception business", "流程-异常业务流程", "流程-異常業務流程"],
                    "40": ["Logic: positive logic", "逻辑-正面逻辑", "邏輯-正面邏輯"],
                    "41": ["Logic: negative logic", "逻辑-反面逻辑", "邏輯-反面邏輯"],
                    "43": ["Logic: trigger conditions", "逻辑-逻辑触发条件", "邏輯-邏輯觸發條件"]
                ]

                if (subId && langMap.containsKey(subId)) {
                    def labels = langMap[subId]
                    targetSubCategory = lang == "zh" ? labels[1] : (lang == "tw" ? labels[2] : labels[0])
                } else if (targetSubCategory == null) {
                    targetSubCategory = ""
                }
                
                def row = [
                    title: test_case.title,
                    application: test_case.application,
                    steps: test_case.steps,
                    test_no: test_case.test_no,
                    expected_result: test_case.expected_result,
                    source_value: test_case.source_value,
                    test_type: test_case.test_type,

                    sub_category: targetSubCategory,
                    sub_id: test_case_map.sub_id,
                    design_element: test_case_map.design_element,
                    test_content: test_case_map.test_content,
                    platform: test_case_map.platform,
                    main_module_name: test_case_map.main_module_name,
                    main_module_source: test_case_map.main_module_source,
                    submodule_name: test_case_map.submodule_name,
                    submodule_description: test_case_map.submodule_description,
                    submodule_source: test_case_map.submodule_source,
                    document_id: fn.optional(kbSpaceFileId, ''),
                    version: version
                ]
                design_case_rows.add(row)
            '''
        })
        act_func?script='''
            def acc = fn.get('batch_case_rows')
            if (acc == null || !(acc instanceof List)) acc = []
            def add = fn.get('design_case_rows')
            if (add instanceof List) add.each { acc.add(it) }
            fn.put('batch_case_rows', acc)
        '''
    })
    act_func?script='''
        def arr = fn.get('batch_case_rows')
        if (arr == null || !(arr instanceof List)) { fn.put("flush_data", []); return }
        def norm = { v -> (v == null ? "" : v.toString()).trim().replaceAll(/\\s+/, " ") }
        def containsAny = { txt, kws ->
            def s = (txt == null ? "" : txt.toString().toLowerCase())
            kws.any { k -> s.contains(k) }
        }
        def forbidden = [
            "unregistered","not registered","feature not registered","no permission","without permission",
            "access blocked","navigation denied","button disabled","未注册","無權限","无权限"
        ]
        def seenBiz = (seen_biz_global instanceof List) ? new LinkedHashSet(seen_biz_global) : new LinkedHashSet()
        def seenNo = (seen_no_global instanceof Map) ? new LinkedHashMap(seen_no_global) : new LinkedHashMap()
        int serial = 0
        try { serial = Integer.parseInt((serial_counter == null ? "0" : serial_counter.toString())) } catch (Throwable ignore) { serial = 0 }
        def out = []
        arr.each { row ->
            if (!(row instanceof Map)) return
            def text = [row.title, row.steps, row.expected_result].collect { norm(it) }.join(" ").toLowerCase()
            if (containsAny(text, forbidden)) return
            def bizKey = [norm(row.submodule_name), norm(row.sub_id), norm(row.title), norm(row.steps), norm(row.expected_result), norm(row.version), norm(row.document_id)].join("||").toLowerCase()
            if (!bizKey || seenBiz.contains(bizKey)) return
            seenBiz.add(bizKey)
            def no = norm(row.test_no)
            if (!no) {
                serial = serial + 1
                no = "TC-" + serial
            }
            if (!seenNo.containsKey(no)) {
                seenNo.put(no, 1)
                row.test_no = no
            } else {
                def n = (seenNo.get(no) ?: 1) + 1
                seenNo.put(no, n)
                row.test_no = no + "-D" + n
            }
            out.add(row)
        }
        fn.put("flush_data", out)
        fn.put("seen_biz_global", new ArrayList(seenBiz))
        fn.put("seen_no_global", seenNo)
        fn.put("serial_counter", serial)
    '''
    when('fn.hasValue(flush_data)', {
        act_kb.defineAndImportData?tableName='kb_gen_test_case'&data='{{flush_data}}'
    }, {
        act_flow.success
    })
})

assign('result_test_cases', 'stream_import_done')
