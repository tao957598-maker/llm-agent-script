n_increment_diff_pages,act_func?script='''
    def splitCommaList = { raw ->
        def s = (raw == null ? "" : raw.toString()).trim()
        def out = []
        if (!s) return out
        def current = new StringBuilder()
        for (int i = 0; i < s.length(); i++) {
            def ch = s.charAt(i)
            if (ch == ',') {
                def token = current.toString().trim()
                if (token) out.add(token)
                current.setLength(0)
            } else {
                current.append(ch)
            }
        }
        def token = current.toString().trim()
        if (token) out.add(token)
        return out
    }
    def normalizePageKey = { v ->
        def x = (v == null ? "" : v.toString()).trim().toLowerCase()
        if (x.startsWith("page")) x = x.substring(4).trim()
        return x
    }
    def singlePageText = { inputVal ->
        def s = (inputVal == null ? "" : inputVal.toString()).trim()
        if (!s) return ""
        def token = s.contains(",") ? s.split(",")[0].trim() : s
        if (!token) return ""
        if (token.toLowerCase().startsWith("page")) {
            token = token.substring(4).trim()
        }
        return token
    }
    def scopeKeys = fn.get("increment_scope_page_keys")
    if (!(scopeKeys instanceof List)) scopeKeys = []
    def pickBestContent = { r ->
        if (!(r instanceof Map)) return ""
        def analysis = r.get("analysis_content")
        def content = r.get("content")
        def picked = fn.hasValue(analysis) ? analysis : content
        return picked == null ? "" : picked.toString().trim()
    }
    def aggregateRowsByPage = { rows ->
        def pageMap = new LinkedHashMap()
        (rows instanceof List ? rows : []).each { r ->
            if (!(r instanceof Map)) return
            def pk = normalizePageKey(r.get("page"))
            if (!pk) return
            if (!scopeKeys.isEmpty() && !scopeKeys.contains(pk)) return
            def pageLabel = singlePageText.call(r.get("page"))
            def content = pickBestContent(r)
            if (!fn.hasValue(content)) return
            def bucket = pageMap.get(pk)
            if (bucket == null) {
                bucket = [page: pageLabel, parts: []]
                pageMap.put(pk, bucket)
            }
            if (!fn.hasValue(bucket.page) && fn.hasValue(pageLabel)) {
                bucket.page = pageLabel
            }
            bucket.parts.add(content)
        }
        def out = new LinkedHashMap()
        pageMap.each { k, bucket ->
            def joined = (bucket.parts instanceof List ? bucket.parts : []).join("\n")
            out.put(k, [
                page: (bucket.page ?: "").toString().trim(),
                content: joined,
                part_count: (bucket.parts instanceof List ? bucket.parts.size() : 0)
            ])
        }
        return out
    }
    def oldRows = n_old_parse_rows
    if (!(oldRows instanceof List)) oldRows = []
    def newRows = n_new_parse_rows
    if (!(newRows instanceof List)) newRows = []
    def oldMap = aggregateRowsByPage(oldRows)
    def newMap = aggregateRowsByPage(newRows)
    def toPageList = { pageMap ->
        def out = []
        pageMap.each { k, v ->
            if (v instanceof Map) {
                out.add([
                    page: v.page,
                    content: v.content
                ])
            }
        }
        return out
    }
    def oldList = toPageList(oldMap)
    def newList = toPageList(newMap)
    fn.put("increment_old_page_map", oldMap)
    fn.put("increment_new_page_map", newMap)
    fn.put("increment_old_json_pages", oldList)
    fn.put("increment_new_json_pages", newList)
    return [
        old_json: oldList,
        new_json: newList
    ]
'''

    def splitCommaList = { raw ->
        def s = (raw == null ? "" : raw.toString()).trim()

                if (token) out.add(token)
                current.setLength(0)

                current.append(ch)
            }

        if (token) out.add(token)
        return out

        def x = (v == null ? "" : v.toString()).trim().toLowerCase()
        if (x.startsWith("page")) x = x.substring(4).trim()
        return x

    def normalizePageList = { inputVal ->
        def s = (inputVal == null ? "" : inputVal.toString()).trim()
        if (!s) return ""
        def out = []
        s.split(/\s*,\s*/).each { token ->
            def t = (token == null ? "" : token.toString()).trim()
            if (!t) return
            if (t.toLowerCase().startsWith("page")) t = t.substring(4).trim()
            if (t) out.add(t)
        }
        return out.unique().join(",")
    }

        def s = normalizePageList.call(inputVal)
        if (!s) return []

            if (key) out.add(key)
        }
        return out.unique()

            def mapped = remap.get(key)

        }

    }

        if (!(addKeys instanceof List)) addKeys = []
        def updateKeys = fn.get("increment_update_page_keys")
        if (!(updateKeys instanceof List)) updateKeys = []
        def deleteKeys = fn.get("increment_delete_page_keys")
        if (!(deleteKeys instanceof List)) deleteKeys = []
        def keys = pageKeyList.call(pageText)

        return buckets

        if (remap instanceof Map) {
            def remapped = [:]
            oldBuckets.each { key, bucket ->
                def outKey = remapPageList.call(key, remap)

                def outBucket = remapped.get(outKey)
                if (outBucket == null) {
                    outBucket = []
                    remapped.put(outKey, outBucket)
                }
                outBucket.addAll(bucket instanceof List ? bucket : [])

        orderedKeys.addAll(tempBuckets.keySet())

        def out = []

                    out.add(cloneRow(row, action, docField))
                }
            }
        }
        def pageSample = []

        }
        println("[INCREMENT-FINAL][" + docField + "] oldPages=" + oldBuckets.size() + ", tempPages=" + tempBuckets.size() + ", out=" + out.size() + ", pages=" + pageSample)
