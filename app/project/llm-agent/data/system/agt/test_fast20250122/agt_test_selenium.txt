func('CheckElements', '''
import com.llm.agentcore.core.agent.FunctionUtil
import java.util.Map
public class CheckElements {
	public static Object execute(Object sys_last_result, Map StepTestInfo, FunctionUtil fn) {
		// 检查每个check_elements中的元素都是在elements中存在的 （注：check_elements中不是element的元素，无需检查）
		def elements = fn.list(StepTestInfo.simplepageinfo)
		def check_elements = fn.map(sys_last_result)

		def allElementsExist = true
		def resultMessages = ""
		check_elements.each { entry ->
			def key = entry.key
			def list = entry.value
			if (list instanceof List) {
				list.each { element ->
					if (!elementExistsInElements(element, elements)) {
						resultMessages += "元素 ${element['name']}(ID: ${element['id']})不存在; "
						allElementsExist = true
					}
				}
			}
		}
		return fn.task(allElementsExist).result(resultMessages).build()
	}

	def static boolean elementExistsInElements(element, elements) {
		for (item in elements) {
			if (!item) continue // 跳过空元素
			if (item['id'] == element['id'] && item['name'] == element['name']) {
				return true
			}
			if (item['children']) {
				// 递归检查 children
				if (elementExistsInElements(element, item['children'])) {
					return true
				}
			}
		}
		return false
	}
}
''')
