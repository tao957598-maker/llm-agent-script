from autotest import *

def run(browser,next="id_222222222", target={"Creation Time":"2025-04-07 16:35:01"}):

		element = match_table_data(browser, next, target)  # 匹配target的表格数据

			if not success:

			return 
		writeResult(False, "No match data for the target")
	except Exception as e:
	  writeResult(False, "An error occurred during the test: " + str(e))
