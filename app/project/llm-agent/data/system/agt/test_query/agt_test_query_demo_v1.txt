// 定义异步结果收集器
assign('async_list', '[]')

// 设置问题集合
assign("kb_list_data", '[]')
append("kb_list_data", "{{userInput}}")


// 异步查询
foreach('n_question', 'kb_list_data', {
  // 获取当前用户输入
  assign('question', '{{n_question.item}}')

  when('fn.hasValue(kbData)',{
    //# Execute query from expert knowledge base
    n_expert_query_answer, agt_test_query_expert_0?userInput={{userInput}}&kbData={{kbData}}&async=true
    act_func?script="async_list.add(n_expert_query_answer)"
  })

  when('fn.hasValue(kbSpaceFileId)',{
    // 向量查询
    n_vector_query_answer, agt_test_query_vector_1?userInput={{userInput}}&folderIds={{kbSpaceId!}}&knowledgeIds={{kbSpaceFileId}}&async=true
    act_func?script="async_list.add(n_vector_query_answer)"

    // 文档查询
    n_document_query_answer,agt_test_query_document_2?userInput={{userInput}}&kbSpaceFileId={{kbSpaceFileId}}&async=true
    act_func?script="async_list.add(n_document_query_answer)"

  })
})

// 定义一个空 list
assign('answer_summary_list', '[]')

// 对查询结果进行循环处理并分类
foreach('n_item', 'async_list', {
  // 获取当前循环项的数据
  assign('async_summary', '{{n_item.item}}')
  // 当前异步结果
  assign('answer_summary', '{{async_summary.value}}')
  act_func?script='''
    answer_summary.each { item ->
      answer_summary_list.add(item)
    }
  '''
  when('seek_mode=="快速模式" && answer_summary_list?size>0',{
    // 快速模式
    act_flow.go?plan=break
  })

})

// 并发查询最终结果
assign('n_answers', '{{answer_summary_list}}')
