assign("CODE_GEN_PROMPT", '''
你是一个金融安全代码生成器。请根据任务描述，生成一个**纯函数**，用于在隐私计算环境中执行。

## 安全约束（必须严格遵守！）
1. 函数名必须为 `compute`，且唯一参数为 `inputs`
2. 所有输入数据必须从 `inputs` 中读取，**禁止硬编码任何业务数值**（如金额、汇率、日期等）。
3. 仅允许使用以下语法：
   - 变量赋值、if/else、for 循环（禁止 while）
   - 基本运算：+ - * / **, min, max, abs
   - 数据结构：list, dict, tuple（禁止 set）
4. **允许在函数内部使用 import 语句，但仅限以下安全的计算相关标准库模块**：
   - `math`（如需 sqrt、log、exp 等）
   - 不得导入其他任何模块（如 os、sys、numpy、pandas、json 等）
5. **import 语句必须出现在函数体开头，且仅当函数实际使用该模块时才引入**
6. 禁止以下操作：
   - exec, eval, open, getattr, setattr, __builtins__, I/O, lambda, class, global, while
7. 必须返回一个字典，结构严格符合 output_schema
8. 代码必须能在配置了安全白名单的 RestrictedPython 或 asteval 中执行
9. **禁止输出任何注释、解释、Markdown、或非代码文本** —— 仅输出函数定义

## 输入上下文说明
你将获得两部分信息，用于理解 `inputs` 的结构：
- **input_data**：`inputs` 的一个具体示例值（用于观察字段名和嵌套结构）
- **input_vars**：`inputs` 的类型描述。该描述必须准确反映 `inputs` 的顶层类型：
  - 如果 `inputs` 是一个 **列表**（如 `[{"date": "...", "price": ...}, ...]`），则 `input_vars` 应描述为 `"list of dict with keys: date (str), price (float/int)"` 或类似，**不得使用 dict 形式的键名包装**。
  - 如果 `inputs` 是一个 **字典**（如 `{"prices": [...]}`），则 `input_vars` 应描述为 `"dict with keys: prices (list)"`。

请严格根据 `input_data` 的实际结构解析 `inputs`：
- 若 `input_data` 是一个列表，则 `inputs` 就是该列表，直接遍历即可；
- 若 `input_data` 是一个字典，则按其键访问对应字段。

**禁止假设 `inputs` 包含某个特定键（如 'prices'），除非 `input_data` 明确显示该键存在。**

## 特别规则：何时输出空字符串
- 如果任务明显依赖外部数据（如涉及交易、客户、金额、列表等），但提供的 input_data 和 input_vars 为空、无效或无法推断字段结构，则**直接输出空字符串（即不输出任何字符）**。
- 如果任务不依赖运行时数据（例如返回固定规则或常量），即使上下文为空，仍应生成合法函数。

## 访问原则
始终以 `input_data` 的顶层结构为准。若 `input_data` 是 list，则 `inputs` 是 list；若 `input_data` 是 dict，则 `inputs` 是 dict。不要依据 `input_vars` 的模糊表述覆盖 `input_data` 的实际形态。

## 任务
{{user_question}}

## 输入上下文说明
input_vars: {{input_vars}}
input_data: {{input_data}}

## 输出要求
{{output_schema_description}}
''')

// 辅助函数：检查是否需要生成代码
define('should_generate_code', '''
when('{{input_data}}!', {
    // 如果 input_data 不为空，总是尝试生成
    assign('should_gen', 'true')
}, {
    when('{{input_vars}}!', {
        // 如果 input_vars 不为空，尝试生成
        assign('should_gen', 'true')
    }, {
        // 检查任务是否依赖外部数据
        assign('keywords', '["金额", "利率", "交易", "客户", "列表", "数组", "价格", "数量", "日期"]')
        assign('contains_data_keyword', 'false')
        
        foreach('keyword', '{{keywords}}', {
            when('fn.contains({{user_question}}, {{keyword}})', {
                assign('contains_data_keyword', 'true')
            }, {})
        })
        
        when('{{contains_data_keyword}} == true', {
            // 依赖外部数据但无输入，输出空字符串
            assign('should_gen', 'false')
        }, {
            // 不依赖外部数据，可以生成
            assign('should_gen', 'true')
        })
    })
})
return '{{should_gen}}'
''')

agent("code_generator_agent", {
    // 存储提示词
    assign("prompt_content", "{{CODE_GEN_PROMPT}}")
    expression('''act_memory.assign?var=code_gen_prompt&value={{prompt_content}}&global=true''')
    
    // 检查是否需要生成代码
    assign("check_result", "{{should_generate_code}}")
    
    when('{{check_result}} == false', {
        // 返回空字符串
        return ""
    }, {
        // 准备参数 - 使用 expression 包装所有函数调用
        assign("final_prompt", expression('''fn.replace("{{prompt_content}}", "{{user_question}}", "{{user_question}}")'''))
        assign("final_prompt", expression('''fn.replace("{{final_prompt}}", "{{input_vars}}", "{{input_vars}}")'''))
        assign("final_prompt", expression('''fn.replace("{{final_prompt}}", "{{input_data}}", "{{input_data}}")'''))
        assign("final_prompt", expression('''fn.replace("{{final_prompt}}", "{{output_schema_description}}", "{{output_schema_description}}")'''))
        
        // 调用生成逻辑
        assign("generated_code", expression('''act_llm.generate?prompt={{final_prompt}}&model=gpt-4&temperature=0.1'''))
        
        // 清理结果，确保只包含代码
        assign("cleaned_code", expression('''fn.strip("{{generated_code}}")'''))
        
        // 移除可能的Markdown代码块标记
        // 简化条件判断，使用字符串方法
        assign("starts_with_python", expression('''fn.startsWith("{{cleaned_code}}", "```python")'''))
        when('{{starts_with_python}} == true', {
            assign("cleaned_code", expression('''fn.replace("{{cleaned_code}}", "```python", "")'''))
            assign("cleaned_code", expression('''fn.replace("{{cleaned_code}}", "```", "")'''))
        }, {})
        
        assign("starts_with_backticks", expression('''fn.startsWith("{{cleaned_code}}", "```")'''))
        when('{{starts_with_backticks}} == true', {
            assign("cleaned_code", expression('''fn.replace("{{cleaned_code}}", "```", "")'''))
        }, {})
        
        return "{{cleaned_code}}"
    })
})
