// Initialize variables
assign("page_content", '[]')
assign("n_source_table", 'kbs_doc_page_parse')

// Step 1: Query document metadata by kbSpaceFileId
n_metadata, act_kb.select?name='kb_doc_req_metadata'&column='page,description'&where='{"documentId":"{{kbSpaceFileId}}"}'&orderBy='page asc'

// Step 2: Use LLM to judge relevance and return relevant page numbers

assign("pages", '{{n_relevant_pages.pages}}')

// Step 3: Get page content based on relevant page numbers and filename
foreach('n_page_num', 'pages', {

	assign('page_num', '{{n_page_num.item}}')

    n_page_content, act_kb.select?name='kbs_doc_page_parse'&column='id,content'&where='{"documentId":"{{kbSpaceFileId}}","page":"{{page_num}}"}'

	// Get page id from kb
	n_id,act_func?script='''return n_page_content[0].id.toString()'''
	// Get page content from kb
  	n_content,act_func?script='''return n_page_content[0].content'''

	// Set page content
	assign("page_content", '{{fn.json("page,page_num,id,n_id,documentId,kbSpaceFileId,source_table,n_source_table,content,n_content")}}')

})


// The finally result of document query
assign("ducoment_query_answer", '{{page_content}}')
