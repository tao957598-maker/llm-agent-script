def data = n_vl_concurrency

def cleanAndParseJson(str) {
    if (!(str instanceof String)) return str
    
    def cleaned = str.replaceAll('(?s)^```(?:json)?\s*', '')
                    .replaceAll('\s*```$', '')
    
    return cleaned
}

def flattenedData = []
data.each { item ->
    if (item instanceof Collection) {
        flattenedData.addAll(item)
    } else {
        flattenedData.add(item)
    }
}

def mid_data = flattenedData.collect { it.collect { cleanAndParseJson(it.value ?: it) } }
return mid_data.findResults { it[1] }
'''
