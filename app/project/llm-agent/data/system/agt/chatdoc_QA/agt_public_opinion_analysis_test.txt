// 1. Initialize and configure data collection
assign('topic', '{{analysis_topic}}')

assign('channels', '{{collection_channels}}')

assign('period_days', '{{analysis_period}}')

// Define prompt for data collection configuration


// Generate collection configuration
n_collection_config, tpl_collection_config, psr_json

// 2. Multi-channel data collection
foreach('n_channel_item', '{{n_collection_config.sources}}', {
	assign('source_type', '{{n_channel_item.type}}')

	assign('source_name', '{{n_channel_item.name}}')

	assign('source_url', '{{n_channel_item.url}}')

	when('{{source_type == "API"}}', {
		// API-based collection
		n_api_data, act_api?name="collect_public_data"&params='{{fn.json("topic,topic,source,source_name,url,source_url,keywords,n_collection_config.keywords")}}'
	
		append('collected_data', '{{n_api_data}}')
	})

	when('{{source_type == "Web"}}', {
		// Web crawler collection
		n_web_data, act_func.run?script='''
		            // Web crawling logic
		            def crawler = new WebCrawler()
		            def content = crawler.fetch("{{source_url}}")
		            return content
		        '''
	
		append('collected_data', '{{n_web_data}}')
	})

	when('{{source_type == "RSS"}}', {
		// RSS feed collection
		n_rss_data, act_func.run?script='''
		            // RSS parsing logic
		            def rssParser = new RSSParser()
		            def feed = rssParser.parse("{{source_url}}")
		            return feed.items
		        '''
	
		append('collected_data', '{{n_rss_data}}')
	})
})

// 3. Intelligent content processing


// Process collected content
n_processed_data, tpl_content_processing, psr_json

// 4. Entity recognition and sentiment analysis enhancement


// Perform enhanced analysis
n_enhanced_analysis, tpl_enhanced_analysis, psr_json

// Convert processed content to vectors
n_vector_data, act_kb.convert?data='{{n_processed_data.processed_content}}'&convert=vector

// Store in vector database
act_kb.importData?name='kb_public_opinion'&data='{{fn.json("topic,topic,content,n_processed_data.processed_content,vector,n_vector_data,entities,n_enhanced_analysis.enhanced_entities,sentiment,n_processed_data.sentiment,tags,n_enhanced_analysis.dynamic_tags,metadata,n_processed_data.metadata,timestamp,fn.now()")}}'

// 6. Semantic similarity search capability


// Perform semantic search
n_similar_content, act_kb.search?requirement='{{tpl_semantic_search}}'&name='kb_public_opinion'

// 7. Automated report generation


// Generate final report
n_final_report, tpl_report_generation, psr_md

// Create monitoring task
assign('monitoring_task', '{{fn.json("name,Public Opinion Monitoring - topic,topic,channels,channels,frequency,daily,status,ACTIVE,last_run,fn.now(),next_run,fn.addDays(fn.now(), 1)")}}')

// Store task configuration
act_kb.importData?name='kb_monitoring_tasks'&data='{{monitoring_task}}'

// 9. Create one-time or periodic collection jobs
when('{{uiEntrance.schedule_type == "periodic"}}', {
	assign('jobData', '{{fn.json("agentName,agt_public_opinion,jobDesc,Periodic Public Opinion Collection,parameters,fn.json("analysis_topic,topic,collection_channels,channels,analysis_period,period_days"),schedule,0 0 * * *,notifyStrategy,On Failure")}}')

	n_job_created, act_job.createAndTrigger?data='{{jobData}}'&isOneOffTask=false
})

when('{{uiEntrance.schedule_type == "one_time"}}', {
	assign('jobData', '{{fn.json("agentName,agt_public_opinion,jobDesc,One-time Public Opinion Analysis,parameters,fn.json("analysis_topic,topic,collection_channels,channels,analysis_period,period_days"),notifyStrategy,全部通知")}}')

	n_job_created, act_job.createAndTrigger?data='{{jobData}}'&isOneOffTask=true
})

// 10. Error handling and logging
on_failure({
	assign('error_message', '{{sys_last_error}}')

	log('Public opinion analysis failed: {{error_message}}')

	// Send notification
	act_func.run?script='''
	        // Send error notification
	        def notification = new NotificationService()
	        notification.sendAlert(
	            "Public Opinion Analysis Failed",
	            "Topic: {{topic}}
	Error: {{error_message}}",
	            "HIGH"
	        )
	    '''

	// Store error for analysis
	act_kb.importData?name='kb_analysis_errors'&data='{{fn.json("topic,topic,error,error_message,timestamp,fn.now(),agent,agt_public_opinion")}}'
})

// 11. Return final results
assign('final_result', '{{fn.json("report,n_final_report,analysis,n_enhanced_analysis,similar_content,n_similar_content,monitoring_task,monitoring_task,job_created,n_job_created")}}')

// Store analysis history
act_kb.importData?name='kb_analysis_history'&data='{{fn.json("topic,topic,report_summary,fn.substring(n_final_report, 0, 500),sentiment_score,n_processed_data.sentiment.score,risk_level,n_enhanced_analysis.risk_level,analysis_date,fn.now()")}}'

// Output final result
act_memory.assign?var=analysis_result&value='{{final_result}}'&type=json

  "sources": [

    }
  ],
  "schedule": "daily/hourly",
  "filters": {
    "language": "en/zh",
