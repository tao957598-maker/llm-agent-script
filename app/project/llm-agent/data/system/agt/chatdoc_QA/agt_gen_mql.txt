// params - list metadata
assign("param_mongo_metadata", '''
{
  'folders': {{kbSpacePaths}},
  'fileNames': {{kbSpaceFilePaths}},
  'description': '{{indicatorDesc}}'
}
''')
 
// Query metadata in Milvus
n_mongo_metadatas_response, act_plan.run?toolType=API&code='get_reportData_metadata'&parameters='{{fn.str(param_mongo_metadata)}}'
assign('n_mongo_metadatas_data', '{{n_mongo_metadatas_response.data}}')
assign('metadata_data_list','[]')

when('!fn.hasValue(n_mongo_metadatas_data)', {
    // Empty data
	assign('mql', '')
}, {
    // Query metadata in MongoDB
    foreach('mongo_metadata', 'n_mongo_metadatas_data', {
        assign('mongo_metadata_data', '{{mongo_metadata.item}}')
        // params - query by titles
        assign("queryByTitles", '{"fileNamePrefix": "{{mongo_metadata_data.fileName}}", "titles": "{{mongo_metadata_data.title}}", "dataLimit": 10}')

        // Query metadata by title
        n_titleMetaData, act_api.doExecuteDynamic?name=ReportMetaData&dynamicParamData='{{queryByTitles}}'
        n_metadata_data_list,act_func?script='''
            def list_temp = metadata_data_list
            list_temp.add(n_titleMetaData)
            return list_temp
        '''
        assign('metadata_data_list', '{{n_metadata_data_list}}')
	})
})
log('metadata_data_list = {{metadata_data_list}}')


// Generate MQL 
tpl_gen_mql_by_metadata, psr_all, n_reportData, act_api.doExecuteDynamic?name=ReportDataSearch&dynamicParamData={{fn.json("searchMQL, tpl_gen_mql_by_metadata, size, '1', filterNull, 'true'", true)}}

assign('mql', '{{tpl_gen_mql_by_metadata}}')
