//## 第一步：将自然语言需求转换为 MongoDB 查询语句
n_table_search_data,agt_text_to_mongoql.report?kbsFileName="wp_demo"&indicatorDesc="{{userInput}}"

// ##第二步：执行 MongoDB 查询，获取原始数据
n_sql_result, act_api.doExecuteDynamic?name=ReportDataSearch&dynamicParamData='{{fn.json("searchMQL, n_table_search_data", true)}}'

// ##第三步：从返回的结果中提取名称字段，纯文本输出
n_final, act_func.run?script='''
    // 1. 获取原始返回内容（可能是字符串或对象）
    def raw = fn.get("n_sql_result")
    // 如果 raw 不是字符串，转为字符串（如 Groovy Map 的 toString 会是类似 {data=[...]} 的格式）
    def text = raw?.toString() ?: ""
    
    // 2. 用正则匹配所有 "name":"..." 片段
    def namePattern = /"name"\s*:\s*"([^"]+)"/
    def matcher = (text =~ namePattern)
    
    // 3. 收集所有名称，取前 30 个
    def names = []
    while (matcher.find() && names.size() < 30) {
        names.add(matcher.group(1))
    }
    
    // 4. 获取用户输入，用于提示
    def userInput = fn.get("userInput")?.trim()
    
    // 5. 返回结果
    names.isEmpty() ? "未找到 rating 为 ${userInput} 的店铺" : names.join("\\n")
'''
