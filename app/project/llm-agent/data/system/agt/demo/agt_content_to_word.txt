	"parameter":[
		{

		},
		{

			"label":"报告标题",
			"required":true,
			"type":"text"
		},
		{
			"name":"folder",
			"label":"目标知识空间文件夹",
			"required":false,
			"type":"text",
			"default":"wp_demo"
		},
		{
			"name":"file_name",
			"label":"自定义文件名（不含扩展名，留空则自动生成）",
			"required":false,
			"type":"text"

// 1. 处理自定义文件名
assign('clean_name', '{{fn.optional(file_name, "")}}')
when ("clean_name != ''", {
    assign('base_name', '{{clean_name}}')
}, {
    assign('base_name', 'report_{{fn.now().replaceAll("[-: ]", "")}}')
})

// 2. 构建 Markdown 内容
assign('md_content', '''# {{title}}

{{content}}''')

// 3. 写入临时文件
assign('tmp_path', 'tmp/{{base_name}}.md')
n_write, act_func.write?filename="{{tmp_path}}"&data="{{md_content}}"

// 4. 转换为 Word
assign('converter_type', '{"converter_type":"md_to_word"}')
n_word_url, act_api.doExecute?name='report_word_to_pdf'&fileNames='{{tmp_path}}'&params='{{converter_type}}'

// 5. 提取 Word 相对路径
n_word_rel, act_func?script='''
    def url = n_word_url.relativePath
    return url.replace("/app/data/", "/")
'''

// 6. 上传到知识空间
n_upload, act_api.doExecute?name='report_kbs_upload_user_file'&fileNames="{{n_word_rel}}"&params='{"folderName":"{{folder}}"}'

// 7. 提取文件 ID 作为最终输出
n_final_file_id, act_func?script='''
    if(n_upload.code == 200){
        def m = n_upload.data[0]
        def k = m.keySet().iterator().next()
        return m[k]
    }
    return "上传失败"
'''

//## 最终输出
assign('final_output', '{{n_final_file_id}}')
