// === Step 3: 增强聚类（全维度 + 标准化 + 动态k最少8组） ===
n_py_file, act_python.writeTmp?requirement='聚类'&path='tmp/cluster_analysis.py'&sourceCode='''
import json, math, random, sys

raw_block = """{{raw_data_block}}"""

vectors = []
file_names = []
for line in raw_block.strip().split("\n"):
    line = line.strip()
    if not line: continue
    idx = line.index("::")
    fname = line[:idx].strip()
    vec_str = line[idx+2:].strip()
    vec = json.loads(vec_str)
    vec = [float(x) for x in vec]
    vectors.append(vec)
    file_names.append(fname)

if not vectors:
    print("ERROR: no vectors")
    sys.exit(0)

n = len(vectors)
d = len(vectors[0])

# 关键改进1：使用全部维度，不做硬截断
X = [v[:] for v in vectors]

# 关键改进2：Z-score 标准化，消除量纲差异
means = [sum(vals)/n for vals in zip(*X)]
stds = []
for j in range(d):
    var = sum((v[j]-means[j])**2 for v in X) / n
    stds.append(var**0.5)
for v in X:
    for j in range(d):
        v[j] = (v[j] - means[j]) / (stds[j] if stds[j] > 0 else 1.0)

def kmeans(X, k, max_iters=100):   # 迭代次数增至100
    random.seed(42)
    centers = random.sample(X, k)
    labels = [0] * len(X)
    for _ in range(max_iters):
        changed = False
        for i, point in enumerate(X):
            dists = [sum((p-c)**2 for p,c in zip(point, center)) for center in centers]
            new_label = dists.index(min(dists))
            if new_label != labels[i]:
                labels[i] = new_label
                changed = True
        if not changed: break
        for ci in range(k):
            pts = [X[j] for j in range(len(X)) if labels[j] == ci]
            if pts:
                centers[ci] = [sum(dim)/len(pts) for dim in zip(*pts)]
    return labels

k_input = sys.argv[1] if len(sys.argv) > 1 else None
# 关键改进3：自动k值下限提升至8，上限20，公式 sqrt(n)
if k_input and k_input.strip():
    k = int(k_input)
else:
    k = max(8, min(20, int(math.sqrt(n))))
k = min(k, n)

labels = kmeans(X, k)

clusters = {}
for i, lbl in enumerate(labels):
    clusters.setdefault(str(lbl), []).append(file_names[i])
output = [{"cluster_id": cid, "members": members} for cid, members in clusters.items()]
print(json.dumps(output, ensure_ascii=False))
'''

n_cluster_json, act_python.run?filepath='{{n_py_file.kbspath}}'&argument='{{cluster_k}}'
