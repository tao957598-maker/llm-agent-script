// Main execution logic based on analysis type
when('{{analysisType == "Topic Report"}}', {
    // Generate topic report
    assign('report_title', 'Public Opinion Analysis Report: {{topicKeywords}}')
    
    // Define data collection parameters
    assign('collection_params', '{"keywords":"{{topicKeywords}}","period":"{{timePeriod}}","sources":{{fn.json(dataSources)}}}')
    
    // Collect data from multiple sources
    n_collected_data, act_kb.search?requirement='{{topicKeywords}}'&name='public_opinion_data'&maxSize=50
    
    when('{{fn.hasValue(n_collected_data)}}', {
        // Generate analysis report using LLM
        tpl_topic_analysis, psr_json
        
        // Format and present the report
        assign('formatted_report', '{{tpl_topic_analysis}}')
        act_memory.assign?var=analysis_result&value={{formatted_report}}&type=md
    }, {
        // No data found, trigger data collection
        assign('error_msg', 'No existing data found. Starting data collection...')
        log('{{error_msg}}')
        
        // Call data collection agent
        agt_data_collection?keywords={{topicKeywords}}&period={{timePeriod}}&async=true
        assign('collection_result', '{{agt_data_collection.value}}')
        
        // Retry analysis with collected data
        tpl_topic_analysis, psr_json
        assign('formatted_report', '{{tpl_topic_analysis}}')
        act_memory.assign?var=analysis_result&value={{formatted_report}}&type=md
    })
})

when('{{analysisType == "Configure Collection"}}', {
    // Configure data collection task
    assign('task_config', '''{
        "taskName": "Collection_{{fn.timestamp()}}",
        "keywords": "{{topicKeywords}}",
        "frequency": "daily",
        "sources": {{fn.json(dataSources)}},
        "startDate": "{{customStartDate}}",
        "endDate": "{{customEndDate}}"
    }''')
    
    // Create collection configuration
    n_task_config, act_kb.importData?name='collection_tasks'&data='[{{task_config}}]'
    
    // Schedule the task
    assign('job_data', '{"agentName":"agt_data_collection","jobDesc":"Public Opinion Collection","parameters":{{task_config}},"schedule":"0 0 * * *"}')
    n_job_created, act_job.createAndTrigger?data={{job_data}}&isOneOffTask=false
    
    assign('result_msg', 'Collection task configured successfully. Task ID: {{n_job_created.jobId}}')
    act_memory.assign?var=configuration_result&value={{result_msg}}
})

when('{{analysisType == "Monitor Task"}}', {
    // Monitor and manage existing collection tasks
    n_existing_tasks, act_kb.select?name='collection_tasks'&column='*'
    
    when('{{fn.hasValue(n_existing_tasks)}}', {
        // Display existing tasks
        tpl_task_monitor, psr_json
        
        assign('monitoring_result', '{{tpl_task_monitor}}')
        act_memory.assign?var=task_list&value={{monitoring_result}}&type=md
    }, {
        assign('no_tasks_msg', 'No active collection tasks found.')
        act_memory.assign?var=task_list&value={{no_tasks_msg}}
    })
})

// Define sub-agent for data collection
agent('agt_data_collection', {
    assign('collection_keywords', '{{keywords}}')
    assign('collection_period', '{{period}}')
    assign('collection_sources', '{{sources}}')
    
    // Simulate data collection from different sources
    assign('collected_items', '[]')
    
    foreach('n_source', 'collection_sources', {
        when('{{n_source.item == "News Websites"}}', {
            // Collect from news APIs
            n_news_data, act_api?name="news_collector"&params='{"keywords":"{{collection_keywords}}","period":"{{collection_period}}"}'
            append('collected_items', '{{n_news_data}}')
        })
        
        when('{{n_source.item == "Social Media"}}', {
            // Collect from social media APIs
            n_social_data, act_api?name="social_media_collector"&params='{"keywords":"{{collection_keywords}}","period":"{{collection_period}}"}'
            append('collected_items', '{{n_social_data}}')
        })
        
        when('{{n_source.item == "Forums"}}', {
            // Collect from forums
            n_forum_data, act_api?name="forum_collector"&params='{"keywords":"{{collection_keywords}}","period":"{{collection_period}}"}'
            append('collected_items', '{{n_forum_data}}')
        })
    })
    
    // Store collected data
    when('{{fn.hasValue(collected_items)}}', {
        n_stored_data, act_kb.importData?name='public_opinion_data'&data='{{collected_items}}'
        assign('collection_result', 'Successfully collected {{fn.size(collected_items)}} items')
    }, {
        assign('collection_result', 'No data collected')
    })
})

// Define prompt templates

// Error handling
on_failure({
    assign('error_result', 'Analysis failed: {{sys_last_error}}')
    act_memory.assign?var=error_output&value={{error_result}}
    act_flow.go?plan=end
})
