## AgentFlow 基础语法（基于简洁的groovy语法）
// 用途：编写Agent流程脚本，由引擎解析执行。脚本内变量可自定义或来自顶部Agent定义块 /*{ "parameter":[...] }*/，如 userInput。
// 术语：Plan=一条expression或等价链；n_xxx=步骤位置标记(用于 act_flow.go?plan=n_xxx 跳转)；tpl_xxx用于调用define定义的LLM模板；sys_=系统内置变量。assign/expression/append/when 等均为引擎提供，脚本中无需定义。
// 参数解析：URL 中可用 plainProperty=key1,key2（逗号分隔的参数字段名）指定这些字段按字面量传入、不做模板解析；须指定具体字段名。
### 变量与数据结构
// 注释为//开头的行
// 字符串
assign('var', 'num')
// 引用变量
assign('referenced_value', '{{var}}')
// 数字/布尔等字面量：用 fn.raw 保持类型写入，否则按字符串解析
assign('num', '{{fn.raw(100)}}')
// 嵌套模板使用%{}%语法
assign('nested_template', "{{fn.when('condition', '今天是2024-10-03', '今天是%{fn.today()}%')}}")
// 列表
assign('list', '["a","b"]')
// JSON/Map对象
assign('json', '{"name":"jack"}')
// 链式定义(变量1,值1,变量2,值2)
assign("product", "oa",
		"requirement", '''无
		''',
		"requirement_score", ">=7")
// 给对象的属性赋值（map 语法）
assign("user.name", "user1")
// 计划表达式：rag(检索)→tpl(LLM模板)→psr(解析)→vrf(校验)→act(执行)
expression('rag_xxx', 'tpl_xxx', 'psr_xxx', 'vrf_xxx', 'act_xxx', 'act_xxx2')
// 简写：一行仅含 rag_/tpl_/psr_/vrf_/act_/n_ 逗号列表时可省略 expression(...)
rag_xxx, tpl_xxx, psr_xxx, vrf_xxx, act_xxx, act_xxx2
// 通过memory工具赋值（简写语法)
act_memory.assign?var=result&value=xxx
// 拼接值val1,val2...给到变量var
append('var', 'val1', 'val2'...)
// 记录日志
log('log info {{var}}')

// vrf_xxx可接任意工具，如果tpl_xxx中有配置sys_hint，将激活自动修正，最多尝试3次
// 完整的Plan事务调用：读取知识库，调用LLM输出数据，抽取markdown内容，并检查LLM输出是否符合要求，最终保存为md_data变量，并在界面中的显示格式为markdown(注：最后的\为换行转义字符)
rag_kb.select?name=kb_md_example, tpl_md_output, psr_md, vrf_flag?value='{{fn.when(fn.hasValue(sys_last_result,"optional_sub_property_name"),"true","false")}}',\
	act_memory.assign?var=md_data&value={{tpl_md_output}}&type=md

### 控制流
when('条件', { ... })
elsewhen('条件', { ... })
foreach ('var_name', 'list', { ... })
loop('var_name', '条件', { ... })
// 异常处理
on_failure({ ... })
// 条件判断，{{}}中为freemarker表达式
when('{{product_id == compared_product_id}}', {
	assign("score", "9")
})
// when支持if-else（省略{{}}不推荐）
when('list?size gt 0',{
	assign("score", "9")
},{
	assign("score", "10")
})
elsewhen('{{condition}}', {
	assign("score", "11")
})
// elsewhen支持else（无参数时）
elsewhen({
	assign("fallback_score", "12")
})
// for 循环，基于普通list数据，或task list数据. n_plan_tasks为json结构, 包含属性：index, item, size
foreach('n_plan_tasks', 'llm_data', {
    // index 下标
    assign('loop_task_index', '{{n_plan_tasks.index}}')
    // 下标对应内容
    assign('loop_task_item', '{{n_plan_tasks.item}}')
})
// 结合条件判断的for循环，构建ReAct思维链
foreach('n_plan_analysis', 'fn.get("tpl_react.name")! != "finish" && n_plan_analysis.index < 5', {
	// tpl_react选择工具(api, agent etc)，交给plan运行
	tpl_react, act_plan.run?code={{tpl_react.name}}&parameters={{tpl_react.parameters}}
})
// 获取汇总的非空List结果
assign('n_summary', '{{n_plan_analysis.validSummary}}')
