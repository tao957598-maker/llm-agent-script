// 1. 参数处理
when('!fn.hasValue(city)', {
  assign("city", "北京")  // 默认查询北京
})

// 2. 调用天气API（真实接口）
n_run_sql_res, act_api.doExecute?name="tool_get_weather"&params='{"city":"{{city}}"}' 

// 3. 处理API响应
when('fn.hasValue(n_run_sql_res)', {
  // 成功获取数据时解析JSON
  assign("api_response", "{{n_run_sql_res}}")
  // final_resultfinal_resultfinal_result
  assign("final_result", "{'🌆 城市':{{api_response.data.position.city}},'🌆 天气':{{api_response.data.weather.weather}}}")
  
},{
  // API调用失败处理
  assign("final_result", "⚠️ 天气查询失败，请检查城市名称或稍后重试")
})


// 5. 最终输出
assign("final_result", "{{final_result}}")
