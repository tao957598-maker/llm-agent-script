n_modules, act_kb.selectBy?sql='SELECT distinct main_module_name,main_module_source FROM kb_gen_test_module WHERE documentId = "{{documentId}}" and version = "{{version}}"'

// 初始化统计变量
assign("total_count", "0")
assign("covered_count", "0")
assign("modules", '{{n_modules}}')
assign("modules_coverage_report", '[]')
// 循环处理每个main_module_name
foreach('n_module', 'modules', {
    // 获取当前模块名称
    assign('current_module', '{{n_module.item}}')
    // 定义main_module_name
    assign('main_module_name', '{{current_module.main_module_name}}')
    // 定义current_module_source
    assign('current_module_source', '{{current_module.main_module_source}}')
    // 页面数据
    n_pages_data,agt_get_page_data?page={{current_module_source}}&documentId="{{documentId}}"
    // 查询出所有的测试案例数据
    n_test_cases, act_kb.selectBy?sql='SELECT title,application,steps,test_no,expected_result,source_value,test_type FROM kb_gen_test_case3 WHERE main_module_name = "{{current_module.main_module_name}}" and documentId = "{{documentId}}" and version = "{{version}}"'
    // 计算数据
    n_report_data,tpl_coverage_report,psr_json
    
    act_func?script="modules_coverage_report.add(n_report_data)"

})

// 定义agent运行结果
assign("result", "{{modules_coverage_report}}")
