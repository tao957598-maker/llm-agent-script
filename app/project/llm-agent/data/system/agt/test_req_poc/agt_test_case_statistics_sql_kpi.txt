n_sql_kpi_data, act_kb.selectBy?sql='''
SELECT
    design_element,
    sub_id,
    COUNT(*) AS total_cases,
    SUM(CASE WHEN test_type = 1 THEN 1 ELSE 0 END) AS positive_cases,
    SUM(CASE WHEN test_type = -1 THEN 1 ELSE 0 END) AS negative_cases,
    ROUND(SUM(CASE WHEN test_type = 1 THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 2) AS positive_ratio,
    ROUND(SUM(CASE WHEN test_type = -1 THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 2) AS negative_ratio
FROM kb_gen_test_case
WHERE documentId = '{{documentId}}' and version = "{{version}}"
GROUP BY sub_id,design_element
'''

n_md_table, act_func?script='''
def md = new StringBuilder()
// 添加表头
md.append("| design_element | sub_name | total_cases | positive_cases | negative_cases | status | proportion |\\n")
md.append("|--|--|--|--|--|--|--|\\n")
// 添加数据行
n_sql_kpi_data.each { row ->
    def sub = kpi_dc[row.sub_id]
    if(sub) {
        sub_name = sub.name
        positive_ratio = sub.positive_ratio
        row_positive_ratio = row.positive_ratio
        status = Math.abs(positive_ratio - row_positive_ratio) <= 5
        status_value = status ? "True" : '<span style="color:red">False</span>'

        md.append("| " + row.design_element + " | " + sub_name + " | " + row.total_cases + " | " + row.positive_cases + " | " + row.negative_cases +" | " + status_value + " | " + sub.proportion + " |\\n")
    }
}
return md.toString()
'''
