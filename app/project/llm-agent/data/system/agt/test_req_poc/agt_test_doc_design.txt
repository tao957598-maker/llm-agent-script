    n_pages_data, act_kb.selectBy?sql='''
    select 
    page,
    CASE
        WHEN analysis_content IS NULL OR analysis_content = '' THEN content
        ELSE analysis_content
    END AS content
    from kbs_doc_page_parse 
    where documentId="{{documentId}}" and page in {{n_pages.pages}} order by page asc'''

    n_get_design,tpl_get_design2
    tpl_design_rm
    // 获取大模型处理结果
    assign('processed_data', '{{tpl_design_rm.data}}')
    assign('processed_data_list', '[]')
    foreach('n_data', 'processed_data', {
        assign('value', '{{n_data.item}}')
        n_c_map_list,act_func?script='''
        def result = []
        if(value.test_content instanceof List) {
            def contentList = value.test_content as List
            contentList.each { item ->
                if(item instanceof List) {
                    def itemList = item as List
                    if(itemList.size() >= 2) {
                        result.add([
                            "sub_category": itemList[0]?.toString() ?: '',
                            "sub_id": itemList[1]?.toString() ?: ''
                        ])
                    }
                }
            }
        }
        value.test_content = result
        return value
        '''
        act_func?script='''processed_data_list.add(value)'''
    })
    // 将结果存入input_data
    act_func?script="design_input_data.put('data', processed_data_list)"
    assign("final_result",'{{design_input_data}}')
})
