// 2. Get info for KB
n_kb_datas,act_kb.selectBy?sql='SELECT knowledge_name,extension,store_uri FROM kb_spaces_folder_knowledge WHERE id in ({{kbSpaceFileId}})'


// 使用展平模块数据
assign('result_list', '[]')
// 3. Recognize Cheque by LLM
foreach('kb_data', 'n_kb_datas', {
    // 在循环内部可以访问循环变量
    assign('current_index', '{{kb_data.index}}')
    assign('current_item', '{{kb_data.item}}')
	assign('file_name', '{{current_item.knowledge_name}}.{{current_item.extension}}')
	assign('image_path', '{{current_item.store_uri}}')
    
	// Recognize Cheque by LLM
	n_datas,tpl_image_chat,psr_json

	// Import Data
	n_import_data,act_func?script='''
		def new_datas = n_datas ? [:] + n_datas : [:]
		new_datas.fileName = file_name
		new_datas.imagePath = image_path
		new_datas.is_modified = new_datas.is_modified ? 1 : 0
		new_datas.signature_exists = new_datas.signature_exists ? 1 : 0
		return new_datas
	'''
  act_func?script='''
            result_list.add(n_import_data)
        '''
 
})

//# data
assign('data', '{{result_list}}')
