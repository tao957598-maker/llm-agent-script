	"parameter": [
		{"name":"userInput", "label":"用户输入", "required":true},
		{"name":"loginUserName", "label":"登录用户名"},

		{"name":"allowHighRiskAction", "label":"是否允许高风险动作"}

assign('opsclaw_root', '',
	'allow_high_risk_action_safe', 'false')

when('fn.hasValue(allowHighRiskAction)', {
	assign('allow_high_risk_action_safe', '{{allowHighRiskAction}}')
})

when('fn.hasValue(loginUserName)', {
	assign('opsclaw_root', 'user/{{loginUserName}}/appdata/opsclaw')
})

// 通过独立 agent 加载 profile / key-memories / 语义检索上下文
n_load_memory_result, agt_opsclaw_memory_load.opsclaw?userInput={{userInput}}&opsclaw_root={{opsclaw_root}}&loginUserId={{loginUserId}}

// 优先基于平台稳定能力发现 API；agent 级动态工具发现后续可逐步接入
n_api_candidates, act_api.search?query='{{userInput}}'&size=8&toolType='API'

n_candidates_display,act_func.run?script='''
    def display = ''
    if (n_api_candidates && n_api_candidates.size() > 0) {
        n_api_candidates.eachWithIndex { item, idx ->
            def displayItem = "${idx + 1}. [${item.code}] ${item.description}\\n"
            display += displayItem
        }
    } else {
        display = 'No matching API candidates found.\\n'
    }
	return display
'''

//## 开始搜索与用户输入最匹配的 API 候选（最多 8 条）
assign('n_candidates_display',"{{n_candidates_display}}")

// 给提示词显式暴露“动态工具集”这个候选 agent，避免模型只盯着 API 候选。
assign('n_dynamic_toolset_guide', '''
- type: AGENT
- code: agt_all_scenario_dynamic_toolset.toolset
- usage: 当无法直接确定合适的 API 或 agent 时，调用该 agent 让系统基于用户问题动态组织工具集，并继续完成工具选择、参数提取与执行。
- requiredParams:
  {
    "userInput": "{{userInput}}"
  }

- type: AGENT
- code: agt_opsclaw_memory_write.opsclaw
- usage: 当用户说「帮我记一下 xxx」「记住 xxx」或「记错了，应该是 yyy」时调用；由该 agent 内 LLM 决定追加或替换，无需调用方区分。
- requiredParams:
  {
    "opsclaw_root": "{{opsclaw_root}}",
    "memoryType": "profile 或 key_memories",
    "content": "用户原始表述（如「帮我记一下我喜欢用 Python」或「记错了，应该是 Java」）"
  }
''')

// 把“直接走 API”与“升级到动态工具集”的路由规则固化到提示词里，减少随机决策。
assign('n_tool_routing_guide', '''
## 工具路由规则

### 优先直接使用 `act_api.search` 产出的 API 候选
满足以下任一组合时，优先选择 `RUN_TOOL + API`，不要调用 `agt_all_scenario_dynamic_toolset.toolset`：
1. `n_api_candidates` 中已经有单个明显匹配的 API，且用户目标是单步完成的查询、提交、更新、触发。
2. 用户问题明确指向某类系统能力，API 名称或描述与用户意图高度一致。
3. 所需参数相对直接，可根据用户输入或上下文直接填写，不需要再次做复杂工具编排。
4. 当前任务更像“调用一个现成接口”而不是“先找一组工具再决定怎么做”。

### 优先调用 `agt_all_scenario_dynamic_toolset.toolset`
满足以下任一情况时，优先选择 `RUN_TOOL + AGENT(agt_all_scenario_dynamic_toolset.toolset)`：
1. `n_api_candidates` 为空、过于泛化、或没有明显高匹配项。
2. 用户问题本质是“有哪些工具可用”“应该调用哪个 agent/API”“帮我自动找合适工具”。
3. 任务可能需要先发现 agent，再做参数提取与执行，而不是直接命中单个 API。
4. 任务跨多个领域，单个 API 很难覆盖，可能更适合由动态工具集先做聚合和分派。
5. 当前上下文不足以确认应走 API、AGENT 还是其他工具类型。

### 优先调用 `agt_opsclaw_memory_write.opsclaw`（写入记忆）
当用户明确要求「记一下/帮我记住/记住/写进记忆」或「记错了/应该是/修正为」时，优先选择 `RUN_TOOL + AGENT(agt_opsclaw_memory_write.opsclaw)`，parameters 填 opsclaw_root、memoryType、content（用户原话）；追加或替换由 memory_write 内 LLM 决定。

### 不要调用 `agt_all_scenario_dynamic_toolset.toolset` 的情况
1. 已经存在明确可执行的 API 候选，且本轮目标只是执行该 API。
2. 最近一次 Observation 已经足够回答用户，无需继续工具发现。
3. 当前只是对已有执行结果做总结、解释、转述。
''')

assign('n_init_result', '{{fn.json("opsclaw_root, opsclaw_root, allow_high_risk_action_safe, allow_high_risk_action_safe, n_load_memory_result, n_load_memory_result, n_api_candidates, n_api_candidates, n_dynamic_toolset_guide, n_dynamic_toolset_guide, n_tool_routing_guide, n_tool_routing_guide")}}')
