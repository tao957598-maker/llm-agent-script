// Use selenium
assign('pythonType', 'selenium', 'subfolder', '{{test_no}}', 'sys_hint_agt', '')

//# Get the page information
agt_collect_page

// Record orignal page information
assign('last_simplepageinfo', '{{StepTestInfo.simplepageinfo}}')

// Determine whether the check element exists
tpl_check_element_exists, vrf_flag?value='sys_last_result.thought.re_reasoning'&cmpValue=false

// Check element exists flag
when('tpl_check_element_exists.has_element', {
	//# Call element check agent to verify element
	agt_test_checkelement?test_no={{test_no}}&current_step={{current_step}}
	
	// Found target element
	act_flow.go?plan=end
})

// ==== The analysis process officially begins ====
//# Call agent to collect the new page information
n_collect_page, agt_collect_page

//# Make decision
tpl_make_plan

//# Call agent to verify decision
agt_verify_plan

// Handle possible error
on_failure('agt_verify_plan', {
	// Try it again from beginning
	act_flow.go?plan=n_collect_page&hintError={{fn.not(code_error)}}
})

//# Final test - call agent to collect the new page information
agt_collect_page

//# Final test - Analyze and check if element exists
tpl_check_element_exists, vrf_flag?value='sys_last_result.thought.re_reasoning'&cmpValue=false

// Final test - Check flag
when('fn.not(tpl_check_element_exists.has_element)', {
	// Analyze the failure reason
	tpl_cmp_pageresult
	
	// Try again from beginning
	act_flow.go?plan=n_collect_page
}, {	
	//# Ouput the python script result
	n_python_file, act_python.write?requirement={{current_step}}
	
	// Write execution result
	act_kb.merge?name=kb_testcase_step&columnValues={{fn.json('result,n_python_result.screenshot_filename', true)}}&where={{fn.json('test_no,test_no,step,current_step')}}
})

// Verify plan through python code
agent('agt_verify_plan', {
	//# Analyse page elements
	tpl_task_element_info, psr_json?filter=id&format='id_\\d+'
	
	//# Generate python code according to decision
	tpl_generate_verify_script, psr_json_python, act_python.writeTmp?pythonCode={{sys_last_result}}
		
	//# Run selenium script through RPA client
	n_python_result, act_rpa.send?name={{python_file}}
	
	// Handle the error
	on_failure({
		// Check hint condition
		act_memory.canHint?hintCount=3
		
		// Call agent to fix python code
		agt_fix_python_code?hint_info={{sys_hint_agt}}
	})
})

// Fix python code
agent('fix_python_code', {
	// Check if code exists
	act_memory.hasValue?name=python_code
	
	//# Try to fix python code
	tpl_self_fix, psr_json_python
	
	// Check code error flag
	when('sys_last_result=="NO_FIX"', {
		// Update code error flag
		assign('code_error', 'false')
		
		// Mark result as failed and prepare to try again
		act_flow.fail
	}, {
		// Update code error flag
		assign('code_error', 'true')
		
		// Generate temporary python code
		act_python.writeTmp?pythonCode={{tpl_self_fix}}
		
		// Run fix script through RPA client
		n_python_result, act_rpa.send?name={{python_file}}
	})
})

// Collect page information
agent('agt_collect_page', {
	// Collect the new page information
	act_rpa.send?command=CollectPageInfo
		
	// Populate StepTestInfo structure
	act_func?name=StepTestInfo
})
