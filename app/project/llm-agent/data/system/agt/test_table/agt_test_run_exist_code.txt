// Use selenium
assign('pythonType', 'selenium', 'subfolder', '{{test_no}}')

//## Prepare to run test step
n_python_file, act_python.read?pythonType={{pythonType}}&requirement={{current_step}}&subfolder={{subfolder}}

// Check script type
when("python_code=='jsonl'", {
	//## Call new Agent to run subtasks
	agt_test_steps_run?test_no={{sub_test_no}}&system={{system}}
}, {
	// Check if script is generated
	when("!n_python_file.hasFilename()", {
		//## Check similiar file, score similar 0 is perfectly matched
		n_similar_file, act_python.readSimilar?pythonType={{pythonType}}&requirement={{current_step}}&filePath={{subfolder}}&score='0.05'
		// resultCode: 0:不匹配  1:完全匹配  2:三元组 3:向量库匹配 去掉这个功能 代码没有审核
		// Check if it is reusable
		when('n_similar_file.resultCode>0', {
			//# Collect page information
			n_collect_page, act_rpa.send?command=CollectPageInfo\
				act_func?name=StepTestInfo
				
			// Check the similarity metric
			when('n_similar_file.resultCode==2', {
				//## Reuse code directly
				agt_run?python_file={{n_similar_file.filename}}&parameter={{n_similar_file.parameter}}&current_step={{current_step}}
				
				//## Write selenium script to the artifact repository
				n_python_file_2, act_python.write?requirement={{current_step}}&pythonCode={{fn.readFile(n_similar_file.fullFilename)}}
			}
			
			)
			when('n_similar_file.resultCode==3', {
				act_flow.fail 
			}
			
			)
			
			//# Collect page information again
			n_collect_page_again, act_rpa.send?command=CollectPageInfo

			//# Check and compare the execution result
			n_cmp1, tpl_cmp_pageresult, vrf_flag?value='fn.str(sys_last_result.thought.re_reasoning)=="false" || sys_last_result.hasDialog',\
				agt_save_explorer?columns='page_change,tpl_cmp_pageresult.thought.change,reasoning,tpl_cmp_pageresult.thought.reasoning'

			// Check if page doesnot change
			when('fn.not(n_cmp1.run_success)', {
				// Show warning information
				// log('Failed to verify page result', true)
				
				// Proceed to error handling process
				act_flow.go?plan=n_fail_rpa_coll
			}, {
				// When the page changes and no dialog appears
				when('fn.hasValue(n_cmp1.change_element) && n_cmp1.change_behavior=="add" && !fn.bool(n_cmp1.hasDialog)', {
					// Add page verify element
					tpl_self_add_wait, psr_python, act_memory.assign?var=python_code
					
					n_gen_file, act_python.writeTmp?requirement={{current_step}}&pythonCode={{sys_last_result}}
				})
			})

			//# Write selenium script to the artifact repository
			n_python_file_3, act_python.write?requirement={{current_step}}
			
			// Result
			assign('result', '{{n_python_file_3.fullFilename}}')
		}, {
			// Anounce no code found
			act_flow.fail?result='No reusable code'&message=true
		})
	}, {
		// Load test data
		n_testdata, act_kb.select?name=kb_testcase_data&where={{fn.json('test_no,test_no,name,system')}}
		
		//## Run selenium code
		agt_run?python_file={{n_python_file.filename}}&parameter={{n_testdata.data}}
	})
})


// Parameters: 1.python_file; 2.parameter, 3. current_step
// Agent for run script
agent('agt_run', {
	//## Run test step
	n_python_result, act_rpa.send?name={{python_file}}&command=RunScript&parameter={{parameter}}
	
	// Save execution result to KB
	act_kb.merge?name=kb_testcase_step&columnValues={{fn.json('result,n_python_result.screenshot_filename', true)}}&where={{fn.json('test_no,test_no,step,current_step')}}
	
	// Result
	assign('result', '{{pythonType}}/{{python_file}}')
	
	// 2表示顺序执行，直到一个可成功执行的任务
	// Try to run test step again
	on_failure({
		// Waiting 5 seconds
		expression(true, 'act_flow.wait?second=5')
		
		// Try to Run test step again
		n_python_result2, act_rpa.send?name={{python_file}}&command=RunScript
	}, 2)
})

// Save explorer
agent('agt_save_explorer', {
	// Save to kb_testcase_step_explorer
	act_kb.merge?name=kb_testcase_step_explorer&columnValues='''
			{{fn.json(columns)}}
		'''&where='''
			{{fn.json("test_no, test_no, step, StepTestInfo.current_step")}}
		'''
})
