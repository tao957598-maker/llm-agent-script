n_data_item, act_func?js='''
(function() {
    let arrList = {{n_kb_data}};
    let nData_Arr = [];
    let nResult_Arr = [];
    let nTitle_Arr = [];
    let n_Method_Arr = [];
    for(let i = 0; i < arrList.length; i++) {
        let jsonArr = JSON.parse(arrList[i].data);
        for(let j = 0; j < jsonArr.length; j++) {
            nData_Arr.push(jsonArr[j]);
            nResult_Arr.push(arrList[i].expected_result);
            nTitle_Arr.push(arrList[i].title);
            n_Method_Arr.push(arrList[i].test_method);
            
        }
       
        
    }
    
    let obj = {
        "title": JSON.stringify(nTitle_Arr),
        "test_no": arrList[0].test_no,
        "system_name": arrList[0].system_name,
        "metadata_id": arrList[0].metadata_id,
        "test_method": JSON.stringify(n_Method_Arr),
        "data": JSON.stringify(nData_Arr),
        "expected_result": JSON.stringify(nResult_Arr)
    };
    
    return JSON.stringify(obj);
})();
'''

assign("n_result_data", "{{n_data_item}}")
