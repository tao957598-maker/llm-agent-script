    def arrList = n_kb_data;
    def newArr = new ArrayList<>();
    def nResArr = new ArrayList<>();
    def nTitleArr = new ArrayList<>();
    def test_step = "";

    for (int i = 0; i < arrList.size(); i++) {
        def dataList = com.alibaba.fastjson2.JSON.parseObject(arrList[i].data, java.util.List.class);
        for (int j = 0; j < dataList.size(); j++) {
            newArr.add(dataList[j]);
        }

        nResArr.add(arrList[i].expected_result);
        if(i == 0){
            nTitleArr.add(arrList[i].title);
            test_step = arrList[i].steps;
        }
    }

    // 构造最终对象
    def obj = new HashMap();
    obj.put("title", nTitleArr);
    obj.put("test_no", test_no);
    obj.put("system_name", system);
    obj.put("metadata_id", arrList[0].metadata_id);
    obj.put("test_method", arrList[0].test_method);
    obj.put("data", newArr);
    obj.put("test_step", test_step);
    obj.put("expected_result", nResArr);
    obj.put("scene", module_name);
    obj.put("version", version);
    obj.put("tag", testTag);

    return obj
'''
