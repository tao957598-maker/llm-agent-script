    def isBlank = { v -> v == null || (v instanceof String && v.trim().isEmpty()) }
    def norm = { v -> (v == null) ? "" : v.toString().trim().replaceAll(/\s+/, " ").toLowerCase() }
    def containsAny = { s, arr ->
        def x = norm(s)
        for (k in arr) { if (x.contains(k)) return true }
        return false
    }

        "project management",

        if (it == null) return
        def reqType = norm(it["requirement_type"])
        def section = it["section"]
        def title = it["title"]
        def desc = it["description"]

        // 跳过明显的非需求/管理信息
        if (containsAny(section, blacklist) || containsAny(title, blacklist) || containsAny(desc, blacklist)) return
        if (containsAny(title, ["project management / task tracking"])) return

        if (isBlank(title) && isBlank(desc)) return

        if (seen.contains(key)) return
        seen.add(key)
        filtered.add(it)
    }
    page_summary_list = filtered
'''

// Step1 去重（先不遗漏）：仅合并“同主模块 + 同子模块名(规范化后)”的同义行，来源页码做并集
act_func?script='''
    def normName = { s ->
        def x = (s == null ? "" : s.toString()).toLowerCase()
        x = x.replaceAll(/[\\[\\]【】()]/, " ")
        x = x.replaceAll(/\\s*-\\s*/, " ")
        x = x.replaceAll(/\\s+/, " ").trim()
        return x
    }
    def mergePages = { a, b ->
        def ps = new LinkedHashSet()
        [a, b].each { v ->
            def t = (v == null ? "" : v.toString())
            t.split(/\\s*,\\s*/).each { p ->
                def pp = p == null ? "" : p.trim()
                if (pp) ps.add(pp)
            }
        }
        return ps.join(", ")
    }
    def merged = new LinkedHashMap()
    result_list.each { r ->
        if (!(r instanceof Map)) return
        def k = normName(r.main_module_name) + "||" + normName(r.submodule_name)
        if (!merged.containsKey(k)) {
            merged.put(k, r)
        } else {
            def ex = merged.get(k)
            ex.main_module_source = mergePages(ex.main_module_source, r.main_module_source)
            ex.submodule_source = mergePages(ex.submodule_source, r.submodule_source)
            def d1 = (ex.submodule_description ?: "").toString().trim()
            def d2 = (r.submodule_description ?: "").toString().trim()
            if (d2.length() > d1.length()) ex.submodule_description = d2
            merged.put(k, ex)
        }
    }
    result_list = new ArrayList(merged.values())
'''
