    act_func?script='''
        def pageTxt = ""
        try {
            if (n_pages_data != null) pageTxt = n_pages_data.toString().toLowerCase()
        } catch (Exception ignore) { }

        def modTxt = ((input_data.submodule_name ?: "") + " " + (input_data.submodule_description ?: "") + " "
            + (input_data.submodule_source ?: "") + " " + (input_data.main_module_source ?: "") + " "
            + (input_data.main_module_name ?: "")).toLowerCase()
        def txt = (modTxt + " " + pageTxt)
        def hasAny = { s, ks -> ks.any { k -> s.contains(k) } }

        // Step2 收口但不漏：按模块类型设置默认候选，避免非界面模块被 UI 类测试点放大
        def moduleText = (((input_data.submodule_name ?: "") + " " + (input_data.submodule_description ?: "")).toLowerCase())
        def isNonInterface = moduleText.contains("non-interface")
        def allow = isNonInterface
            ? new LinkedHashSet(["31","32","40","43"])
            : new LinkedHashSet(["13","21","31","32","40","43"])

        if (hasAny(txt, ["button","click","upload","download","navigate","input step","guide","guideline","link","template"])) {
            allow.add("10")
            allow.add("11")
        }

        if (hasAny(txt, ["multi-language","multi language","three language","en"," tc"," sc","en/tc","tc/sc","localization","i18n","简体","繁体","英文","traditional chinese","simplified chinese"])) {
            allow.add("15")
        }

        if (hasAny(txt, ["register","registration","access control","permission","authorized","profile","entitle"])) {
            allow.add("41")
            allow.add("42")
            allow.add("44")
        }

        if (hasAny(txt, ["browser","ie","edge","chrome","safari","firefox","ios","android","harmonyos","device"])) {
            allow.add("16")
        }
        if (hasAny(txt, ["font","icon","spacing","visual"])) {
            allow.add("17")
        }
        if (hasAny(txt, ["responsive","resolution","layout","横竖屏"])) {
            allow.add("18")
        }

        processed_data_list.each { row ->
            def tc = row.test_content instanceof List ? (row.test_content as List) : []
            def filtered = tc.findAll { m ->
                def id = (m["sub_id"] ?: "").toString()
                allow.contains(id)
            }
            def seen = new LinkedHashSet()
            def deduped = []

            def idSet = new LinkedHashSet(deduped.collect { (it["sub_id"] ?: "").toString() })

            return hit ?: (processed_data_list ? processed_data_list[0] : null)
        }

        def target = pickRow()
        if (target != null) {

            if (needNavInteract && !allIds.contains("10") && !allIds.contains("11")) {
                tc.add(["sub_category": "Navigation flow: menu hierarchy, sub-page navigation", "sub_id": "10"])

                tc.add(["sub_category": "Localization and internationalization: EN TC SC UI text", "sub_id": "15"])

            fallbackIds.each { fid ->

                    tc.add(["sub_category": (idNameMap[fid] ?: ("Category " + fid)), "sub_id": fid])
                    localSeen.add(fid)
                }

            row.test_content = tc

            def rank = { sid ->

                return i >= 0 ? i : 999
            }

            def maxKeep = 8

            row.test_content = deduped

            def tc = row.test_content instanceof List ? (row.test_content as List) : []
            return tc.size() > 0
        }
