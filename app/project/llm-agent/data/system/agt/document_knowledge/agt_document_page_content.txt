// 1. 读取数据库信息
assign('kbSpaceFileId', '{{kbSpaceFileId}}')
assign('pages', '{{pages}}')
when('!fn.hasValue(pages)', {
//# 取数据 排除空内容
n_kb_data,act_kb.selectBy?sql='''select
    *
from
    (SELECT
    page,
    CASE
        WHEN analysis_content IS NULL OR analysis_content = '' THEN content
        ELSE analysis_content
    END AS analysis_content
FROM
    kbs_doc_page_parse
WHERE
    documentId = "{{kbSpaceFileId}}"
ORDER BY
    CAST(page AS SIGNED) ASC) as a where a.analysis_content is not null'''
},{
n_kb_data,act_kb.selectBy?sql='''select
    *
from
    (SELECT
    page,
    CASE
        WHEN analysis_content IS NULL OR analysis_content = '' THEN content
        ELSE analysis_content
    END AS analysis_content
FROM
    kbs_doc_page_parse
WHERE
    documentId = "{{kbSpaceFileId}}"
    and page in ({{pages}})
ORDER BY
    CAST(page AS SIGNED) ASC) as a where a.analysis_content is not null '''
})
