// 构建SQL查询条件
assign("sqlCondition", '')

// 执行SQL查询
n_queryResult, act_kb.selectBy?sql='SELECT fileName, page, COALESCE(analysis_content, content) AS content FROM kbs_doc_page_parse WHERE documentId IN ({{kbSpaceFileId}})'
when('!fn.hasValue(n_queryResult)', {
	// 无查询结果的情况
	n_file_name, act_kb.selectBy?sql='select  file_name,store_uri  from kb_spaces_folder_knowledge where id="{{kbSpaceFileId}}" '
	n_file_data,act_func.read?filename="{{n_file_name[0].store_uri}}"
	assign("file_name", '{{n_file_name[0].file_name}}');
	assign("file_content", '{{n_file_data}}');
	assign("queryResult", '{{fn.json("file_name,file_name,file_content,file_content")}}')
},{
	// 有查询结果的情况
	assign("file_name", '{{n_queryResult[0].fileName}}');
	assign("file_content", '{{n_queryResult[0].content}}');
	assign("queryResult", '{{fn.json("file_name,file_name,file_content,file_content")}}')
}
)
// 返回查询结果
assign("result", '{{queryResult}}')
