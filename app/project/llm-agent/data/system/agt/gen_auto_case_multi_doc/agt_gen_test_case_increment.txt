act_func?script='''
    def splitCommaList = { raw ->
        def s = (raw == null ? "" : raw.toString()).trim()
        def out = []
        if (!s) return out
        def current = new StringBuilder()
        for (int i = 0; i < s.length(); i++) {
            def ch = s.charAt(i)
            if (ch == ',') {
                def token = current.toString().trim()

                current.setLength(0)
            } else {
                current.append(ch)
            }
        }
        def token = current.toString().trim()
        if (token) out.add(token)
        return out
    }
    def normalizePageKey = { v ->
        def x = (v == null ? "" : v.toString()).trim().toLowerCase()
        if (x.startsWith("page")) x = x.substring(4).trim()
        return x
    }
    def singlePageText = { inputVal ->
        def s = (inputVal == null ? "" : inputVal.toString()).trim()
        if (!s) return ""
        def token = s.contains(",") ? s.split(",")[0].trim() : s
        if (!token) return ""
        if (token.toLowerCase().startsWith("page")) {
            token = token.substring(4).trim()
        }
        return token
    }
    def scopeKeys = fn.get("increment_scope_page_keys")
    if (!(scopeKeys instanceof Set)) scopeKeys = new LinkedHashSet()
    def pickBestContent = { r ->
        if (!(r instanceof Map)) return ""
        def analysis = r.get("analysis_content")
        def content = r.get("content")
        def picked = fn.hasValue(analysis) ? analysis : content
        return picked == null ? "" : picked.toString().trim()
    }
    def aggregateRowsByPage = { rows ->
        def pageMap = new LinkedHashMap()
        (rows instanceof List ? rows : []).each { r ->
            if (!(r instanceof Map)) return
            def pk = normalizePageKey(r.get("page"))
            if (!pk) return
            if (!scopeKeys.isEmpty() && !scopeKeys.contains(pk)) return
            def pageLabel = singlePageText.call(r.get("page"))
            def content = pickBestContent(r)
            if (!fn.hasValue(content)) return
            def bucket = pageMap.get(pk)
            if (bucket == null) {
                bucket = [page: pageLabel, parts: []]
                pageMap.put(pk, bucket)
            }
            if (!fn.hasValue(bucket.page) && fn.hasValue(pageLabel)) {
                bucket.page = pageLabel
            }
            bucket.parts.add(content)
        }
        def out = new LinkedHashMap()
        pageMap.each { k, bucket ->
            def joined = (bucket.parts instanceof List ? bucket.parts : []).join("\n")
            out.put(k, [
                page: (bucket.page ?: "").toString().trim(),
                content: joined,
                part_count: (bucket.parts instanceof List ? bucket.parts.size() : 0)
            ])
        }
        return out
    }
    def oldRows = n_old_parse_rows
    if (!(oldRows instanceof List)) oldRows = []
    def newRows = n_new_parse_rows
    if (!(newRows instanceof List)) newRows = []
    def oldMap = aggregateRowsByPage(oldRows)
    def newMap = aggregateRowsByPage(newRows)
    def commonPages = []
    def addPages = []
    def deletePages = []
    def promptLines = []
    promptLines.add("Compare the following page pairs semantically.")
    promptLines.add("Return a pure JSON array only.")
    promptLines.add("Each item must contain page, status, and reason.")
    promptLines.add("Use Same when the page meaning, structure, and paragraph semantics are the same even if formatting, OCR noise, or punctuation differs.")
    promptLines.add("Use Update when the page meaning or structure changed materially.")
    promptLines.add("Pages that exist only in the new document are Add.")
    promptLines.add("Pages that exist only in the old document are Delete.")
    newMap.each { k, n ->
        def o = oldMap.remove(k)
        if (o == null) {
            addPages.add(n.page)
        } else {
            commonPages.add([
                page_key: k,
                page: n.page,
                old_content: o.content,
                new_content: n.content,
                old_len: o.content.length(),
                new_len: n.content.length(),
                old_parts: (o.part_count ?: 0),
                new_parts: (n.part_count ?: 0)
            ])
            promptLines.add("=== PAGE " + n.page + " ===")
            promptLines.add("OLD:")
            promptLines.add(o.content)
            promptLines.add("NEW:")
            promptLines.add(n.content)
            promptLines.add("")
        }
    }
    oldMap.each { k, o ->
        deletePages.add(o.page)
    }
    def promptData = promptLines.join("\n").trim()
    fn.put("increment_add_pages", addPages)
    fn.put("increment_delete_pages", deletePages)
    fn.put("increment_common_pages", commonPages)
    fn.put("increment_page_diff_prompt_data", promptData)
    fn.put("increment_page_diff_prompt_pages", commonPages.collect { it.page })
'''

define('tpl_increment_page_diff', '''
You are comparing old and new document pages for incremental testcase generation.

Decide the semantic status for each common page pair.

Rules:
- Use `Same` when the meaning, structure, and paragraph semantics are the same even if spacing, punctuation, OCR noise, or minor formatting differ.
- Use `Update` when the meaning or structure changed materially, sections were added or removed, or the page content is clearly different.
- Use `Add` only for pages present only in the new document.
- Use `Delete` only for pages present only in the old document.
- Prefer semantic equivalence over literal string comparison.
- Return pure JSON only.
- Return an array of objects.
- Each object must have keys: `page`, `status`, `reason`.
- `status` must be one of `Same`, `Update`, `Add`, `Delete`.
- Keep `reason` short.
- If there are no common pages, return `[]`.

Page pairs:
{{increment_page_diff_prompt_data}}
''')
