        if (!txt) return out

        if (!m.matches()) {
            m = (txt =~ /^(\d+)(?:\s*-\s*(\d+))?$/)
            if (!m.matches()) return out
        }
        int start = Integer.parseInt(m[0][1].toString())
        int end = m[0][2] ? Integer.parseInt(m[0][2].toString()) : start
        if (end < start) { int tmp = start; start = end; end = tmp }
