n_func_modules_compact_input,act_func?script='''
    def clip = { v, maxLen ->
        def s = v == null ? "" : v.toString().trim()
        if (!s) return ""
        return s.length() <= maxLen ? s : s.substring(0, maxLen)
    }
    def takeList = { v, maxItems, itemLen ->
        def out = []
        if (v instanceof List) {
            for (int i = 0; i < v.size() && i < maxItems; i++) {
                def t = clip(v.get(i), itemLen)
                if (t) out.add(t)
            }
        } else {
            def t = clip(v, itemLen)
            if (t) out.add(t)
        }
        return out
    }
    def rows = input_data
    if (rows == null) return []
    if (!(rows instanceof List)) rows = [rows]
    def out = []
    rows.each { item ->
        if (!(item instanceof Map)) return
        out.add([
            platform: clip(item.get("platform"), 60),
            source_type: clip(item.get("source_type"), 20),
            section: clip(item.get("section"), 120),
            title: clip(item.get("title"), 160),
            description: clip(item.get("description"), 320),
            business_rules: takeList(item.get("business_rules"), 4, 120),
            acceptance_criteria: takeList(item.get("acceptance_criteria"), 4, 120),
            dependencies: takeList(item.get("dependencies"), 4, 80),
            page: clip(item.get("page"), 40),
            source: clip(item.get("source") ?: item.get("source_ref"), 120)
        ])
    }
    return out
'''

define('tpl_get_functional', '''
You are generating a compact functional module tree for test-case generation.

Goal:
- Convert the input functional summary list into a stable JSON module tree.
- Keep output bounded and evidence-based.

Hard rules:
1. Output JSON only.
2. Output language: {{lang_output_text}}.
3. `platform` must be exactly `DEMO-platform`.
4. Do not invent modules or submodules not supported by the input.
5. Do not split by wording variants, examples, labels, or channel notes alone.
6. Prefer merge over split when two items represent the same executable test intent.
7. Keep descriptions short, usually one sentence.
8. Keep total output size close to the number of distinct functional intents in the input.
9. Preserve `page` and `source`. If `source` contains `fileName#PageN`, keep that format.
10. Each submodule must be directly traceable to the input.

Output format:
```json
{
  "platform": "DEMO-platform",
  "modules": [
    {
      "name": "Main module name",
      "page": "1,2",
      "source": "Page1 or fileName#Page1",
      "submodules": [
        {
          "name": "Submodule name",
          "description": "Short functional description",
          "page": "1,2",
          "source": "Page1 or fileName#Page1"
        }
      ]
    }
  ]
}
```

Input data:
{{n_func_modules_compact_input}}
''')
