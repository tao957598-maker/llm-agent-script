assign('n_relationship_cypher', 'MATCH (n:{{graph_content}})-[r]->() RETURN DISTINCT type(r) as relationship_type')
// 1.查询出当前图谱库中所有的关系类型
n_level_relationship,act_kb.searchGraph?cypher='{{n_relationship_cypher}}'

// 2.查询出标签下节点的参数信息
assign('n_label_node_field_cypher', 'MATCH (n:{{graph_content}}) RETURN properties(n) AS nodeProperties limit 1')
n_label_node_field,act_kb.searchGraph?cypher='{{n_label_node_field_cypher}}'

// 3.通过gpt生成cypher查询语句
n_result_data,tpl_graph_neo4j_cypher
assign('return', '{{n_result_data}}')

    label是：{{graph_content}}
    现在有图谱关系：{{n_level_relationship}}
    允许跳节点次数为：{{maxHops}}

    {{question}}

    MATCH path = (start:kb_cy_test_114_graph_content)-[rels:level_relationship|level_relationship_01*1..3]->(end:kb_cy_test_114_graph_content)

    	RETURN
          [node IN nodes(path) | { mainLabel: labels(node)[0],properties: properties(node)}] AS nodeProperties,
          [rel IN relationships(path) | {type: type(rel),properties: properties(rel)}] AS relProperties,
