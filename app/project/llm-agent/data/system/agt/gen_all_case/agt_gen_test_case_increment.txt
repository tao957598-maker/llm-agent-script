//# ========== Step 3: 模块级别对比 ==========
n_diff_result, act_func?script='''
	def normalize = { s ->
		def x = (s == null ? "" : s.toString()).toLowerCase()
		x = x.replaceAll(/[\\[\\]【】()]/, " ")
		x = x.replaceAll(/\\s*-\\s*/, " ")
		x = x.replaceAll(/\\s+/, " ").trim()
		return x
	}
	
	def matchKey = { mod ->
		def main = normalize(mod.main_module_name)
		def sub = normalize(mod.submodule_name)
		return "${main}||${sub}"
	}
	
	def isModified = { oldMod, newMod ->
		if (!oldMod || !newMod) return true
		if (normalize(oldMod.submodule_description) != normalize(newMod.submodule_description)) return true
		if (normalize(oldMod.main_module_name) != normalize(newMod.main_module_name)) return true
		return false
	}
	
	def oldList = fn.get("old_module_list") ?: []
	def newList = fn.get("new_module_list") ?: []
	
	def oldMap = new LinkedHashMap()
	oldList.each { r ->
		if (r instanceof Map) {
			def k = matchKey(r)
			oldMap.put(k, r)
		}
	}
	
	def diff = [
		unchanged: [],
		modified: [],
		new: [],
		deleted: []
	]
	
	newList.each { newMod ->
		if (!(newMod instanceof Map)) return
		def k = matchKey(newMod)
		if (oldMap.containsKey(k)) {
			def oldMod = oldMap.get(k)
			if (isModified(oldMod, newMod)) {
				diff.modified.add(newMod)
			} else {
				diff.unchanged.add(oldMod)  // 保留旧记录用于后续复用
			}
			oldMap.remove(k)
		} else {
			diff.new.add(newMod)
		}
	}
	
	// 剩余的旧模块就是被删除的
	oldMap.each { k, oldMod ->
		diff.deleted.add(oldMod)
	}
	
	fn.put("diff_result", diff)
	return diff
'''
assign("n_diff_result", "{{n_diff_result}}")

	]
	return stats
'''
