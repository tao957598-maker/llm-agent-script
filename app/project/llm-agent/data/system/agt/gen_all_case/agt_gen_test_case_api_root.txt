    def toCamel = { String s ->
        if (s == null || s.length() == 0) return s
        def p = s.split("_")
        def sb = new StringBuilder(p[0])
        for (int i = 1; i < p.length; i++) {
            if (p[i].length() > 0) sb.append(p[i].substring(0, 1).toUpperCase()).append(p[i].substring(1))
        }
        return sb.toString()
    }
    def wrapKeys = ["value", "result", "data", "output", "outputs", "results", "agentResult", "returnValue", "payload", "planResult", "response", "body", "extra", "context", "bindings", "ret"]
    def deepPick
    deepPick = { obj, String key, int depth ->
        if (depth > 12 || obj == null) return null
        def keysTry = [key, toCamel(key)]
        if (obj instanceof Map) {
            def m = obj as Map
            for (def kk : keysTry) {
                if (kk != null && m.containsKey(kk)) return m.get(kk)
            }
            for (def w : wrapKeys) {
                if (!m.containsKey(w)) continue
                def inner = deepPick(m.get(w), key, depth + 1)
                if (inner != null) return inner
            }
            for (def e : m.entrySet()) {
                def inner = deepPick(e.value, key, depth + 1)
                if (inner != null) return inner
            }
        }
        if (obj instanceof List) {
            for (def it : (obj as List)) {
                def inner = deepPick(it, key, depth + 1)
                if (inner != null) return inner
            }
        }
        return null
    }
    def shallowPick = { raw, String k ->
        try {
            if (raw instanceof Map) {
                if (raw.containsKey(k)) return raw.get(k)
                def ck = toCamel(k)
                if (raw.containsKey(ck)) return raw.get(ck)
                for (def alt : wrapKeys) {
                    def inner = raw.get(alt)
                    if (inner instanceof Map) {
                        if (inner.containsKey(k)) return inner.get(k)
                        if (inner.containsKey(ck)) return inner.get(ck)
                    }
                }
            }
            if (raw != null && !(raw instanceof Map)) {
                try { return raw."${k}" } catch (Throwable ignore0) {}
                try {
                    def cap = k.length() > 0 ? k.substring(0, 1).toUpperCase() + k.substring(1) : k
                    def m = raw.getClass().getMethod("get" + cap)
                    return m.invoke(raw)
                } catch (Throwable ignore1) {}
            }
        } catch (Throwable ignore2) {}
        return null
    }
    def r = n_api_result
    def idx = [status: 0, n_combined_result: 1, n_doc_mind_map: 2]
    def pickOne = { String k ->
        def v = shallowPick(r, k)
        if (v != null) return v
        v = deepPick(r, k, 0)
        if (v != null) return v
        if (r instanceof List && idx.containsKey(k)) {
            def L = r as List
            int i = idx.get(k)
            if (L.size() > i) return L.get(i)
        }
        if (r instanceof Map) {
            def innerList = r.get("resultPlanResults")
            if (innerList == null) innerList = r.get("planResults")
            if (innerList instanceof List && idx.containsKey(k)) {
                def L = innerList as List
                int i = idx.get(k)
                if (L.size() > i) return L.get(i)
            }
        }
        return null
    }
    def s = { x -> x == null ? "" : x.toString() }
    fn.put("api_root_mmd_str", s(pickOne("n_doc_mind_map")))
    return ""
'''
