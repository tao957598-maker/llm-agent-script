// 处理返回结果
when('{{fn.hasValue(n_openapi_result)}}', {
	assign("openapi_result", "OpenAPI规范生成成功")
	log("OpenAPI规范已生成：{{fn.toString(n_openapi_result)}}")
}, {
	assign("openapi_result", "OpenAPI规范生成失败")
	log("生成OpenAPI规范时出现错误")
})

- 版本号：{{version}}
