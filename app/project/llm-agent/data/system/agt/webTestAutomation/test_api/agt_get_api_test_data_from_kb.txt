 	n_info, act_kb.selectBy?sql='SELECT name, method, description, function_data FROM kb_api_test_data WHERE version="{{n_version}}"'

		def data = n_info

	'''

{
	// Get  all api
	assign("n_all_api", "{{api}}")
	n_apis, act_func?script='''
		def inputStr = "{{n_all_api}}"
		def parts = inputStr.split(",")
		return parts.collect { it.trim() }.findAll { it } 
	'''

	n_str_apis,act_func?js='''
	(function() {
		let str_apis = JSON.stringify({{n_apis}})
		const res = '(' + JSON.parse(str_apis).map(item => `"${item}"`).join(',') + ')';
		return res
	})();
	'''

	// Get api data through version and api name
 	n_info, act_kb.selectBy?sql='SELECT name, method, description, function_data FROM kb_api_test_data WHERE version="{{n_version}}" AND name IN {{n_str_apis}}'

})

// Get hash suffix 
n_suffix,act_func?js='''
	(function() {
		let str = '{{n_all_api}}'
		let hash = 0;
		for (let i = 0; i < str.length; i++) {
			const char = str.charCodeAt(i);
			hash = ((hash << 5) - hash) + char;
			hash = hash & hash;
		}
		return Math.abs(hash).toString(16);
	})();
'''

// datas
n_datas,act_func?js='''
	(function() {
		return JSON.stringify({{n_info}});
	})();
'''
