assign("system", "{{platform}}")
assign("module", "{{main_module_name}}")

// Use selenium
assign('explore_mode_data', '{{fn.optional(explore_mode, 1)}}')

when('fn.hasValue(restart_browser)&restart_browser=1', {
	//# Send product mode information to RPA
	act_rpa.send?command=SwitchExploreMode&name={{fn.json('explore_mode,explore_mode_data,test_system,system,test_no,test_no')}}
},{
	assign('msg', 'Not restart Browser')
})


when('fn.hasValue(dependent_test_no)', {
	// Send the dependent test number
	n_rpa_test, act_rpa.send?command=dependOn&parameters='{{fn.json("test_no,dependent_test_no")}}'
	
	// Result of dependent
  	on_failure({
		act_flow.go?plan=end
	})
	
})
	
when('!fn.hasValue(module)', {
	assign("n_module","def_module")
},
{
	assign("n_module","{{module}}")
})

when('!fn.hasValue(draft_tests)', {
	assign("n_draft_tests","draft_steps")
},
{
	assign("n_draft_tests","{{draft_tests}}")
})

when('!fn.hasValue(programming_language)', {
	assign("n_programming_language","selenium")
},
{
	assign("n_programming_language","{{programming_language}}")
})

when("n_programming_language=='selenium'", {
	assign("file_type",".py")
})
when("n_programming_language=='python'", {
	assign("file_type",".py")
})
when("n_programming_language=='java'", {
	assign("file_type",".java")
})
// script subfolder
assign("script_subfolder","/{{system}}/{{n_programming_language}}/workflows/{{n_module}}/{{n_draft_tests}}/{{test_no}}")
// script tmp subfolder
assign("script_subfolder_tmp","/{{system}}/{{n_programming_language}}/tmp/{{test_no}}")

// feature subfolder
assign("feature_subfolder_main_code","/{{system}}/features/{{n_module}}/{{test_no}}{{file_type}}")
assign("feature_subfolder_main_steps","/{{system}}/features/{{n_module}}/{{test_no}}_steps.txt")
// global variable
assign("global_variable","{{fn.json('system,system,programming_language,n_programming_language,module,n_module,draft_tests,n_draft_tests,test_no,test_no,script_subfolder,script_subfolder,script_subfolder_tmp,script_subfolder_tmp,file_type,file_type,feature_subfolder_main_code,feature_subfolder_main_code,feature_subfolder_main_steps,feature_subfolder_main_steps')}}")


act_memory.assign?var=global_variable&value={{global_variable}}&global=true

assign("page_code_info","[]")
act_memory.assign?var=global_variable_page_code_info&value={{page_code_info}}&global=true

// Call agent to generate test steps
agt_api_steps_gen?system={{system}}&test_no={{test_no}}&version={{version}}&steps={{steps}}&expected_result={{expected_result}}&force_update=false&step_supplement={{step_supplement}}
