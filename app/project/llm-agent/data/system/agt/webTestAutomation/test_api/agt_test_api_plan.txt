// Get hash suffix 
n_suffix,act_func?js='''
	(function() {
		let str = '{{n_all_api}}'
		let hash = 0;
		for (let i = 0; i < str.length; i++) {
			const char = str.charCodeAt(i);
			hash = ((hash << 5) - hash) + char;
			hash = hash & hash;
		}
		return Math.abs(hash).toString(16);
	})();
'''
