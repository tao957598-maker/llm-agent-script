assign("sql", """
SELECT 
    t1.current_uri, 
    CASE 
        WHEN t1.analysis_content IS NULL OR t1.analysis_content = '' THEN t1.text_data
        ELSE t1.analysis_content
    END as content
FROM llm_agent.kb_spiderdeep_content t1
INNER JOIN (
    SELECT current_uri, MAX(update_time) as max_update_time
    FROM llm_agent.kb_spiderdeep_content
    WHERE system_name = '{{system_name}}'
    GROUP BY current_uri
) t2 ON t1.current_uri = t2.current_uri 
    AND t1.update_time = t2.max_update_time
WHERE t1.system_name = '{{system_name}}';
""")
// 执行SQL查询
n_data,act_kb.selectBy?sql={{sql}}
