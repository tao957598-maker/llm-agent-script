when("operation==1", {
    // 切片向量化
    foreach('n_loop_id', 'sectionIds', {
        assign('sectionId', '{{n_loop_id.item}}')

        // 1.找到切片数据
        n_data,act_kb.select?name='kb_spaces_knowledge_section'&maxSize=1&where='{"id":{{n_loop_id.item}}}'

        n_update_data,act_func?script='''
                                        def data = n_data[0];
                                        data.put("id", n_loop_id.item+"")
                                        data.put("status", "1")
                                        data.put("update_time","now")
                                        data.put("page",data.page +"")
                                        return data
                                    '''
        // 2.将切片数据修改状态，并向量化
        act_kb.merge?name='kb_spaces_knowledge_section'&columnValues={{n_update_data}}&where='{"id":{{n_loop_id.item}}}'&addColumnsMilvus=true&embedding=true
    })

    assign('last_result', '{{sys_last_result}}')
})




when("operation==2",{
    // 新增切片
    // 1.找到对应的知识信息，获取相关参数
    n_data,act_kb.selectBy?sql='select * from kb_spaces_knowledge_section where document_id = ? order by id desc limit 1'&parameters='{{knowledgeId}}'
    n_import_data,act_func?script='''
                                            def data = n_data[0];
                                            data.remove("id")
                                            data.put("status", "0")
                                            data.put("content", content)
                                            data.put("update_time","now")
                                            data.put("order_by_num", (Integer.parseInt(String.valueOf(data.order_by_num)) + 1 ) + "")
                                            return data
                                        '''
    // 2.保存db数据信息
    act_kb.importData?name='kb_spaces_knowledge_section'&data={{n_import_data}}&embedding=false
    assign('last_result', '{{sys_last_result}}')
})

when("operation==4",{
    // 对话检索
    // 1.获取向量查询参数
    n_data,act_kbs.callQuestionKnowledges?callType={{callType}}&value={{value}}&knowledgeName={{knowledgeName}}

    n_has_value,act_func?script='''
        def list = n_data
        def size = list.size()
        return size
    '''

    when("n_has_value > 0",{
        n_filter_data,act_func?script='''
                def item = new HashMap()
                // logicType 逻辑类型，空字符串表示简单条件，and、or、not表示复杂逻辑
                item.put("logicType", "")
                item.put("filterValue", n_data)
                item.put("operator", "in")
                item.put("filterKey", "document_id")
                return item
            '''

        // 2.找到对应的知识向量信息
            n_result_data,act_kb.search?requirement='{{question}}'&name='kb_spaces_knowledge_section'&column='content'&top='10'&filterItem='{{n_filter_data}}'
            n_result_parse_data,act_func?script='''
                        def n_parse_data = n_result_data
                        for (int i = 0; i < n_parse_data.size(); i++) {
                            def map = n_parse_data[i]
                            map.put("id", map.id + "")
                        }
                        return n_parse_data
                    '''
            assign('last_result', '{{n_result_parse_data}}')
    })
})

when("operation==5",{
    // 对话检索-ink
    // 1.获取向量查询参数
    n_data,act_kbs.callQuestionKnowledges?callType={{callType}}&value={{value}}&knowledgeName={{knowledgeName}}

    n_has_value,act_func?script='''
        def list = n_data
        def size = list.size()
        return size
    '''

    when("n_has_value > 0",{
        n_api_data,act_api.doExecute?name='get_gientech_retrieval'&params='{"question":'{{question}}',"collection":"kb_spaces_knowledge_ink_section","document_id":{{n_data}}}'

        n_merge_data,act_kbs.removeProp?params='{{n_api_data.reference_details}}'&properties='emb_content'

        n_result_data,act_func?script='''
            def map = new HashMap()
            map.put("reference_contents", n_api_data.reference_contents)
            map.put("sections", n_merge_data)
            return map
        '''
        assign('last_result', '{{n_result_data}}')
    })
})

assign('return', '{{last_result}}')
