# -*- coding: utf-8 -*-
# run function dynamic parameters:
'''{"item":{"id": "登录按钮的ID", "name": "xxx"}, "verify_data": None}'''  # 由于是验证按钮存在，不校验按钮的文案，所以不需要校验verify_data期望值，故verify_data为None
from autotest import getXpathByItem, writeResult
from selenium.webdriver.common.by import By
from datetime import datetime


def run(browser, item={"id": "登录按钮的ID", "name": "Login"}, verify_data=None):
    try:
        # 示例验证逻辑
        login_xpath = getXpathByItem(browser, item)  # 使用getXpathByItem获取xpath
        if not login_xpath:
            writeResult(False, "Login button not found")
            return

        # Locate element using the obtained xpath
        login_element = browser.find_element(By.XPATH, login_xpath) # 使用xpath必须来自getXpathByItem获取的，禁止使用这种xpath: "./xxxxx"

        # Verify login element is displayed
        assert login_element.is_displayed(), "Login button element is not visible"
        writeResult(True, "Login button verification passed")

    except Exception as e:
        writeResult(False, f"Element verification failed: {str(e)}")

from autotest import getXpathByItem, writeResult
from selenium.webdriver.common.by import By


def run(browser, item={"id": "输入框的ID", "name": "username"}, verify_data="admin"):
    try:
        # 示例验证逻辑

            writeResult(False, "username not found")
            return False

        username_element = browser.find_element(By.XPATH, username_xpath)
        username_text = username_element.text
        
        assert username_text == verify_data, f"username verification failed. Expected: {verify_data}, Actual: {username_text}"
		writeResult(True, "Element verification passed")

    except Exception as e:
        writeResult(False, f"Element verification failed: {str(e)}")
