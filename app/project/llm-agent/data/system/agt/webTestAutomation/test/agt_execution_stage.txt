def data = steps
def executorToAgent = [
    "VERIFICATION_TASK": "agt_async_verify.webTestAutomation.test",
    "VERIFICATION_TRIPLE_TASK": "agt_async_verify.webTestAutomation.test",
    "VERIFICATION_EXECUTE_TASK": "agt_async_verify.webTestAutomation.test",

    "TRIPLE_TASK": "agt_test_triple.webTestAutomation.test",
    "OPERATION_EXECUTE_TASK": "agt_test_selenium.webTestAutomation.test",
    "OPERATION_EXIST_TASK": "agt_test_selenium.webTestAutomation.test"
]

def result = []
int idx = 0

while (idx < data.size()) {
    def current = data[idx]
    def currentExecutor = current.executor
    
    boolean isVerificationNotTask = currentExecutor.startsWith("VERIFICATION_") && currentExecutor != "VERIFICATION_TASK"
    
    if (isVerificationNotTask && idx + 1 < data.size()) {
        def nextItem = data[idx + 1]
        
        current.agent = executorToAgent[currentExecutor]
        nextItem.agent = executorToAgent[nextItem.executor]
        
        result.add([current, nextItem])
        idx += 2
    } else {
        current.agent = executorToAgent[currentExecutor]
        result.add([current])
        idx += 1
    }
}

return result
'''
