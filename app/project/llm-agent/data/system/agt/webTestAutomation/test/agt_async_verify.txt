when('VerifyExist',{

when('step_type=="Verify"', {
    // 需要从原来执行那里给参数过来，领导说的是存储到数据库，再查出来
    // Try to run existed code
    agt_test_run_exist_code?test_no={{test_no}}&current_step={{current_step}}&steps={{steps}}&system={{system}}&envinfo={{envinfo}}

    // Run code successfully, go to end
    act_flow.go?plan=end

    // Handle the error
    on_failure({
        // Force to success status
        act_flow.success
    })

    // Find and check target element
    agt_test_findelement?test_no={{test_no}}&current_step={{current_step}}&steps={{steps}}&envinfo={{envinfo}}
    // Run Find and check target element successfully, go to end
    act_flow.go?plan=end
})

when('step_type=="VerifyTriple"', {
	//# 从数据库查询用于验证的数据
	n_verify_data, act_kb.selectBy?sql='select * from kb_test_case_instance_intermediates where test_step_id="{{step_id}}" '

    // 需要从原来执行那里给参数过来，领导说的是存储到数据库，再查出来

    n_cmp1, tpl_cmp_pageresult_Triple, vrf_flag?value='fn.str(sys_last_result.thought.re_reasoning)=="false" || sys_last_result.hasDialog',\
    agt_save_explorer?columns='page_change,tpl_cmp_pageresult.thought.change,reasoning,tpl_cmp_pageresult.thought.reasoning'
    assign('python_code', '{{fn.readFile(n_similar_file.filename)}}')
    // Check if page doesnot change
    when('fn.not(n_cmp1.run_success)', {
        // 基于现有代码生成失败，直接返回，让后续步骤生成新代码
        // Anounce no valid code
        act_flow.fail?result='No valid code'&message=true
    }, {
        // When the page changes and no dialog appears
        when('fn.hasValue(n_cmp1.change_element) && n_cmp1.change_behavior=="add" && !fn.bool(n_cmp1.hasDialog)', {
            // save tmp
            n_python_file_2, act_python.writeTmp?path={{n_global_variable.script_subfolder_tmp}}&requirement={{current_step}}&sourceCode={{python_code}}&parameters={{n_similar_file.parameter}}
            assign('python_file', '{{n_python_file_2.kbspath}}')
            assign('python_code', '{{n_python_file_2.sourceCode}}')
            // Add page verify element
            n_python_code,tpl_self_add_wait, psr_python, act_memory.assign?var=python_code
            // return code
            assign('python_code', '{{n_python_code}}')
        })
    })
})

when('step_type=="VerifyExecute"', {
	//# 从数据库查询用于验证的数据
	n_verify_data, act_kb.selectBy?sql='select * from kb_test_case_instance_intermediates where test_step_id="{{step_id}}" '

    when('n_result.action_type=="3"', {
        //# Check and compare the execution result
        n_cmp1, tpl_cmp_pageresult_Execute, vrf_flag?value='fn.str(sys_last_result.thought.re_reasoning)=="false" || sys_last_result.hasDialog',\
                agt_save_explorer?columns='page_change,tpl_cmp_pageresult.thought.change,reasoning,tpl_cmp_pageresult.thought.reasoning'
    },
    {   
        assign('action_result', '{{n_result.action_result}}')
        
        vrf_flag?value='fn.str(action_result.thought.re_reasoning)=="false" || action_result.hasDialog',\
            agt_save_explorer?columns='page_change,action_result.thought.change,reasoning,action_result.thought.reasoning'
        assign('n_cmp1', '{{n_result.action_result}}')
    })

	// Check if page doesnot change
	when('fn.not(n_cmp1.run_success)', {
		// Show warning information
		// log('Failed to verify page result', true)
		
		// Proceed to error handling process
		act_flow.go?plan=n_fail_rpa_coll
	}, {
		// n_cmp1n_cmp1n_cmp1
		assign('n_cmp1n_cmp1', '{{n_cmp1}}')
		// tpl_task_element_infotpl_task_element_info
		assign('tpl_task_element_infotpl_task_element_info', '{{tpl_task_element_info}}')
		// 无论n_cmp1的新界面还是tpl_task_element_info的原有界面都没有对话框时 <== 对话框的开关，通常不会有基础元素的变化
		// When the page changes and no dialog appears
		when('fn.hasValue(n_cmp1.change_element) && n_cmp1.change_behavior=="add" && !fn.bool(n_cmp1.hasDialog) && !fn.bool(tpl_task_element_info.hasDialog)', {
			// Add page verify element
			tpl_self_add_wait, psr_python, act_memory.assign?var=python_code
			assign('python_code', "{{sys_last_result}}")
			n_python_file_3, act_python.writeTmp?path={{n_global_variable.script_subfolder_tmp}}&requirement={{current_step}}&sourceCode={{sys_last_result}}
			assign('python_code', '{{n_python_file_3.sourceCode}}')
			assign('python_file', '{{n_python_file_3.kbspath}}')
		})
		when('fn.hasValue(n_cmp1.change_element) && n_cmp1.change_behavior=="remove" && !fn.bool(n_cmp1.hasDialog) && !fn.bool(tpl_task_element_info.hasDialog)', {
			// Remove page verify element
			tpl_self_remove_wait, psr_python, act_memory.assign?var=python_code
			assign('python_code', "{{sys_last_result}}")
			n_python_file_4, act_python.writeTmp?path={{n_global_variable.script_subfolder_tmp}}&requirement={{current_step}}&sourceCode={{sys_last_result}}
			assign('python_code', '{{n_python_file_4.sourceCode}}')
			assign('python_file', '{{n_python_file_4.kbspath}}')
		})
	})

})
