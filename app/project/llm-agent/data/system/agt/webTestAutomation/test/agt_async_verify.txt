// 异常处理
// Try to fix script error
on_failure({
	//# 构建StepTestInfo
	n_tmp_info,act_func?script='''
	return [
		current_step: current_step ?: "",
		steps: steps ?: [],
		tpl_task_element_info: n_tpl_task_element_info ?: [:],
		simplepageinfo: n_verify_data.dom_before ?: "",
		new_collect_page: n_verify_data.dom_after ?: ""
	]
	'''
	// Collect page information when failure
	n_fail_rpa_coll, act_rpa.send?command=CollectPageInfo

	// Analysis page information
	rag_func?name=StepTestInfo, tpl_task_page_info, agt_save_explorer?columns='page,tpl_task_page_info.pageName,url,tpl_task_page_info.pageUrl,page_type,tpl_task_page_info.pageType,tip_message,tpl_task_page_info.tipMessage!'

	// Check for multiple failures
	when('sys_run_count>3', {
		// Multiple failures, announce task failure
		act_flow.fail?result={{fn.json("failure_count, sys_run_count, error_info, sys_hint_agt")}}
	})
	
	// Try to restore original page
	when('fn.bool(tpl_task_page_info.hasDialog) && !fn.bool(tpl_task_element_info.hasDialog)', {
		// Click the cancel button
		act_rpa.send?command=RunDynamicScript&name='clickElementByItem(browser, "{{tpl_task_page_info.cancelElementId}}")'
	})

	// Try to fix the script
	when('fn.hasValue(python_code)', {
		// Collect page elements
		tpl_task_element_info
	
		// Fix the script
		n_fix_python_code,tpl_self_fix,psr_json_python
		assign('python_code', "{{n_fix_python_code}}")
		act_flow.go?plan={{fn.optional(plan, 'n_python_file_2')}}&step=act
	}, {
		// No python code, announce task failure 
		act_flow.fail?result={{fn.json("failure_count, sys_run_count, error_info, sys_hint_agt")}}
	})
})
