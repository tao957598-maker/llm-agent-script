	assign('n_directory_id', "{{n_content.directory_id}}")
	assign('n_collect_data', "{{n_content.collect_data}}")

	// 查询库内directory_id相同的数据作对比
	n_content_list,act_kb.selectBy?sql="SELECT content_id, directory_id, collect_data, content_version FROM kb_spiderdeep_content WHERE directory_id = '{{n_directory_id}}'"

	// 计算是否重复
	n_same_content, act_func?script='''
		// 当前页面的 id 集合
		def currentIds = n_collect_data
				.findAll { it.containsKey('id') }
				.collect { it.id }
				.toSet()
		def maxVersion = 0
		def res = [:]
		for (item in n_content_list) {
			def libData = item.collect_data
			def version = item.content_version ?: 0
			maxVersion = Math.max(maxVersion, version)
			def libIds_str = fn.str(libData,"id")
			// .toSet() 会先去重再排序
            def libIds_list = fn.list(libIds_str).findAll { it != null }.toSet()

			// id 完全一致（忽略顺序）
			if (currentIds == libIds_list) {
				res.have_content = true
				res.directory_id = item.directory_id
				res.content_id = item.content_id
				return res
			}
		}

		// 都不匹配，返回最大版本号 + 1
		def newVersion = maxVersion + 1
		def contentId = "${n_directory_id}${newVersion}".md5()

		res.have_content = false
		res.max_version = newVersion
		res.directory_id = n_directory_id
		res.content_id = contentId
		return res
	'''

	when("n_same_content.have_content != true", {
		// 本内容为首次进入，需要入内容表
  		n_save_content, act_func?script='''
  			def res = [:]
     		res = n_content
			res.content_id = n_same_content.content_id
			res.content_version = n_same_content.max_version
  			return res
		'''
		// first time, save content
		act_kb.defineAndImportData?tableName=kb_spiderdeep_content&data='{{n_save_content}}'
	})

	// 开始更新任务表，判断执行后biz_id是否存在
	n_old_same_data,act_kb.selectBy?sql='SELECT * FROM kb_spiderdeep_directory WHERE biz_id = "{{n_old_data.biz_id}}"'
	when('{{(n_old_same_data![])?size > 0}}', {
		// 找到任务表并更新任务表状态
		act_kb.selectBy?sql='UPDATE kb_spiderdeep_directory SET content_id="{{n_same_content.content_id}}", run_success = {{n_old_data.run_success}},is_return_able = {{n_old_data.is_return_able}}, directory_id = "{{n_directory_id}}", target_uri = "{{n_old_data.target_uri}}" WHERE id = "{{n_parameters.id}}"'
	},{
		// 未找到，直接新增
		n_save_old_data, act_func?script='''
			def res = [:]
			res = n_old_data
			res.content_id = n_same_content.content_id
			return res
		'''
		when('!fn.hasValue(n_parameters.id)', {
			// 首次执行没有id，直接入库
			act_kb.defineAndImportData?tableName=kb_spiderdeep_directory&data='{{n_save_old_data}}'
		}, {
			// 非首次运行，根据id更新任务
			act_kb.update?name='kb_spiderdeep_directory'&columnValues='{{n_save_old_data}}'&where='{"id":"{{n_parameters.id}}"}'
		})
	})
})
