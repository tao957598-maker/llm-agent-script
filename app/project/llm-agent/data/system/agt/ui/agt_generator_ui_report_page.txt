// 1.检查现有页面列表能否满足需求
rag_api?name=ui_dev_search&params='{"pageType":{{pageType}}}', tpl_ui_select
when("tpl_ui_select.pageName=='NEW_PAGE' && fn.hasValue(tpl_ui_select.template)", {
	// 2.生成新页面属性信息，并保存到db
	tpl_ui_generator, act_api?name=ui_dev_save&params={{sys_last_result}}
	assign('n_last_result_data', '{"result":"success","pageName":"{{tpl_ui_generator.pageName}}","msg":"Generation completed"}')
}, {
    when("fn.hasValue(tpl_ui_select.pageName) && fn.hasValue(tpl_ui_select.template)", {
        // 2.页面已存在
        assign('n_last_result_data', '{"result":"fail","pageName":"{{tpl_ui_select.pageName}}","msg":"Page already exists"}')
    })

    when("tpl_ui_select.success == 'fail'", {
        // 2.异常，无法识别用户输入的信息
        assign('n_last_result_data', '{"result":"fail","msg":"{{tpl_ui_select.remark}}"')
    })
});

assign('return', '{{n_last_result_data}}')

define('tpl_ui_select', '''
	你是一名前端程序开发人员，基于用户的描述，对比现有的页面名称及描述信息，按严格匹配，找出哪个页面能符合用户需求，返回该页面名称及页面类型（如果没有页面，则：pageName:'NEW_PAGE'）。返回结果要求用json格式

    用户需求:
    {{requirement}}

	现有的页面列表：(注：页面列表中的pageName为页面名称)
	{{ui_dev_search.data}}

	页面模板类型为template，取值为数字1-3，含义如下：
    1： 柱形图表
    2： 饼形图表
    3： 折线图表

	返回结果示例：
	{
		pageName: 'NEW_PAGE'
		template: '1'
	}

	异常信息处理：
    如果无法识别用户的需求，那么返回示例:
    {
        success: 'fail',
        msg: 'Unable to recognize input information',
        remark: 'xxxxx为什么无法识别'
    }

    提示：pageCode:不允许重复

''')
