def decision = "INSERT_ONLY"
def upper = fn.str(n_llm_decision,"decision")
if (upper.contains("MERGE")) {
    decision = "MERGE"
} else if (upper.contains("SOFT_DELETE")) {
    decision = "SOFT_DELETE_INSERT"
} else if (upper.contains("INSERT_ONLY") || upper.contains("INSERT")) {
    decision = "INSERT_ONLY"
} else if (upper.contains("SKIP")) {
    decision = "SKIP"
}

'''
assign('return', '{{n_decision}}')

// 构建决策提示
define('tpl_llm_decision', '''
    你是一个Hint记录合并决策专家。请根据以下信息判断应该执行哪种合并策略。

    #新记录信息
    - activation_rules: {{newActivationRules}}
    - solution: {{newSolution}}
    - type: {{newType}}
    - positive: {{newPositive}}

    #现有相似记录
    {{n_history_list}}

    #决策选项
    1. MERGE - 合并到现有记录：当新记录与现有记录的activation_rules、solution、type高度相似时选择
    2. SOFT_DELETE_INSERT - 软删除现有记录并新增：当新记录与现有记录相似但有显著差异时选择
    3. INSERT_ONLY - 仅新增记录：当没有查到相似记录或相似度很低时选择
    4. SKIP - 跳过处理：当新记录是重复的或无价值时选择

    #判断规则
    - 如果相似度score >= 0.8 且内容高度相似，应该选择 MERGE
    - 如果相似但有显著差异，应该选择 SOFT_DELETE_INSERT
    - 如果没有相似记录或相似度很低，应该选择 INSERT_ONLY
    - 如果是重复记录或无价值记录，应该选择 SKIP

    #输出格式
    {"decision": "MERGE(决策选项的某一种）", "reason": "原因描述"}

''')
