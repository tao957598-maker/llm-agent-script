assign("python_file_name", "python/{{n_python_file.filename}}")
// Run python file return file
n_res_file_list, act_api.doExecute?name="run_py_return_file"&params='{"instance_name": "{{instance_name}}"}'&fileNames='["{{python_file_name}}"]'

// foreach csv file list
assign("res_file_list", "{{n_res_file_list.file_list}}")
foreach('this_file', 'res_file_list', {
  assign("this_file_item", "{{this_file.item}}")
  assign("file_name", "{{this_file_item.file_name}}")
  assign("file_content", "{{this_file_item.file_content}}")

  assign("file_path", "python/{{python_folder}}/{{file_name}}")
  // write to csv
  act_func.write?filename="{{file_path}}"&data="{{file_content}}"

  // save to mysql
  n_save_csv_res, act_api.doExecute?name="mysql_save_csv"&params='{"mysql_config": {{mysqlConfig}}, "table_name": "{{table_name}}"}'&fileNames='["{{file_path}}"]'
})


assign("result", 'Agent task ended')

    输入指令：
    ```

    ```
    输出数据：
    ```json
    {"sql": "SHOW CREATE TABLE `login_user`;"}
    ```
  - 例子2：
    获取login_user表中全部数据
    ```
    {"sql": "SELECT * FROM `login_user`;"}
    ```

# Input
需要执行的指令: {{sql_cmd}}
''')
