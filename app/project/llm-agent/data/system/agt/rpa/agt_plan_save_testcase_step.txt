n_load_target_data, act_kb.select?name='kb_testcase_step'&where='{"test_no":"{{test_no}}","application":"{{system}}","version":"{{version}}"}'&column='application,test_no,step,result,version'

// Save test step
act_kb.delete?name=kb_testcase_step&where='{"test_no":"{{test_no}}","application":"{{system}}","version":"{{version}}"}'&maxSize=100\
	act_kb.convert?data={{steps}}&converter=mergeList&storeColumn='application,test_no,step,result,executor,version'&parameters='{"uniqueKey":"step","targetData": {{n_load_target_data}},"test_no":"{{test_no}}","application":"{{system}}","version":"{{version}}"}'\
	act_kb.importData?name=kb_testcase_step&data={{sys_last_result}}&uniqueKey='application,test_no,step,version'
