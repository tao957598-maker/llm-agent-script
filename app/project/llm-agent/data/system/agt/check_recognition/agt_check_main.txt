// 将 n_file_name 中每条 store_uri 转为路径列表，并取出 image_marked 对应路径
n_image_path_list, act_func.run?script='''
def rows = fn.list(fn.get("n_file_name"))
def paths = new ArrayList()
def marked = ""
rows.each { row ->
  def m = fn.map(row)
  def uri = fn.str(m, "store_uri")
  paths.add(uri)
  
}
return paths
'''

n_image_marked_uri,act_func.run?script='''
def rows = fn.list(fn.get("n_file_name"))
def result = null

rows.each { row ->
  def m = fn.map(row)
  def uri = fn.str(m, "store_uri")
  
  if (uri.contains("_marked")) {
    result = uri   
    return   
  }
}

return result   
'''

// 获取原始支票图片路径
n_original_image,act_func.run?script='''
def rows = fn.list(fn.get("n_file_name"))
def original = null

rows.each { row ->
  def m = fn.map(row)
  def uri = fn.str(m, "store_uri")
  
  if (!uri.contains("_marked") && !uri.contains("image_")) {
	original = uri   
	return   
  }
}

return original   
'''
