// 判断图片来源
when('fn.hasValue(kbSpaceFileId)', {
	// documentId
	assign("documentId", "{{kbSpaceFileId}}")
	// 查询文档所有信息
	n_file_infos, act_kb.selectBy?sql='select id as document_id,knowledge_name as document_name,extension as document_type,store_uri as source_path from kb_spaces_folder_knowledge where id="{{documentId}}" '
	// image path
	assign("check_image_path", "{{n_file_infos[0].source_path}}")
},{
	// image path
	assign("check_image_path", "{{image_path}}")
})

// 定义系统提示词
assign("system_prompt",'''
请仔细识别支票图片中的所有内容信息，并按照以下JSON格式返回结构化数据：

{
  "date": "日期（年/月/日格式，如2026/04/16）",
  "payee": "收款人名称",
  "amount_figures": "小写金额（数字格式，如12345.00）",
  "amount_words": "大写金额（如壹万贰仟叁佰肆拾伍元陆角柒分）",
  "currency": "币种",
  "memo": "备注/用途",
  "signature_present": true/false,
  "signature_clarity": 1-10的清晰度评分,
  "bank_name": "开户银行名称（如果可见）",
  "account_number": "账号（如果可见）"
  "credibility": "可信度"
}

识别规则：
1. 日期：识别年月日，按yyyy/mm/dd格式
2. 收款人：识别支票上的收款人名称
3. 小写金额：准确识别数字金额，保留两位小数
4. 大写金额：识别中文大写金额
5. 币种：识别币种
5. 备注：识别支票上的用途或备注信息
6. 签名：判断是否存在签名，以及清晰度评分
7. 可信度： 判断支票的真实有效性给出答复

请优先识别图片中的文字内容，并包含可信度，如果某个字段无法识别则返回null。
输出markdown格式数据，内容是JSON数据，不要解释。
''')

assign("user_prompt",'''
请提取支票内容
''')

// 调用LLM获取json结果
tpl_extract_content_recognition,psr_all

// 输出结果
assign("result", "{{tpl_extract_content_recognition}}")

// define定义对话信息
define('tpl_extract_content_recognition', '''
/*{
   "llmType":"alias.vision"
}*/
  {{user_prompt}}
	{{fn.chatImageFlag('%{system_prompt}%', 'user_prompt', '', 'check_image_path', 1)}}
''')
