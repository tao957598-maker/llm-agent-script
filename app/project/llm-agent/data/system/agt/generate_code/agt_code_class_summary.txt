when("!fn.hasValue(dbSchema)", {
    assign('dbSchema', 'gen_code')
})

when("fn.hasValue(fileId)", {
    assign('n_where_data', 'id = {{fileId}} and ')
})

// 1.查询没有注释或者总结的文件，如果fileId不为空，那么精确查询
n_class_data,act_kb.selectBy?sql='select * from {{dbSchema}}.code_gen_file_structure where {{n_where_data}} (file_summary is null or file_summary = '') order by id desc limit 1000'

foreach('n_loop_data', 'n_class_data', {
    assign('n_data', '{{n_loop_data.item}}')
    // 2.生成Class的提示信息
    n_prompt_data,act_code.generateClassPrompt?fileId={{n_data.id}}

    // 3.通过gpt，生成注释信息
    n_result_data,tpl_code_class_generate_summary_agent

    // 4.存储注释信息
    act_kb.merge?name='code_gen_file_structure'&columnValues='{"file_summary":"{{n_result_data.fileSummary}}"}'&where='{"id":"{{n_data.id}}"}'&embedding=true

    // 5.修改java原始代码，将注释信息写入
    n_modify_data,act_code.modifyClassDocComment?classPath='{{n_data.file_path}}'&className='{{n_data.class_name}}'&type='class'&docComment="{{n_result_data.fileSummary}}"
})


define('tpl_code_class_generate_summary_agent', '''
    你是一个高级JAVA代码开发工程师、需求分析师，请根据以下JSON定义说明，根据对应文件属性、函数定义、属性定义分析出该class的作用,精简回答
    参数属性说明如下：
        className:类名
        classFields:类属性列表
            filedName:属性名称
            fieldSummary:属性注释或总结
        fieldSummary:属性注释或总结
        classFunctions:类函数列表
            functionName:函数名称
            functionSummary:函数注释或总结
            body:函数代码内容
        fileType:文件类型：1.controller、2.service、3.interface、4.abstract、5.model、6.dto、7.util、8.未知类型
        classType:java类：1.普通类 2.内部类-成员内部类 3.内部类-静态内部类

    分析代码如下：

    {{n_prompt_data}}

    分析结果严格按json格式返回,格式如:
    {
        "id":"10001",
        "className":"ClassNameXX",
        "fileSummary":"获取工具总结"
    }
''')
